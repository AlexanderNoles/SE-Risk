157 bool CheckAnimStateChanged(out int stateHash, out float normalizedTime, int layerId)
158 {
159 bool change = false;
160 stateHash = 0;
161 normalizedTime = 0;
162
163 float lw = animator.GetLayerWeight(layerId);
164 if (Mathf.Abs(lw - layerWeight[layerId]) > 0.001f)
165 {
166 layerWeight[layerId] = lw;
167 change = true;
168 }
169
170 if (animator.IsInTransition(layerId))
171 {
172 AnimatorTransitionInfo tt = animator.GetAnimatorTransitionInfo(layerId);
173 if (tt.fullPathHash != transitionHash[layerId])
174 {
175 // first time in this transition
176 transitionHash[layerId] = tt.fullPathHash;
177 animationHash[layerId] = 0;
178 return true;
179 }
180 return change;
181 }
182
183 AnimatorStateInfo st = animator.GetCurrentAnimatorStateInfo(layerId);
184 if (st.fullPathHash != animationHash[layerId])
185 {
186 // first time in this animation state
187 if (animationHash[layerId] != 0)
188 {
189 // came from another animation directly - from Play()
190 stateHash = st.fullPathHash;
191 normalizedTime = st.normalizedTime;
192 }
193 transitionHash[layerId] = 0;
194 animationHash[layerId] = st.fullPathHash;
195 return true;
196 }
197 return change;
198 }

277 ulong NextDirtyBits()
278 {
279 ulong dirtyBits = 0;
280 for (int i = 0; i < parameters.Length; i++)
281 {
282 AnimatorControllerParameter par = parameters[i];
283 bool changed = false;
284 if (par.type == AnimatorControllerParameterType.Int)
285 {
286 int newIntValue = animator.GetInteger(par.nameHash);
287 changed = newIntValue != lastIntParameters[i];
288 if (changed)
289 lastIntParameters[i] = newIntValue;
290 }
291 else if (par.type == AnimatorControllerParameterType.Float)
292 {
293 float newFloatValue = animator.GetFloat(par.nameHash);
294 changed = Mathf.Abs(newFloatValue - lastFloatParameters[i]) > 0.001f;
295 // only set lastValue if it was changed, otherwise value could slowly drift within the 0.001f limit each frame
296 if (changed)
297 lastFloatParameters[i] = newFloatValue;
298 }
299 else if (par.type == AnimatorControllerParameterType.Bool)
300 {
301 bool newBoolValue = animator.GetBool(par.nameHash);
302 changed = newBoolValue != lastBoolParameters[i];
303 if (changed)
304 lastBoolParameters[i] = newBoolValue;
305 }
306 if (changed)
307 {
308 dirtyBits |= 1ul << i;
309 }
310 }
311 return dirtyBits;
312 }

314 bool WriteParameters(NetworkWriter writer, bool forceAll = false)
315 {
316 // fix: https://github.com/MirrorNetworking/Mirror/issues/2852
317 // serialize parameterCount to be 100% sure we deserialize correct amount of bytes.
318 // (255 parameters should be enough for everyone, write it as byte)
319 byte parameterCount = (byte)parameters.Length;
320 writer.WriteByte(parameterCount);
321
322 ulong dirtyBits = forceAll ? (~0ul) : NextDirtyBits();
323 writer.WriteULong(dirtyBits);
324
325 // iterate on byte count. if it's >256, it won't break
326 // serialization - just not serialize excess layers.
327 for (int i = 0; i < parameterCount; i++)
328 {
329 if ((dirtyBits & (1ul << i)) == 0)
330 continue;
331
332 AnimatorControllerParameter par = parameters[i];
333 if (par.type == AnimatorControllerParameterType.Int)
334 {
335 int newIntValue = animator.GetInteger(par.nameHash);
336 writer.WriteInt(newIntValue);
337 }
338 else if (par.type == AnimatorControllerParameterType.Float)
339 {
340 float newFloatValue = animator.GetFloat(par.nameHash);
341 writer.WriteFloat(newFloatValue);
342 }
343 else if (par.type == AnimatorControllerParameterType.Bool)
344 {
345 bool newBoolValue = animator.GetBool(par.nameHash);
346 writer.WriteBool(newBoolValue);
347 }
348 }
349 return dirtyBits != 0;
350 }

352 void ReadParameters(NetworkReader reader)
353 {
354 // fix: https://github.com/MirrorNetworking/Mirror/issues/2852
355 // serialize parameterCount to be 100% sure we deserialize correct amount of bytes.
356 // mismatch shows error to make this super easy to debug.
357 byte parameterCount = reader.ReadByte();
358 if (parameterCount != parameters.Length)
359 {
360 Debug.LogError($"NetworkAnimator: serialized parameter count={parameterCount} does not match expected parameter count={parameters.Length}. Are you changing animators at runtime?", gameObject);
361 return;
362 }
363
364 bool animatorEnabled = animator.enabled;
365 // need to read values from NetworkReader even if animator is disabled
366 ulong dirtyBits = reader.ReadULong();
367 for (int i = 0; i < parameterCount; i++)
368 {
369 if ((dirtyBits & (1ul << i)) == 0)
370 continue;
371
372 AnimatorControllerParameter par = parameters[i];
373 if (par.type == AnimatorControllerParameterType.Int)
374 {
375 int newIntValue = reader.ReadInt();
376 if (animatorEnabled)
377 animator.SetInteger(par.nameHash, newIntValue);
378 }
379 else if (par.type == AnimatorControllerParameterType.Float)
380 {
381 float newFloatValue = reader.ReadFloat();
382 if (animatorEnabled)
383 animator.SetFloat(par.nameHash, newFloatValue);
384 }
385 else if (par.type == AnimatorControllerParameterType.Bool)
386 {
387 bool newBoolValue = reader.ReadBool();
388 if (animatorEnabled)
389 animator.SetBool(par.nameHash, newBoolValue);
390 }
391 }
