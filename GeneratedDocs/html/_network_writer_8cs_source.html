79 // WriteBlittable<T> from DOTSNET.
80 // this is extremely fast, but only works for blittable types.
81 //
82 // Benchmark:
83 // WriteQuaternion x 100k, Macbook Pro 2015 @ 2.2Ghz, Unity 2018 LTS (debug mode)
84 //
85 // | Median | Min | Max | Avg | Std | (ms)
86 // before | 30.35 | 29.86 | 48.99 | 32.54 | 4.93 |
87 // blittable* | 5.69 | 5.52 | 27.51 | 7.78 | 5.65 |
88 //
89 // * without IsBlittable check
90 // => 4-6x faster!
91 //
92 // WriteQuaternion x 100k, Macbook Pro 2015 @ 2.2Ghz, Unity 2020.1 (release mode)
93 //
94 // | Median | Min | Max | Avg | Std | (ms)
95 // before | 9.41 | 8.90 | 23.02 | 10.72 | 3.07 |
96 // blittable* | 1.48 | 1.40 | 16.03 | 2.60 | 2.71 |
97 //
98 // * without IsBlittable check
99 // => 6x faster!
100 //
101 // Note:
102 // WriteBlittable assumes same endianness for server & client.
103 // All Unity 2018+ platforms are little endian.
104 // => run NetworkWriterTests.BlittableOnThisPlatform() to verify!
105 //
106 // This is not safe to expose to random structs.
107 // * StructLayout.Sequential is the default, which is safe.
108 // if the struct contains a reference type, it is converted to Auto.
109 // but since all structs here are unmanaged blittable, it's safe.
110 // see also: https://docs.microsoft.com/en-us/dotnet/api/system.runtime.interopservices.layoutkind?view=netframework-4.8#system-runtime-interopservices-layoutkind-sequential
111 // * StructLayout.Pack depends on CPU word size.
112 // this may be different 4 or 8 on some ARM systems, etc.
113 // this is not safe, and would cause bytes/shorts etc. to be padded.
114 // see also: https://docs.microsoft.com/en-us/dotnet/api/system.runtime.interopservices.structlayoutattribute.pack?view=net-6.0
115 // * If we force pack all to '1', they would have no padding which is
116 // great for bandwidth. but on some android systems, CPU can't read
117 // unaligned memory.
118 // see also: https://github.com/vis2k/Mirror/issues/3044
119 // * The only option would be to force explicit layout with multiples
120 // of word size. but this requires lots of weaver checking and is
121 // still questionable (IL2CPP etc.).
122 //
123 // Note: inlining WriteBlittable is enough. don't inline WriteInt etc.
124 // we don't want WriteBlittable to be copied in place everywhere.
125 internal unsafe void WriteBlittable<T>(T value)
126 where T : unmanaged
127 {
128 // check if blittable for safety
129#if UNITY_EDITOR
130 if (!UnsafeUtility.IsBlittable(typeof(T)))
131 {
132 Debug.LogError($"{typeof(T)} is not blittable!");
133 return;
134 }
135#endif
136 // calculate size
137 // sizeof(T) gets the managed size at compile time.
138 // Marshal.SizeOf<T> gets the unmanaged size at runtime (slow).
139 // => our 1mio writes benchmark is 6x slower with Marshal.SizeOf<T>
140 // => for blittable types, sizeof(T) is even recommended:
141 // https://docs.microsoft.com/en-us/dotnet/standard/native-interop/best-practices
142 int size = sizeof(T);
143
144 // ensure capacity
145 // NOTE that our runtime resizing comes at no extra cost because:
146 // 1. 'has space' checks are necessary even for fixed sized writers.
147 // 2. all writers will eventually be large enough to stop resizing.
148 EnsureCapacity(Position + size);
149
150 // write blittable
151 fixed (byte* ptr = &buffer[Position])
152 {
153#if UNITY_ANDROID
154 // on some android systems, assigning *(T*)ptr throws a NRE if
155 // the ptr isn't aligned (i.e. if Position is 1,2,3,5, etc.).
156 // here we have to use memcpy.
157 //
158 // => we can't get a pointer of a struct in C# without
159 // marshalling allocations
160 // => instead, we stack allocate an array of type T and use that
161 // => stackalloc avoids GC and is very fast. it only works for
162 // value types, but all blittable types are anyway.
163 //
164 // this way, we can still support blittable reads on android.
165 // see also: https://github.com/vis2k/Mirror/issues/3044
166 // (solution discovered by AIIO, FakeByte, mischa)
167 T* valueBuffer = stackalloc T[1]{value};
168 UnsafeUtility.MemCpy(ptr, valueBuffer, size);
169#else
170 // cast buffer to T* pointer, then assign value to the area
171 *(T*)ptr = value;
172#endif
173 }
174 Position += size;
175 }
