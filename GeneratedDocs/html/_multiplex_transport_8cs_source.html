16 // underlying transport connectionId to multiplexed connectionId lookup.
17 //
18 // originally we used a formula to map the connectionId:
19 // connectionId * transportAmount + transportId
20 //
21 // if we have 3 transports, then
22 // transport 0 will produce connection ids [0, 3, 6, 9, ...]
23 // transport 1 will produce connection ids [1, 4, 7, 10, ...]
24 // transport 2 will produce connection ids [2, 5, 8, 11, ...]
25 //
26 // however, some transports like kcp may give very large connectionIds.
27 // if they are near int.max, then "* transprotAmount + transportIndex"
28 // will overflow, resulting in connIds which can't be projected back.
29 // https://github.com/vis2k/Mirror/issues/3280
30 //
31 // instead, use a simple lookup with 0-indexed ids.
32 // with initial capacity to avoid runtime allocations.
33
34 // (original connectionId, transport#) to multiplexed connectionId
35 readonly Dictionary<KeyValuePair<int, int>, int> originalToMultiplexedId =
36 new Dictionary<KeyValuePair<int, int>, int>(100);
37
38 // multiplexed connectionId to (original connectionId, transport#)
39 readonly Dictionary<int, KeyValuePair<int, int>> multiplexedToOriginalId =
40 new Dictionary<int, KeyValuePair<int, int>>(100);
41
42 // next multiplexed id counter. start at 1 because 0 is reserved for host.
43 int nextMultiplexedId = 1;
44
45 // prevent log flood from OnGUI or similar per-frame updates
46 bool alreadyWarned;

84 // add to bidirection lookup. returns the multiplexed connectionId.
85 public int AddToLookup(int originalConnectionId, int transportIndex)
86 {
87 // add to both
88 KeyValuePair<int, int> pair = new KeyValuePair<int, int>(originalConnectionId, transportIndex);
89 int multiplexedId = nextMultiplexedId++;
90
91 originalToMultiplexedId[pair] = multiplexedId;
92 multiplexedToOriginalId[multiplexedId] = pair;
93
94 return multiplexedId;
95 }
96
97 public void RemoveFromLookup(int originalConnectionId, int transportIndex)
98 {
99 // remove from both
100 KeyValuePair<int, int> pair = new KeyValuePair<int, int>(originalConnectionId, transportIndex);
101 if (originalToMultiplexedId.TryGetValue(pair, out int multiplexedId))
102 {
103 originalToMultiplexedId.Remove(pair);
104 multiplexedToOriginalId.Remove(multiplexedId);
105 }
106 }
107
108 public bool OriginalId(int multiplexId, out int originalConnectionId, out int transportIndex)
109 {
110 if (!multiplexedToOriginalId.ContainsKey(multiplexId))
111 {
112 originalConnectionId = 0;
113 transportIndex = 0;
114 return false;
115 }
116
117 KeyValuePair<int, int> pair = multiplexedToOriginalId[multiplexId];
118 originalConnectionId = pair.Key;
119 transportIndex = pair.Value;
120 return true;
121 }
122
123 public int MultiplexId(int originalConnectionId, int transportIndex)
124 {
125 KeyValuePair<int, int> pair = new KeyValuePair<int, int>(originalConnectionId, transportIndex);
126 if (originalToMultiplexedId.TryGetValue(pair, out int multiplexedId))
127 return multiplexedId;
128 else
129 return 0;
130 }

252 void AddServerCallbacks()
253 {
254 // all underlying transports should call the multiplex transport's events
255 for (int i = 0; i < transports.Length; i++)
256 {
257 // this is required for the handlers, if I use i directly
258 // then all the handlers will use the last i
259 int transportIndex = i;
260 Transport transport = transports[i];
261
262 transport.OnServerConnected = (originalConnectionId =>
263 {
264 // invoke Multiplex event with multiplexed connectionId
265 int multiplexedId = AddToLookup(originalConnectionId, transportIndex);
266 OnServerConnected.Invoke(multiplexedId);
267 });
268
269 transport.OnServerDataReceived = (originalConnectionId, data, channel) =>
270 {
271 // invoke Multiplex event with multiplexed connectionId
272 int multiplexedId = MultiplexId(originalConnectionId, transportIndex);
273 if (multiplexedId == 0)
274 {
275 if (Utils.IsHeadless())
276 {
277 Console.ForegroundColor = ConsoleColor.Yellow;
278 Console.WriteLine($"[Multiplexer] Received data for unknown connectionId={originalConnectionId} on transport={transportIndex}");
279 Console.ResetColor();
280 }
281 else
282 Debug.LogWarning($"[Multiplexer] Received data for unknown connectionId={originalConnectionId} on transport={transportIndex}");
283
284 return;
285 }
286 OnServerDataReceived.Invoke(multiplexedId, data, channel);
287 };
288
289 transport.OnServerError = (originalConnectionId, error, reason) =>
290 {
291 // invoke Multiplex event with multiplexed connectionId
292 int multiplexedId = MultiplexId(originalConnectionId, transportIndex);
293 if (multiplexedId == 0)
294 {
295 if (Utils.IsHeadless())
296 {
297 Console.ForegroundColor = ConsoleColor.Red;
298 Console.WriteLine($"[Multiplexer] Received error for unknown connectionId={originalConnectionId} on transport={transportIndex}");
299 Console.ResetColor();
300 }
301 else
302 Debug.LogError($"[Multiplexer] Received error for unknown connectionId={originalConnectionId} on transport={transportIndex}");
303
304 return;
305 }
306 OnServerError.Invoke(multiplexedId, error, reason);
307 };
308
309 transport.OnServerDisconnected = originalConnectionId =>
310 {
311 // invoke Multiplex event with multiplexed connectionId
312 int multiplexedId = MultiplexId(originalConnectionId, transportIndex);
313 if (multiplexedId == 0)
314 {
315 if (Utils.IsHeadless())
316 {
317 Console.ForegroundColor = ConsoleColor.Yellow;
318 Console.WriteLine($"[Multiplexer] Received disconnect for unknown connectionId={originalConnectionId} on transport={transportIndex}");
319 Console.ResetColor();
320 }
321 else
322 Debug.LogWarning($"[Multiplexer] Received disconnect for unknown connectionId={originalConnectionId} on transport={transportIndex}");
323
324 return;
325 }
326 OnServerDisconnected.Invoke(multiplexedId);
327 RemoveFromLookup(originalConnectionId, transportIndex);
328 };
329 }

401 public override int GetMaxPacketSize(int channelId = 0)
402 {
403 // finding the max packet size in a multiplex environment has to be
404 // done very carefully:
405 // * servers run multiple transports at the same time
406 // * different clients run different transports
407 // * there should only ever be ONE true max packet size for everyone,
408 // otherwise a spawn message might be sent to all tcp sockets, but
409 // be too big for some udp sockets. that would be a debugging
410 // nightmare and allow for possible exploits and players on
411 // different platforms seeing a different game state.
412 // => the safest solution is to use the smallest max size for all
413 // transports. that will never fail.
414 int mininumAllowedSize = int.MaxValue;
415 foreach (Transport transport in transports)
416 {
417 int size = transport.GetMaxPacketSize(channelId);
418 mininumAllowedSize = Mathf.Min(size, mininumAllowedSize);
419 }
420 return mininumAllowedSize;
421 }
