179 public BufferPool(int bucketCount, int smallest, int largest)
180 {
181 if (bucketCount < 2) throw new ArgumentException("Count must be at least 2");
182 if (smallest < 1) throw new ArgumentException("Smallest must be at least 1");
183 if (largest < smallest) throw new ArgumentException("Largest must be greater than smallest");
184
185 this.bucketCount = bucketCount;
186 this.smallest = smallest;
187 this.largest = largest;
188
189 // split range over log scale (more buckets for smaller sizes)
190 double minLog = Math.Log(this.smallest);
191 double maxLog = Math.Log(this.largest);
192 double range = maxLog - minLog;
193 double each = range / (bucketCount - 1);
194
195 buckets = new BufferBucket[bucketCount];
196
197 for (int i = 0; i < bucketCount; i++)
198 {
199 double size = smallest * Math.Pow(Math.E, each * i);
200 buckets[i] = new BufferBucket((int)Math.Ceiling(size));
201 }
202
203 Validate();
204
205 // Example
206 // 5 count
207 // 20 smallest
208 // 16400 largest
209
210 // 3.0 log 20
211 // 9.7 log 16400
212
213 // 6.7 range 9.7 - 3
214 // 1.675 each 6.7 / (5-1)
215
216 // 20 e^ (3 + 1.675 * 0)
217 // 107 e^ (3 + 1.675 * 1)
218 // 572 e^ (3 + 1.675 * 2)
219 // 3056 e^ (3 + 1.675 * 3)
220 // 16,317 e^ (3 + 1.675 * 4)
221
222 // precision wont be lose when using doubles
223 }
224
225 [Conditional("UNITY_ASSERTIONS")]
226 void Validate()
227 {
228 if (buckets[0].arraySize != smallest)
229 Log.Error($"[SWT-BufferPool]: BufferPool Failed to create bucket for smallest. bucket:{buckets[0].arraySize} smallest:{smallest}");
230
231 int largestBucket = buckets[bucketCount - 1].arraySize;
232 // rounded using Ceiling, so allowed to be 1 more that largest
233 if (largestBucket != largest && largestBucket != largest + 1)
234 Log.Error($"[SWT-BufferPool]: BufferPool Failed to create bucket for largest. bucket:{largestBucket} smallest:{largest}");
235 }
236
237 public ArrayBuffer Take(int size)
238 {
239 if (size > largest)
240 throw new ArgumentException($"Size ({size}) is greater than largest ({largest})");
241
242 for (int i = 0; i < bucketCount; i++)
243 if (size <= buckets[i].arraySize)
244 return buckets[i].Take();
245
246 throw new ArgumentException($"Size ({size}) is greater than largest ({largest})");
247 }
