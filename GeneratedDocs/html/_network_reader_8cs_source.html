91 // ReadBlittable<T> from DOTSNET
92 // this is extremely fast, but only works for blittable types.
93 // => private to make sure nobody accidentally uses it for non-blittable
94 //
95 // Benchmark: see NetworkWriter.WriteBlittable!
96 //
97 // Note:
98 // ReadBlittable assumes same endianness for server & client.
99 // All Unity 2018+ platforms are little endian.
100 //
101 // This is not safe to expose to random structs.
102 // * StructLayout.Sequential is the default, which is safe.
103 // if the struct contains a reference type, it is converted to Auto.
104 // but since all structs here are unmanaged blittable, it's safe.
105 // see also: https://docs.microsoft.com/en-us/dotnet/api/system.runtime.interopservices.layoutkind?view=netframework-4.8#system-runtime-interopservices-layoutkind-sequential
106 // * StructLayout.Pack depends on CPU word size.
107 // this may be different 4 or 8 on some ARM systems, etc.
108 // this is not safe, and would cause bytes/shorts etc. to be padded.
109 // see also: https://docs.microsoft.com/en-us/dotnet/api/system.runtime.interopservices.structlayoutattribute.pack?view=net-6.0
110 // * If we force pack all to '1', they would have no padding which is
111 // great for bandwidth. but on some android systems, CPU can't read
112 // unaligned memory.
113 // see also: https://github.com/vis2k/Mirror/issues/3044
114 // * The only option would be to force explicit layout with multiples
115 // of word size. but this requires lots of weaver checking and is
116 // still questionable (IL2CPP etc.).
117 //
118 // Note: inlining ReadBlittable is enough. don't inline ReadInt etc.
119 // we don't want ReadBlittable to be copied in place everywhere.
120 internal unsafe T ReadBlittable<T>()
121 where T : unmanaged
122 {
123 // check if blittable for safety
124#if UNITY_EDITOR
125 if (!UnsafeUtility.IsBlittable(typeof(T)))
126 {
127 throw new ArgumentException($"{typeof(T)} is not blittable!");
128 }
129#endif
130
131 // calculate size
132 // sizeof(T) gets the managed size at compile time.
133 // Marshal.SizeOf<T> gets the unmanaged size at runtime (slow).
134 // => our 1mio writes benchmark is 6x slower with Marshal.SizeOf<T>
135 // => for blittable types, sizeof(T) is even recommended:
136 // https://docs.microsoft.com/en-us/dotnet/standard/native-interop/best-practices
137 int size = sizeof(T);
138
139 // ensure remaining
140 if (Remaining < size)
141 {
142 throw new EndOfStreamException($"ReadBlittable<{typeof(T)}> not enough data in buffer to read {size} bytes: {ToString()}");
143 }
144
145 // read blittable
146 T value;
147 fixed (byte* ptr = &buffer.Array[buffer.Offset + Position])
148 {
149#if UNITY_ANDROID
150 // on some android systems, reading *(T*)ptr throws a NRE if
151 // the ptr isn't aligned (i.e. if Position is 1,2,3,5, etc.).
152 // here we have to use memcpy.
153 //
154 // => we can't get a pointer of a struct in C# without
155 // marshalling allocations
156 // => instead, we stack allocate an array of type T and use that
157 // => stackalloc avoids GC and is very fast. it only works for
158 // value types, but all blittable types are anyway.
159 //
160 // this way, we can still support blittable reads on android.
161 // see also: https://github.com/vis2k/Mirror/issues/3044
162 // (solution discovered by AIIO, FakeByte, mischa)
163 T* valueBuffer = stackalloc T[1];
164 UnsafeUtility.MemCpy(valueBuffer, ptr, size);
165 value = valueBuffer[0];
166#else
167 // cast buffer to a T* pointer and then read from it.
168 value = *(T*)ptr;
169#endif
170 }
171 Position += size;
172 return value;
173 }
