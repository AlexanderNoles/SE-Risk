354 public static void WinCheck(Player current)
355 {
356 //We get the current player as an argument in case of a break in some other part of the code
357 //that would cause a player to attack not on their turn (or more likely the code doesn't think it is their turn)
358
359 if (current != null)
360 {
361 if (PlayOptionsManagement.IsConquestMode())
362 {
363 //Check if the current player has all the capitals in their possesion
364 if (Map.DoesPlayerHoldAllCapitals(current.GetIndex()))
365 {
366 gameOver = true;
367 }
368 }
369 else
370 {
371 //Check if only one player is left, this is pretty easy
372 //We can't check if only one player is left in the list
373 //as they are removed during THEIR deploy phase
374 //So we need to simply check if the amount of territories the current player holds is
375 //equal to all the territories
376
377 if (current.GetTerritories().Count == Map.GetTerritories().Count)
378 {
379 Debug.Log("won");
380 //This Player has won!
381 gameOver = true;
382 }
383 }
384 }
385
386 if (gameOver)
387 {
388 //Create game won info, to be used by game won screen
389 gameWonInfo = new GameWonInfo(!instance.playerList[0].IsDead());
390 if (current != null)
391 {
392 gameWonInfo.winnerName = current.GetColorName();
393
394 string winnerColour = "#" + current.GetColor().ToHexString();
395
396 gameWonInfo.winnerColor = winnerColour;
397 }
398 else
399 {
400 gameWonInfo.winnerName = "None";
401 gameWonInfo.winnerColor = "white";
402 }
403
404 //Load win screen menu
