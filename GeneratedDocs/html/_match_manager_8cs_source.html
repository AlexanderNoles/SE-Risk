422 public static void WinCheck(int current)
423 {
424 if (NetworkManagement.GetClientState() == NetworkManagement.ClientState.Client)
425 {
426 //Make request to server for win check
427 NetworkConnection.ServerWinCheck(current);
428 return;
429 }
430
431 //We get the current player as an argument in case of a break in some other part of the code
432 //that would cause a player to attack not on their turn (or more likely the code doesn't think it is their turn)
433
434 if (current != -1)
435 {
436 if (PlayOptionsManagement.IsConquestMode())
437 {
438 //Check if the current player has all the capitals in their possesion
439 if (Map.DoesPlayerHoldAllCapitals(current))
440 {
441 //This Player has won!
442 gameOver = true;
443 }
444 }
445 else
446 {
447 //Check if only one player is left, this is pretty easy
448 //Simply get all alive players and check if the current player is the only one alive
449
450 List<int> alivePlayers = Map.GetAlivePlayers();
451 if (alivePlayers.Count == 1 && alivePlayers[0] == current)
452 {
453 //This Player has won!
454 gameOver = true;
455 }
456 }
457 }
458
459 if (gameOver)
460 {
461 //Create game won info, to be used by game won screen
462 gameWonInfo = new GameWonInfo(current == PlayerInputHandler.GetLocalPlayerIndex());
463 if (current != -1)
464 {
465 gameWonInfo.winnerName = GetPlayerFromIndex(current).GetColorName();
466
467 string winnerColour = "#" + GetPlayerFromIndex(current).GetColor().ToHexString();
468
469 gameWonInfo.winnerColor = winnerColour;
470 }
471 else
472 {
473 gameWonInfo.winnerName = "None";
474 gameWonInfo.winnerColor = "white";
475 }
476
477 if (NetworkManagement.GetClientState() == NetworkManagement.ClientState.Offline)
478 {
479 StartExitTransition();
480 }
481 else
482 {
483 NetworkConnection.StartGameExitAcrossLobby(gameWonInfo.winnerName, gameWonInfo.winnerColor, current);
484 }
485 }
