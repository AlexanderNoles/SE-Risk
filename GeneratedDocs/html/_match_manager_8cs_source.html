220 public static void WinCheck(Player current)
221 {
222 //We get the current player as an argument in case of a break in some other part of the code
223 //that would cause a player to attack not on their turn (or more likely the code doesn't think it is their turn)
224
225 if (current != null)
226 {
227 if (PlayOptionsManagement.IsConquestMode())
228 {
229 //Check if the current player has all the capitals in their possesion
230 if (Map.DoesPlayerHoldAllCapitals(current))
231 {
232 gameOver = true;
233 }
234 }
235 else
236 {
237 //Check if only one player is left, this is pretty easy
238 //We can't check if only one player is left in the list
239 //as they are removed during THEIR deploy phase
240 //So we need to simply check if the amount of territories the current player holds is
241 //equal to all the territories
242
243 if (current.GetTerritories().Count == Map.GetTerritories().Count)
244 {
245 //This Player has won!
246 gameOver = true;
247 }
248 }
249 }
250
251 if (gameOver)
252 {
253 if (Map.IsSimulated())
254 {
255 //Restart game
256 instance.ResetGame(false);
257 }
258 else
259 {
260 //Create game won info, to be used by game won screen
261 gameWonInfo = new GameWonInfo();
262 if (current != null)
263 {
264 gameWonInfo.winnerName = current.GetColorName();
265 gameWonInfo.winnerColor = current.GetColorName().ToLower();
266 }
267 else
268 {
269 gameWonInfo.winnerName = "None";
270 gameWonInfo.winnerColor = "white";
271 }
272
273 //Load win screen menu
274 TransitionControl.onTransitionOver.AddListener(OnOutTransitionOver);
275 TransitionControl.RunTransition(TransitionControl.Transitions.SwipeIn);
276 }
277 }
