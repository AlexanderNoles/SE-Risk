58 public static string ReadString(this NetworkReader reader)
59 {
60 // read number of bytes
61 ushort size = reader.ReadUShort();
62
63 // null support, see NetworkWriter
64 if (size == 0)
65 return null;
66
67 ushort realSize = (ushort)(size - 1);
68
69 // make sure it's within limits to avoid allocation attacks etc.
70 if (realSize > NetworkWriter.MaxStringLength)
71 throw new EndOfStreamException($"NetworkReader.ReadString - Value too long: {realSize} bytes. Limit is: {NetworkWriter.MaxStringLength} bytes");
72
73 ArraySegment<byte> data = reader.ReadBytesSegment(realSize);
74
75 // convert directly from buffer to string via encoding
76 // throws in case of invalid utf8.
77 // see test: ReadString_InvalidUTF8()
78 return reader.encoding.GetString(data.Array, data.Offset, data.Count);
79 }

200 public static NetworkBehaviour ReadNetworkBehaviour(this NetworkReader reader)
201 {
202 // read netId first.
203 //
204 // IMPORTANT: if netId != 0, writer always writes componentIndex.
205 // reusing ReadNetworkIdentity() might return a null NetworkIdentity
206 // even if netId was != 0 but the identity disappeared on the client,
207 // resulting in unequal amounts of data being written / read.
208 // https://github.com/vis2k/Mirror/issues/2972
209 uint netId = reader.ReadUInt();
210 if (netId == 0)
211 return null;
212
213 // read component index in any case, BEFORE searching the spawned
214 // NetworkIdentity by netId.
215 byte componentIndex = reader.ReadByte();
216
217 // NOTE: a netId not being in spawned is common.
218 // for example, "[SyncVar] NetworkIdentity target" netId would not
219 // be known on client if the monster walks out of proximity for a
220 // moment. no need to log any error or warning here.
221 NetworkIdentity identity = Utils.GetSpawnedInServerOrClient(netId);
222
223 return identity != null
224 ? identity.NetworkBehaviours[componentIndex]
225 : null;
226 }

343 public static Texture2D ReadTexture2D(this NetworkReader reader)
344 {
345 // support 'null' textures for [SyncVar]s etc.
346 // https://github.com/vis2k/Mirror/issues/3144
347 short width = reader.ReadShort();
348 if (width == -1) return null;
349
350 // read height
351 short height = reader.ReadShort();
352
353 // prevent allocation attacks with a reasonable limit.
354 // server shouldn't allocate too much on client devices.
355 // client shouldn't allocate too much on server in ClientToServer [SyncVar]s.
356 // log an error and return default.
357 // we don't want attackers to be able to trigger exceptions.
358 int totalSize = width * height;
359 if (totalSize > NetworkReader.AllocationLimit)
360 {
361 Debug.LogWarning($"NetworkReader attempted to allocate a Texture2D with total size (width * height) of {totalSize}, which is larger than the allowed limit of {NetworkReader.AllocationLimit}.");
362 return null;
363 }
364
365 Texture2D texture2D = new Texture2D(width, height);
366
367 // read pixel content
368 Color32[] pixels = reader.ReadArray<Color32>();
369 texture2D.SetPixels32(pixels);
370 texture2D.Apply();
371 return texture2D;
372 }
