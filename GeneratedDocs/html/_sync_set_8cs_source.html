100 public override void OnSerializeDelta(NetworkWriter writer)
101 {
102 // write all the queued up changes
103 writer.WriteUInt((uint)changes.Count);

107 Change change = changes[i];

146 public override void OnDeserializeDelta(NetworkReader reader)
147 {
148 int changesCount = (int)reader.ReadUInt();
149
150 for (int i = 0; i < changesCount; i++)
151 {
152 Operation operation = (Operation)reader.ReadByte();
153
154 // apply the operation only if it is a new change
155 // that we have not applied yet
156 bool apply = changesAhead == 0;
157 T item = default;
158
159 switch (operation)
160 {
161 case Operation.OP_ADD:
162 item = reader.Read<T>();
163 if (apply)
164 {
165 objects.Add(item);
166 // add dirty + changes.
167 // ClientToServer needs to set dirty in server OnDeserialize.
168 // no access check: server OnDeserialize can always
169 // write, even for ClientToServer (for broadcasting).
170 AddOperation(Operation.OP_ADD, item, false);
171 }
172 break;
173
174 case Operation.OP_CLEAR:
175 if (apply)
176 {
177 objects.Clear();
178 // add dirty + changes.
179 // ClientToServer needs to set dirty in server OnDeserialize.
180 // no access check: server OnDeserialize can always
181 // write, even for ClientToServer (for broadcasting).
182 AddOperation(Operation.OP_CLEAR, false);
183 }
184 break;
185
186 case Operation.OP_REMOVE:
187 item = reader.Read<T>();
188 if (apply)
189 {
190 objects.Remove(item);
191 // add dirty + changes.
192 // ClientToServer needs to set dirty in server OnDeserialize.
193 // no access check: server OnDeserialize can always
194 // write, even for ClientToServer (for broadcasting).
195 AddOperation(Operation.OP_REMOVE, item, false);
196 }
197 break;
198 }
199
200 if (!apply)
201 {
202 // we just skipped this change
203 changesAhead--;
204 }
205 }
