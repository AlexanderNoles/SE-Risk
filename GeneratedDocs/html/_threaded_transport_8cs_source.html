189 // worker thread ///////////////////////////////////////////////////////
190 void ProcessThreadQueue()
191 {
192 // TODO deadlock protection. worker thread may be to slow to process all.
193 while (threadQueue.TryDequeue(out ThreadEvent elem))
194 {
195 switch (elem.type)
196 {
197 // SERVER EVENTS ///////////////////////////////////////////
198 case ThreadEventType.DoServerStart: // start listening
199 {
200 // call the threaded function
201 ThreadedServerStart();
202 break;
203 }
204 case ThreadEventType.DoServerSend:
205 {
206 // call the threaded function
207 ConcurrentNetworkWriterPooled writer = (ConcurrentNetworkWriterPooled)elem.param;
208 ThreadedServerSend(elem.connectionId.Value, writer, elem.channelId.Value);
209
210 // recycle writer to thread safe pool for reuse
211 ConcurrentNetworkWriterPool.Return(writer);
212 break;
213 }
214 case ThreadEventType.DoServerDisconnect:
215 {
216 // call the threaded function
217 ThreadedServerDisconnect(elem.connectionId.Value);
218 break;
219 }
220 case ThreadEventType.DoServerStop: // stop listening
221 {
222 // call the threaded function
223 ThreadedServerStop();
224 break;
225 }
226
227 // CLIENT EVENTS ///////////////////////////////////////////
228 case ThreadEventType.DoClientConnect:
229 {
230 // call the threaded function
231 if (elem.param is string address)
232 ThreadedClientConnect(address);
233 else if (elem.param is Uri uri)
234 ThreadedClientConnect(uri);
235 break;
236 }
237 case ThreadEventType.DoClientSend:
238 {
239 // call the threaded function
240 ConcurrentNetworkWriterPooled writer = (ConcurrentNetworkWriterPooled)elem.param;
241 ThreadedClientSend(writer, elem.channelId.Value);
242
243 // recycle writer to thread safe pool for reuse
244 ConcurrentNetworkWriterPool.Return(writer);
245 break;
246 }
247 case ThreadEventType.DoClientDisconnect:
248 {
249 // call the threaded function
250 ThreadedClientDisconnect();
251 break;
252 }
253
254 // SHUTDOWN ////////////////////////////////////////////////
255 case ThreadEventType.DoShutdown:
256 {
257 // call the threaded function
258 ThreadedShutdown();
259 break;
260 }
261 }
262 }

375 // client //////////////////////////////////////////////////////////////
376 // implementations need to use ThreadedEarlyUpdate
377 public override void ClientEarlyUpdate()
378 {
379 // regular transports process OnReceive etc. from early update.
380 // need to process the worker thread's queued events here too.
381 //
382 // process only up to N messages per tick here.
383 // if main thread becomes too slow, we don't want to deadlock.
384 int processed = 0;
385 while (clientMainQueue.TryDequeue(out ClientMainEvent elem))
386 {
387 switch (elem.type)
388 {
389 // CLIENT EVENTS ///////////////////////////////////////////
390 case ClientMainEventType.OnClientConnected:
391 {
392 // call original transport event
393 OnClientConnected?.Invoke();
394 break;
395 }
396 case ClientMainEventType.OnClientSent:
397 {
398 // call original transport event
399 ConcurrentNetworkWriterPooled writer = (ConcurrentNetworkWriterPooled)elem.param;
400 OnClientDataSent?.Invoke(writer, elem.channelId.Value);
401
402 // recycle writer to thread safe pool for reuse
403 ConcurrentNetworkWriterPool.Return(writer);
404 break;
405 }
406 case ClientMainEventType.OnClientReceived:
407 {
408 // call original transport event
409 ConcurrentNetworkWriterPooled writer = (ConcurrentNetworkWriterPooled)elem.param;
410 OnClientDataReceived?.Invoke(writer, elem.channelId.Value);
411
412 // recycle writer to thread safe pool for reuse
413 ConcurrentNetworkWriterPool.Return(writer);
414 break;
415 }
416 case ClientMainEventType.OnClientError:
417 {
418 // call original transport event
419 OnClientError?.Invoke(elem.error.Value, (string)elem.param);
420 break;
421 }
422 case ClientMainEventType.OnClientDisconnected:
423 {
424 // call original transport event
425 OnClientDisconnected?.Invoke();
426 break;
427 }
428 }
429
430 // process only up to N messages per tick here.
431 // if main thread becomes too slow, we don't want to deadlock.
432 if (++processed >= MaxProcessingPerTick)
433 {
434 Debug.LogWarning($"ThreadedTransport processed the limit of {MaxProcessingPerTick} messages this tick. Continuing next tick to prevent deadlock.");
435 break;
436 }
437 }
438 }
