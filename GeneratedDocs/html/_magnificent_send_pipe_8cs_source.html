2// safely sends messages from main thread to send thread.
3// -> thread safety built in

6// => hides all the complexity from telepathy
7// => easy to switch between stack/queue/concurrentqueue/etc.

10using System;
11using System.Collections.Generic;

13namespace Telepathy

15 public class MagnificentSendPipe
16 {
17 // message queue
18 // ConcurrentQueue allocates. lock{} instead.
19 // -> byte arrays are always of MaxMessageSize
20 // -> ArraySegment indicates the actual message content
21 //
22 // IMPORTANT: lock{} all usages!
23 readonly Queue<ArraySegment<byte>> queue = new Queue<ArraySegment<byte>>();
24
25 // byte[] pool to avoid allocations
26 // Take & Return is beautifully encapsulated in the pipe.
27 // the outside does not need to worry about anything.
28 // and it can be tested easily.
29 //
30 // IMPORTANT: lock{} all usages!
31 Pool<byte[]> pool;
32
33 // constructor
34 public MagnificentSendPipe(int MaxMessageSize)
35 {
36 // initialize pool to create max message sized byte[]s each time
37 pool = new Pool<byte[]>(() => new byte[MaxMessageSize]);
38 }

53 // enqueue a message
54 // arraysegment for allocation free sends later.
55 // -> the segment's array is only used until Enqueue() returns!
56 public void Enqueue(ArraySegment<byte> message)
57 {
58 // pool & queue usage always needs to be locked
59 lock (this)
60 {
61 // ArraySegment array is only valid until returning, so copy
62 // it into a byte[] that we can queue safely.
63
64 // get one from the pool first to avoid allocations
65 byte[] bytes = pool.Take();
66
67 // copy into it
68 Buffer.BlockCopy(message.Array, message.Offset, bytes, 0, message.Count);
69
70 // indicate which part is the message
71 ArraySegment<byte> segment = new ArraySegment<byte>(bytes, 0, message.Count);
72
73 // now enqueue it
74 queue.Enqueue(segment);
75 }
76 }

78 // send threads need to dequeue each byte[] and write it into the socket
79 // -> dequeueing one byte[] after another works, but it's WAY slower
80 // than dequeueing all immediately (locks only once)
81 // lock{} & DequeueAll is WAY faster than ConcurrentQueue & dequeue
82 // one after another:
83 //
84 // uMMORPG 450 CCU
85 // SafeQueue: 900-1440ms latency
86 // ConcurrentQueue: 2000ms latency
87 //
88 // -> the most obvious solution is to just return a list with all byte[]
89 // (which allocates) and then write each one into the socket
90 // -> a faster solution is to serialize each one into one payload buffer
91 // and pass that to the socket only once. fewer socket calls always
92 // give WAY better CPU performance(!)
93 // -> to avoid allocating a new list of entries each time, we simply
94 // serialize all entries into the payload here already
95 // => having all this complexity built into the pipe makes testing and
96 // modifying the algorithm super easy!
97 //
98 // IMPORTANT: serializing in here will allow us to return the byte[]
99 // entries back to a pool later to completely avoid
100 // allocations!
101 public bool DequeueAndSerializeAll(ref byte[] payload, out int packetSize)
102 {
103 // pool & queue usage always needs to be locked
104 lock (this)
105 {
106 // do nothing if empty
107 packetSize = 0;
108 if (queue.Count == 0)
109 return false;
110
111 // we might have multiple pending messages. merge into one
112 // packet to avoid TCP overheads and improve performance.
113 //
114 // IMPORTANT: Mirror & DOTSNET already batch into MaxMessageSize
115 // chunks, but we STILL pack all pending messages
116 // into one large payload so we only give it to TCP
117 // ONCE. This is HUGE for performance so we keep it!
118 packetSize = 0;
119 foreach (ArraySegment<byte> message in queue)
120 packetSize += 4 + message.Count; // header + content
121
122 // create payload buffer if not created yet or previous one is
123 // too small
124 // IMPORTANT: payload.Length might be > packetSize! don't use it!
125 if (payload == null || payload.Length < packetSize)
126 payload = new byte[packetSize];
127
128 // dequeue all byte[] messages and serialize into the packet
129 int position = 0;
130 while (queue.Count > 0)
131 {
132 // dequeue
133 ArraySegment<byte> message = queue.Dequeue();
134
135 // write header (size) into buffer at position
136 Utils.IntToBytesBigEndianNonAlloc(message.Count, payload, position);
137 position += 4;
138
139 // copy message into payload at position
140 Buffer.BlockCopy(message.Array, message.Offset, payload, position, message.Count);
141 position += message.Count;
142
143 // return to pool so it can be reused (avoids allocations!)
144 pool.Return(message.Array);
145 }
146
147 // we did serialize something
148 return true;
149 }
150 }

153 {
154 // pool & queue usage always needs to be locked
155 lock (this)
156 {
157 // clear queue, but via dequeue to return each byte[] to pool
158 while (queue.Count > 0)
159 {
160 pool.Return(queue.Dequeue().Array);
161 }
162 }
163 }
