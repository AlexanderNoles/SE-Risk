49 public static void Loop(Config config)
50 {
51 (Connection conn, int maxMessageSize, bool expectMask, ConcurrentQueue<Message> queue, BufferPool _) = config;
52
53 Profiler.BeginThreadProfiling("SimpleWeb", $"ReceiveLoop {conn.connId}");
54
55 byte[] readBuffer = new byte[Constants.HeaderSize + (expectMask ? Constants.MaskSize : 0) + maxMessageSize];
56 try
57 {
58 try
59 {
60 TcpClient client = conn.client;
61
62 while (client.Connected)
63 ReadOneMessage(config, readBuffer);
64
65 Log.Verbose($"[SWT-ReceiveLoop]: {conn} Not Connected");
66 }
67 catch (Exception)
68 {
69 // if interrupted we don't care about other exceptions
70 Utils.CheckForInterupt();
71 throw;
72 }
73 }
74 catch (ThreadInterruptedException e) { Log.InfoException(e); }
75 catch (ThreadAbortException e) { Log.InfoException(e); }
76 catch (ObjectDisposedException e) { Log.InfoException(e); }
77 catch (ReadHelperException e)
78 {
79 Log.InfoException(e);
80 }
81 catch (SocketException e)
82 {
83 // this could happen if wss client closes stream
84 Log.Warn($"[SWT-ReceiveLoop]: ReceiveLoop SocketException\n{e.Message}");
85 queue.Enqueue(new Message(conn.connId, e));
86 }
87 catch (IOException e)
88 {
89 // this could happen if client disconnects
90 Log.Warn($"[SWT-ReceiveLoop]: ReceiveLoop IOException\n{e.Message}");
91 queue.Enqueue(new Message(conn.connId, e));
92 }
93 catch (InvalidDataException e)
94 {
95 Log.Error($"[SWT-ReceiveLoop]: Invalid data from {conn}\n{e.Message}\n{e.StackTrace}\n\n");
96 queue.Enqueue(new Message(conn.connId, e));
97 }
98 catch (Exception e)
99 {
100 Log.Exception(e);
101 queue.Enqueue(new Message(conn.connId, e));
102 }
103 finally
104 {
105 Profiler.EndThreadProfiling();
106 conn.Dispose();
107 }
108 }

110 static void ReadOneMessage(Config config, byte[] buffer)
111 {
112 (Connection conn, int maxMessageSize, bool expectMask, ConcurrentQueue<Message> queue, BufferPool bufferPool) = config;
113 Stream stream = conn.stream;
114
115 Header header = ReadHeader(config, buffer);
116
117 int msgOffset = header.offset;
118 header.offset = ReadHelper.Read(stream, buffer, header.offset, header.payloadLength);
119
120 if (header.finished)
121 {
122 switch (header.opcode)
123 {
124 case 2:
125 HandleArrayMessage(config, buffer, msgOffset, header.payloadLength);
126 break;
127 case 8:
128 HandleCloseMessage(config, buffer, msgOffset, header.payloadLength);
129 break;
130 }
131 }
132 else
133 {
134 // todo cache this to avoid allocations
135 Queue<ArrayBuffer> fragments = new Queue<ArrayBuffer>();
136 fragments.Enqueue(CopyMessageToBuffer(bufferPool, expectMask, buffer, msgOffset, header.payloadLength));
137 int totalSize = header.payloadLength;
138
139 while (!header.finished)
140 {
141 header = ReadHeader(config, buffer, opCodeContinuation: true);
142
143 msgOffset = header.offset;
144 header.offset = ReadHelper.Read(stream, buffer, header.offset, header.payloadLength);
145 fragments.Enqueue(CopyMessageToBuffer(bufferPool, expectMask, buffer, msgOffset, header.payloadLength));
146
147 totalSize += header.payloadLength;
148 MessageProcessor.ThrowIfMsgLengthTooLong(totalSize, maxMessageSize);
149 }
150
151 ArrayBuffer msg = bufferPool.Take(totalSize);
152 msg.count = 0;
153 while (fragments.Count > 0)
154 {
155 ArrayBuffer part = fragments.Dequeue();
156
157 part.CopyTo(msg.array, msg.count);
158 msg.count += part.count;
159
160 part.Release();
161 }
162
163 // dump after mask off
164 Log.DumpBuffer($"[SWT-ReceiveLoop]: Message", msg);
165
166 queue.Enqueue(new Message(conn.connId, msg));
167 }
168 }

170 static Header ReadHeader(Config config, byte[] buffer, bool opCodeContinuation = false)
171 {
172 (Connection conn, int maxMessageSize, bool expectMask, ConcurrentQueue<Message> queue, BufferPool bufferPool) = config;
173 Stream stream = conn.stream;
174 Header header = new Header();
175
176 // read 2
177 header.offset = ReadHelper.Read(stream, buffer, header.offset, Constants.HeaderMinSize);
178 // log after first blocking call
179 Log.Flood($"[SWT-ReceiveLoop]: Message From {conn}");
180
181 if (MessageProcessor.NeedToReadShortLength(buffer))
182 header.offset = ReadHelper.Read(stream, buffer, header.offset, Constants.ShortLength);
183 if (MessageProcessor.NeedToReadLongLength(buffer))
184 header.offset = ReadHelper.Read(stream, buffer, header.offset, Constants.LongLength);
185
186 Log.DumpBuffer($"[SWT-ReceiveLoop]: Raw Header", buffer, 0, header.offset);
187
188 MessageProcessor.ValidateHeader(buffer, maxMessageSize, expectMask, opCodeContinuation);
189
190 if (expectMask)
191 header.offset = ReadHelper.Read(stream, buffer, header.offset, Constants.MaskSize);
192
193 header.opcode = MessageProcessor.GetOpcode(buffer);
194 header.payloadLength = MessageProcessor.GetPayloadLength(buffer);
195 header.finished = MessageProcessor.Finished(buffer);
196
197 Log.Flood($"[SWT-ReceiveLoop]: Header ln:{header.payloadLength} op:{header.opcode} mask:{expectMask}");
198
199 return header;
200 }

214 static ArrayBuffer CopyMessageToBuffer(BufferPool bufferPool, bool expectMask, byte[] buffer, int msgOffset, int payloadLength)
215 {
216 ArrayBuffer arrayBuffer = bufferPool.Take(payloadLength);
217
218 if (expectMask)
219 {
220 int maskOffset = msgOffset - Constants.MaskSize;
221 // write the result of toggle directly into arrayBuffer to avoid 2nd copy call
222 MessageProcessor.ToggleMask(buffer, msgOffset, arrayBuffer, payloadLength, buffer, maskOffset);
223 }
224 else
225 arrayBuffer.CopyFrom(buffer, msgOffset, payloadLength);
226
227 return arrayBuffer;
228 }
