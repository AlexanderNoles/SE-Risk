1// People should be able to see and report errors to the developer very easily.
2//
3// Unity's Developer Console only works in development builds and it only shows
4// errors. This class provides a console that works in all builds and also shows
5// log and warnings in development builds.
6//
7// Note: we don't include the stack trace, because that can also be grabbed from
8// the log files if needed.
9//
10// Note: there is no 'hide' button because we DO want people to see those errors
11// and report them back to us.
12//
13// Note: normal Debug.Log messages can be shown by building in Debug/Development
14// mode.

62 {
63 // only show at runtime, or if showInEditor is enabled
64 if (show)

68 // OnLog logs everything, even Debug.Log messages in release builds
69 // => this makes a lot of things easier. e.g. addon initialization logs.
70 // => it's really better to have than not to have those
71 void OnLog(string message, string stackTrace, LogType type)
72 {
73 // is this important?
74 // => always show exceptions & errors
75 // => usually a good idea to show warnings too, otherwise it's too
76 // easy to miss OnDeserialize warnings etc. in builds
77 bool isImportant = type == LogType.Error || type == LogType.Exception || type == LogType.Warning;
78
79 // use stack trace only if important
80 // (otherwise users would have to find and search the log file.
81 // seeing it in the console directly is way easier to deal with.)
82 // => only add \n if stack trace is available (only in debug builds)
83 if (isImportant && !string.IsNullOrWhiteSpace(stackTrace))
84 message += $"\n{stackTrace}";
85
86 // add to queue
87 log.Enqueue(new LogEntry(message, type));
88
89 // respect max entries
90 if (log.Count > maxLogCount)
91 log.Dequeue();
92
93 // become visible if it was important
94 // (no need to become visible for regular log. let the user decide.)
95 if (isImportant)
96 visible = true;
97
98 // auto scroll
99 scroll.y = float.MaxValue;
100 }
101
102 void Update()
103 {
104 if (show && Input.GetKeyDown(hotKey))
105 visible = !visible;
106 }

117 scroll = GUILayout.BeginScrollView(scroll, "Box", GUILayout.Width(Screen.width - offsetX - 10), GUILayout.Height(height));
118 foreach (LogEntry entry in log)
119 {
120 if (entry.type == LogType.Error || entry.type == LogType.Exception)
121 GUI.color = Color.red;
122 else if (entry.type == LogType.Warning)
123 GUI.color = Color.yellow;
124
125 GUILayout.Label(entry.message);
126 GUI.color = Color.white;
127 }
128 GUILayout.EndScrollView();

131 }
