39 private void ServerRemoveFromPending(EncryptedConnection con)
40 {
41 for (int i = 0; i < _serverPendingConnections.Count; i++)
42 {
43 if (_serverPendingConnections[i] == con)
44 {
45 // remove by swapping with last
46 int lastIndex = _serverPendingConnections.Count - 1;
47 _serverPendingConnections[i] = _serverPendingConnections[lastIndex];
48 _serverPendingConnections.RemoveAt(lastIndex);
49 break;
50 }
51 }

136 private bool HandleClientValidateServerPubKey(PubKeyInfo pubKeyInfo)
137 {
138 switch (clientValidateServerPubKey)
139 {
140 case ValidationMode.Off:
141 return true;
142 case ValidationMode.List:
143 return Array.IndexOf(clientTrustedPubKeySignatures, pubKeyInfo.Fingerprint) >= 0;
144 case ValidationMode.Callback:
145 return onClientValidateServerPubKey(pubKeyInfo);
146 default:
147 throw new ArgumentOutOfRangeException();
148 }

155 public override void ClientConnect(string address)
156 {
157 switch (clientValidateServerPubKey)
158 {
159 case ValidationMode.Off:
160 break;
161 case ValidationMode.List:
162 if (clientTrustedPubKeySignatures == null || clientTrustedPubKeySignatures.Length == 0)
163 {
164 OnClientError?.Invoke(TransportError.Unexpected, "Validate Server Public Key is set to List, but the clientTrustedPubKeySignatures list is empty.");
165 return;
166 }
167 break;
168 case ValidationMode.Callback:
169 if (onClientValidateServerPubKey == null)
170 {
171 OnClientError?.Invoke(TransportError.Unexpected, "Validate Server Public Key is set to Callback, but the onClientValidateServerPubKey handler is not set");
172 return;
173 }
174 break;
175 default:
176 throw new ArgumentOutOfRangeException();
177 }
