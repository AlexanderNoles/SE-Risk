91 public void Connect(string address, ushort port)
92 {
93 if (connected)
94 {
95 Log.Warning("[KCP] Client: already connected!");
96 return;
97 }
98
99 // resolve host name before creating peer.
100 // fixes: https://github.com/MirrorNetworking/Mirror/issues/3361
101 if (!Common.ResolveHostname(address, out IPAddress[] addresses))
102 {
103 // pass error to user callback. no need to log it manually.
104 OnError(ErrorCode.DnsResolve, $"Failed to resolve host: {address}");
105 OnDisconnectedCallback();
106 return;
107 }
108
109 // create fresh peer for each new session
110 // client doesn't need secure cookie.
111 Reset(config);
112
113 Log.Info($"[KCP] Client: connect to {address}:{port}");
114
115 // create socket
116 remoteEndPoint = new IPEndPoint(addresses[0], port);
117 socket = new Socket(remoteEndPoint.AddressFamily, SocketType.Dgram, ProtocolType.Udp);
118 active = true;
119
120 // recv & send are called from main thread.
121 // need to ensure this never blocks.
122 // even a 1ms block per connection would stop us from scaling.
123 socket.Blocking = false;
124
125 // configure buffer sizes
126 Common.ConfigureSocketBuffers(socket, config.RecvBufferSize, config.SendBufferSize);
127
128 // bind to endpoint so we can use send/recv instead of sendto/recvfrom.
129 socket.Connect(remoteEndPoint);
130
131 // immediately send a hello message to the server.
132 // server will call OnMessage and add the new connection.
133 // note that this still has cookie=0 until we receive the server's hello.
134 SendHello();
135 }

137 // io - input.
138 // virtual so it may be modified for relays, etc.
139 // call this while it returns true, to process all messages this tick.
140 // returned ArraySegment is valid until next call to RawReceive.
141 protected virtual bool RawReceive(out ArraySegment<byte> segment)
142 {
143 segment = default;
144 if (socket == null) return false;
145
146 try
147 {
148 return socket.ReceiveNonBlocking(rawReceiveBuffer, out segment);
149 }
150 // for non-blocking sockets, Receive throws WouldBlock if there is
151 // no message to read. that's okay. only log for other errors.
152 catch (SocketException e)
153 {
154 // the other end closing the connection is not an 'error'.
155 // but connections should never just end silently.
156 // at least log a message for easier debugging.
157 // for example, his can happen when connecting without a server.
158 // see test: ConnectWithoutServer().
159 Log.Info($"[KCP] Client.RawReceive: looks like the other end has closed the connection. This is fine: {e}");
160 base.Disconnect();
161 return false;
162 }
163 }

200 // insert raw IO. usually from socket.Receive.
201 // offset is useful for relays, where we may parse a header and then
202 // feed the rest to kcp.
203 public void RawInput(ArraySegment<byte> segment)
204 {
205 // ensure valid size: at least 1 byte for channel + 4 bytes for cookie
206 if (segment.Count <= 5) return;
207
208 // parse channel
209 // byte channel = segment[0]; ArraySegment[i] isn't supported in some older Unity Mono versions
210 byte channel = segment.Array[segment.Offset + 0];
211
212 // server messages always contain the security cookie.
213 // parse it, assign if not assigned, warn if suddenly different.
214 Utils.Decode32U(segment.Array, segment.Offset + 1, out uint messageCookie);
215 if (messageCookie == 0)
216 {
217 Log.Error($"[KCP] Client: received message with cookie=0, this should never happen. Server should always include the security cookie.");
218 }
219
220 if (cookie == 0)
221 {
222 cookie = messageCookie;
223 Log.Info($"[KCP] Client: received initial cookie: {cookie}");
224 }
225 else if (cookie != messageCookie)
226 {
227 Log.Warning($"[KCP] Client: dropping message with mismatching cookie: {messageCookie} expected: {cookie}.");
228 return;
229 }
230
231 // parse message
232 ArraySegment<byte> message = new ArraySegment<byte>(segment.Array, segment.Offset + 1+4, segment.Count - 1-4);
233
234 switch (channel)
235 {
236 case (byte)KcpChannel.Reliable:
237 {
238 OnRawInputReliable(message);
239 break;
240 }
241 case (byte)KcpChannel.Unreliable:
242 {
243 OnRawInputUnreliable(message);
244 break;
245 }
246 default:
247 {
248 // invalid channel indicates random internet noise.
249 // servers may receive random UDP data.
250 // just ignore it, but log for easier debugging.
251 Log.Warning($"[KCP] Client: invalid channel header: {channel}, likely internet noise");
252 break;
253 }
254 }
255 }

257 // process incoming messages. should be called before updating the world.
258 // virtual because relay may need to inject their own ping or similar.
259 public override void TickIncoming()
260 {
261 // recv on socket first, then process incoming
262 // (even if we didn't receive anything. need to tick ping etc.)
263 // (connection is null if not active)
264 if (active)
265 {
266 while (RawReceive(out ArraySegment<byte> segment))
267 RawInput(segment);
268 }
269
270 // RawReceive may have disconnected peer. active check again.
271 if (active) base.TickIncoming();
272 }
273
274 // process outgoing messages. should be called after updating the world.
275 // virtual because relay may need to inject their own ping or similar.
276 public override void TickOutgoing()
277 {
278 // process outgoing while active
279 if (active) base.TickOutgoing();
280 }
281
282 // process incoming and outgoing for convenience
283 // => ideally call ProcessIncoming() before updating the world and
284 // ProcessOutgoing() after updating the world for minimum latency
285 public virtual void Tick()
286 {
287 TickIncoming();
288 TickOutgoing();
289 }
