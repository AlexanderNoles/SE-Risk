35 // Update applies interpolation
36 void Update()
37 {
38 if (isServer) UpdateServerInterpolation();
39 // for all other clients (and for local player if !authority),
40 // we need to apply snapshots from the buffer.
41 // 'else if' because host mode shouldn't interpolate client
42 else if (isClient && !IsClientWithAuthority) UpdateClientInterpolation();
43 }
44
45 // LateUpdate broadcasts.
46 // movement scripts may change positions in Update.
47 // use LateUpdate to ensure changes are detected in the same frame.
48 // otherwise this may run before user update, delaying detection until next frame.
49 // this could cause visible jitter.
50 void LateUpdate()
51 {
52 // if server then always sync to others.
53 if (isServer) UpdateServerBroadcast();
54 // client authority, and local player (= allowed to move myself)?
55 // 'else if' because host mode shouldn't send anything to server.
56 // it is the server. don't overwrite anything there.
57 else if (isClient && IsClientWithAuthority) UpdateClientBroadcast();
58 }

60 protected virtual void CheckLastSendTime()
61 {
62 // We check interval every frame, and then send if interval is reached.
63 // So by the time sendIntervalCounter == sendIntervalMultiplier, data is sent,
64 // thus we reset the counter here.
65 // This fixes previous issue of, if sendIntervalMultiplier = 1, we send every frame,
66 // because intervalCounter is always = 1 in the previous version.
67
68 // Changing == to >= https://github.com/MirrorNetworking/Mirror/issues/3571
69
70 if (sendIntervalCounter >= sendIntervalMultiplier)
71 sendIntervalCounter = 0;
72
73 // timeAsDouble not available in older Unity versions.
74 if (AccurateInterval.Elapsed(NetworkTime.localTime, NetworkServer.sendInterval, ref lastSendIntervalTime))
75 sendIntervalCounter++;
76 }

78 void UpdateServerBroadcast()
79 {
80 // broadcast to all clients each 'sendInterval'
81 // (client with authority will drop the rpc)
82 // NetworkTime.localTime for double precision until Unity has it too
83 //
84 // IMPORTANT:
85 // snapshot interpolation requires constant sending.
86 // DO NOT only send if position changed. for example:
87 // ---
88 // * client sends first position at t=0
89 // * ... 10s later ...
90 // * client moves again, sends second position at t=10
91 // ---
92 // * server gets first position at t=0
93 // * server gets second position at t=10
94 // * server moves from first to second within a time of 10s
95 // => would be a super slow move, instead of a wait & move.
96 //
97 // IMPORTANT:
98 // DO NOT send nulls if not changed 'since last send' either. we
99 // send unreliable and don't know which 'last send' the other end
100 // received successfully.
101 //
102 // Checks to ensure server only sends snapshots if object is
103 // on server authority(!clientAuthority) mode because on client
104 // authority mode snapshots are broadcasted right after the authoritative
105 // client updates server in the command function(see above), OR,
106 // since host does not send anything to update the server, any client
107 // authoritative movement done by the host will have to be broadcasted
108 // here by checking IsClientWithAuthority.
109 // TODO send same time that NetworkServer sends time snapshot?
110 CheckLastSendTime();
111
112 if (sendIntervalCounter == sendIntervalMultiplier && // same interval as time interpolation!
113 (syncDirection == SyncDirection.ServerToClient || IsClientWithAuthority))
114 {
115 // send snapshot without timestamp.
116 // receiver gets it from batch timestamp to save bandwidth.
117 TransformSnapshot snapshot = Construct();
118
119 if (changedDetection)
120 {
121 cachedChangedComparison = CompareChangedSnapshots(snapshot);
122
123 if ((cachedChangedComparison == Changed.None || cachedChangedComparison == Changed.CompressRot) && hasSentUnchangedPosition && onlySyncOnChange) { return; }
124
125 SyncData syncData = new SyncData(cachedChangedComparison, snapshot);
126
127 RpcServerToClientSync(syncData);
128
129 if (cachedChangedComparison == Changed.None || cachedChangedComparison == Changed.CompressRot)
130 {
131 hasSentUnchangedPosition = true;
132 }
133 else
134 {
135 hasSentUnchangedPosition = false;
136 UpdateLastSentSnapshot(cachedChangedComparison, snapshot);
137 }
138 }
139 else
140 {
141 cachedSnapshotComparison = CompareSnapshots(snapshot);
142 if (cachedSnapshotComparison && hasSentUnchangedPosition && onlySyncOnChange) { return; }
143
144 if (compressRotation)
145 {
146 RpcServerToClientSyncCompressRotation(
147 // only sync what the user wants to sync
148 syncPosition && positionChanged ? snapshot.position : default(Vector3?),
149 syncRotation && rotationChanged ? Compression.CompressQuaternion(snapshot.rotation) : default(uint?),
150 syncScale && scaleChanged ? snapshot.scale : default(Vector3?)
151 );
152 }
153 else
154 {
155 RpcServerToClientSync(
156 // only sync what the user wants to sync
157 syncPosition && positionChanged ? snapshot.position : default(Vector3?),
158 syncRotation && rotationChanged ? snapshot.rotation : default(Quaternion?),
159 syncScale && scaleChanged ? snapshot.scale : default(Vector3?)
160 );
161 }
162
163 if (cachedSnapshotComparison)
164 {
165 hasSentUnchangedPosition = true;
166 }
167 else
168 {
169 hasSentUnchangedPosition = false;
170
171 // Fixes https://github.com/MirrorNetworking/Mirror/issues/3572
172 // This also fixes https://github.com/MirrorNetworking/Mirror/issues/3573
173 // with the exception of Quaternion.Angle sensitivity has to be > 0.16.
174 // Unity issue, we are leaving it as is.
175
176 if (positionChanged) lastSnapshot.position = snapshot.position;
177 if (rotationChanged) lastSnapshot.rotation = snapshot.rotation;
178 if (positionChanged) lastSnapshot.scale = snapshot.scale;
179 }
180 }
181 }

184 void UpdateServerInterpolation()
185 {
186 // apply buffered snapshots IF client authority
187 // -> in server authority, server moves the object
188 // so no need to apply any snapshots there.
189 // -> don't apply for host mode player objects either, even if in
190 // client authority mode. if it doesn't go over the network,
191 // then we don't need to do anything.
192 // -> connectionToClient is briefly null after scene changes:
193 // https://github.com/MirrorNetworking/Mirror/issues/3329
194 if (syncDirection == SyncDirection.ClientToServer &&
195 connectionToClient != null &&
196 !isOwned)
197 {
198 if (serverSnapshots.Count == 0) return;
199
200 // step the transform interpolation without touching time.
201 // NetworkClient is responsible for time globally.
202 SnapshotInterpolation.StepInterpolation(
203 serverSnapshots,
204 connectionToClient.remoteTimeline,
205 out TransformSnapshot from,
206 out TransformSnapshot to,
207 out double t);
208
209 // interpolate & apply
210 TransformSnapshot computed = TransformSnapshot.Interpolate(from, to, t);
211 Apply(computed, to);
212 }

401 protected virtual void OnClientToServerSync(Vector3? position, Quaternion? rotation, Vector3? scale)
402 {
403 // only apply if in client authority mode
404 if (syncDirection != SyncDirection.ClientToServer) return;
405
406 // protect against ever growing buffer size attacks
407 if (serverSnapshots.Count >= connectionToClient.snapshotBufferSizeLimit) return;

479 protected virtual void UpdateLastSentSnapshot(Changed change, TransformSnapshot currentSnapshot)
480 {
481 if (change == Changed.None || change == Changed.CompressRot) return;
482
483 if ((change & Changed.PosX) > 0) lastSnapshot.position.x = currentSnapshot.position.x;
484 if ((change & Changed.PosY) > 0) lastSnapshot.position.y = currentSnapshot.position.y;
485 if ((change & Changed.PosZ) > 0) lastSnapshot.position.z = currentSnapshot.position.z;
486
487 if (compressRotation)
488 {
489 if ((change & Changed.Rot) > 0) lastSnapshot.rotation = currentSnapshot.rotation;
490 }
491 else
492 {
493 Vector3 newRotation;
494 newRotation.x = (change & Changed.RotX) > 0 ? currentSnapshot.rotation.eulerAngles.x : lastSnapshot.rotation.eulerAngles.x;
495 newRotation.y = (change & Changed.RotY) > 0 ? currentSnapshot.rotation.eulerAngles.y : lastSnapshot.rotation.eulerAngles.y;
496 newRotation.z = (change & Changed.RotZ) > 0 ? currentSnapshot.rotation.eulerAngles.z : lastSnapshot.rotation.eulerAngles.z;
497
498 lastSnapshot.rotation = Quaternion.Euler(newRotation);
499 }
500
501 if ((change & Changed.Scale) > 0) lastSnapshot.scale = currentSnapshot.scale;
502 }

504 // Returns true if position, rotation AND scale are unchanged, within given sensitivity range.
505 // Note the sensitivity comparison are different for pos, rot and scale.
506 protected virtual Changed CompareChangedSnapshots(TransformSnapshot currentSnapshot)
507 {
508 Changed change = Changed.None;
509
510 if (syncPosition)
511 {
512 bool positionChanged = Vector3.SqrMagnitude(lastSnapshot.position - currentSnapshot.position) > positionSensitivity * positionSensitivity;
513 if (positionChanged)
514 {
515 if (Mathf.Abs(lastSnapshot.position.x - currentSnapshot.position.x) > positionSensitivity) change |= Changed.PosX;
516 if (Mathf.Abs(lastSnapshot.position.y - currentSnapshot.position.y) > positionSensitivity) change |= Changed.PosY;
517 if (Mathf.Abs(lastSnapshot.position.z - currentSnapshot.position.z) > positionSensitivity) change |= Changed.PosZ;
518 }
519 }
520
521 if (syncRotation)
522 {
523 if (compressRotation)
524 {
525 bool rotationChanged = Quaternion.Angle(lastSnapshot.rotation, currentSnapshot.rotation) > rotationSensitivity;
526 if (rotationChanged)
527 {
528 // Here we set all Rot enum flags, to tell us if there was a change in rotation
529 // when using compression. If no change, we don't write the compressed Quat.
530 change |= Changed.CompressRot;
531 change |= Changed.Rot;
532 }
533 else
534 {
535 change |= Changed.CompressRot;
536 }
537 }
538 else
539 {
540 if (Mathf.Abs(lastSnapshot.rotation.eulerAngles.x - currentSnapshot.rotation.eulerAngles.x) > rotationSensitivity) change |= Changed.RotX;
541 if (Mathf.Abs(lastSnapshot.rotation.eulerAngles.y - currentSnapshot.rotation.eulerAngles.y) > rotationSensitivity) change |= Changed.RotY;
542 if (Mathf.Abs(lastSnapshot.rotation.eulerAngles.z - currentSnapshot.rotation.eulerAngles.z) > rotationSensitivity) change |= Changed.RotZ;
543 }
544 }
545
546 if (syncScale)
547 {
548 if (Vector3.SqrMagnitude(lastSnapshot.scale - currentSnapshot.scale) > scaleSensitivity * scaleSensitivity) change |= Changed.Scale;
549 }
550
551 return change;
552 }

626 protected virtual void UpdateSyncData(ref SyncData syncData, SortedList<double, TransformSnapshot> snapshots)
627 {
628 if (syncData.changedDataByte == Changed.None || syncData.changedDataByte == Changed.CompressRot)
629 {
630 syncData.position = snapshots.Count > 0 ? snapshots.Values[snapshots.Count - 1].position : GetPosition();
631 syncData.quatRotation = snapshots.Count > 0 ? snapshots.Values[snapshots.Count - 1].rotation : GetRotation();
632 syncData.scale = snapshots.Count > 0 ? snapshots.Values[snapshots.Count - 1].scale : GetScale();
633 }
634 else
635 {
636 // Just going to update these without checking if syncposition or not,
637 // because if not syncing position, NT will not apply any position data
638 // to the target during Apply().
639
640 syncData.position.x = (syncData.changedDataByte & Changed.PosX) > 0 ? syncData.position.x : (snapshots.Count > 0 ? snapshots.Values[snapshots.Count - 1].position.x : GetPosition().x);
641 syncData.position.y = (syncData.changedDataByte & Changed.PosY) > 0 ? syncData.position.y : (snapshots.Count > 0 ? snapshots.Values[snapshots.Count - 1].position.y : GetPosition().y);
642 syncData.position.z = (syncData.changedDataByte & Changed.PosZ) > 0 ? syncData.position.z : (snapshots.Count > 0 ? snapshots.Values[snapshots.Count - 1].position.z : GetPosition().z);
643
644 // If compressRot is true, we already have the Quat in syncdata.
645 if ((syncData.changedDataByte & Changed.CompressRot) == 0)
646 {
647 syncData.vecRotation.x = (syncData.changedDataByte & Changed.RotX) > 0 ? syncData.vecRotation.x : (snapshots.Count > 0 ? snapshots.Values[snapshots.Count - 1].rotation.eulerAngles.x : GetRotation().eulerAngles.x);
648 syncData.vecRotation.y = (syncData.changedDataByte & Changed.RotY) > 0 ? syncData.vecRotation.y : (snapshots.Count > 0 ? snapshots.Values[snapshots.Count - 1].rotation.eulerAngles.y : GetRotation().eulerAngles.y); ;
