29 // get the two states closest to a given timestamp.
30 // those can be used to interpolate the exact state at that time.
31 public static bool Sample<T>(
32 SortedList<double, T> history,
33 double timestamp, // current server time
34 out T before,
35 out T after,
36 out int afterIndex,
37 out double t) // interpolation factor
38 {
39 before = default;
40 after = default;
41 t = 0;
42 afterIndex = -1;
43
44 // can't sample an empty history
45 // interpolation needs at least two entries.
46 // can't Lerp(A, A, 1.5). dist(A, A) * 1.5 is always 0.
47 if (history.Count < 2) {
48 return false;
49 }
50
51 // older than oldest
52 if (timestamp < history.Keys[0]) {
53 return false;
54 }
55
56 // iterate through the history
57 // TODO this needs to be faster than O(N)
58 // search around that area.
59 // should be O(1) most of the time, unless sampling was off.
60 int index = 0; // manually count when iterating. easier than for-int loop.
61 KeyValuePair<double, T> prev = new KeyValuePair<double, T>();
62 foreach (KeyValuePair<double, T> entry in history) {
63 // exact match?
64 if (timestamp == entry.Key)
65 {
66 before = entry.Value;
67 after = entry.Value;
68 afterIndex = index;
69 t = Mathd.InverseLerp(entry.Key, entry.Key, timestamp);
70 return true;
71 }
72
73 // did we check beyond timestamp? then return the previous two.
74 if (entry.Key > timestamp)
75 {
76 before = prev.Value;
77 after = entry.Value;
78 afterIndex = index;
79 t = Mathd.InverseLerp(prev.Key, entry.Key, timestamp);
80 return true;
81 }
82
83 // remember the last
84 prev = entry;
85 index += 1;
86 }
87
88 return false;
89 }

91 // inserts a server state into the client's history.
92 // readjust the deltas of the states after the inserted one.
93 // returns the corrected final position.
94 public static T CorrectHistory<T>(
95 SortedList<double, T> stateHistory,
96 int stateHistoryLimit,
97 T corrected, // corrected state with timestamp
98 T before, // state in history before the correction
99 T after, // state in history after the correction
100 int afterIndex) // index of the 'after' value so we don't need to find it again here
101 where T: PredictedState
102 {
103 // respect the limit
104 // TODO unit test to check if it respects max size
105 if (stateHistory.Count >= stateHistoryLimit)
106 stateHistory.RemoveAt(0);
107
108 // insert the corrected state into the history, or overwrite if already exists
109 stateHistory[corrected.timestamp] = corrected;
110
111 // the entry behind the inserted one still has the delta from (before, after).
112 // we need to correct it to (corrected, after).
113 //
114 // for example:
115 // before: (t=1.0, delta=10, position=10)
116 // after: (t=3.0, delta=20, position=30)
117 //
118 // then we insert:
119 // corrected: (t=2.5, delta=__, position=25)
120 //
121 // previous delta was from t=1.0 to t=3.0 => 2.0
122 // inserted delta is from t=2.5 to t=3.0 => 0.5
123 // multiplier is 0.5 / 2.0 = 0.25
124 // multiply 'after.delta(20)' by 0.25 to get the new 'after.delta(5)
125 //
126 // so the new history is:
127 // before: (t=1.0, delta=10, position=10)
128 // corrected: (t=2.5, delta=__, position=25)
129 // after: (t=3.0, delta= 5, position=__)
130 //
131 // so when we apply the correction, the new after.position would be:
132 // corrected.position(25) + after.delta(5) = 30
133 //
134 double previousDeltaTime = after.timestamp - before.timestamp; // 3.0 - 1.0 = 2.0
135 double correctedDeltaTime = after.timestamp - corrected.timestamp; // 3.0 - 2.5 = 0.5
136
137 // fix multiplier becoming NaN if previousDeltaTime is 0:
138 // double multiplier = correctedDeltaTime / previousDeltaTime;
139 double multiplier = previousDeltaTime != 0 ? correctedDeltaTime / previousDeltaTime : 0; // 0.5 / 2.0 = 0.25
140
141 // recalculate 'after.delta' with the multiplier
142 after.positionDelta = Vector3.Lerp(Vector3.zero, after.positionDelta, (float)multiplier);
143 after.velocityDelta = Vector3.Lerp(Vector3.zero, after.velocityDelta, (float)multiplier);
144 after.angularVelocityDelta = Vector3.Lerp(Vector3.zero, after.angularVelocityDelta, (float)multiplier);
145 // Quaternions always need to be normalized in order to be a valid rotation after operations
146 after.rotationDelta = Quaternion.Slerp(Quaternion.identity, after.rotationDelta, (float)multiplier).normalized;
147
148 // changes aren't saved until we overwrite them in the history
149 stateHistory[after.timestamp] = after;
150
151 // second step: readjust all absolute values by rewinding client's delta moves on top of it.
152 T last = corrected;
153 for (int i = afterIndex; i < stateHistory.Count; ++i)
154 {
155 double key = stateHistory.Keys[i];
156 T entry = stateHistory.Values[i];
157
158 // correct absolute position based on last + delta.
159 entry.position = last.position + entry.positionDelta;
160 entry.velocity = last.velocity + entry.velocityDelta;
161 entry.angularVelocity = last.angularVelocity + entry.angularVelocityDelta;
162 // Quaternions always need to be normalized in order to be a valid rotation after operations
163 entry.rotation = (entry.rotationDelta * last.rotation).normalized; // quaternions add delta by multiplying in this order
164
165 // save the corrected entry into history.
166 stateHistory[key] = entry;
167
168 // save last
169 last = entry;
170 }
171
172 // third step: return the final recomputed state.
173 return last;
174 }
