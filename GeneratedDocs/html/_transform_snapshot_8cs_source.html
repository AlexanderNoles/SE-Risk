6namespace Mirror

8 // NetworkTransform Snapshot
9 public struct TransformSnapshot : Snapshot
10 {
11 // time or sequence are needed to throw away older snapshots.
12 //
13 // glenn fiedler starts with a 16 bit sequence number.
14 // supposedly this is meant as a simplified example.
15 // in the end we need the remote timestamp for accurate interpolation
16 // and buffering over time.
17 //
18 // note: in theory, IF server sends exactly(!) at the same interval then
19 // the 16 bit ushort timestamp would be enough to calculate the
20 // remote time (sequence * sendInterval). but Unity's update is
21 // not guaranteed to run on the exact intervals / do catchup.
22 // => remote timestamp is better for now
23 //
24 // [REMOTE TIME, NOT LOCAL TIME]
25 // => DOUBLE for long term accuracy & batching gives us double anyway
26 public double remoteTime { get; set; }
27
28 // the local timestamp (when we received it)
29 // used to know if the first two snapshots are old enough to start.
30 public double localTime { get; set; }
31
32 public Vector3 position;
33 public Quaternion rotation;
34 public Vector3 scale;
35
36 public TransformSnapshot(double remoteTime, double localTime, Vector3 position, Quaternion rotation, Vector3 scale)
37 {
38 this.remoteTime = remoteTime;
39 this.localTime = localTime;
40 this.position = position;
41 this.rotation = rotation;
42 this.scale = scale;
43 }
44
45 public static TransformSnapshot Interpolate(TransformSnapshot from, TransformSnapshot to, double t)
46 {
47 // NOTE:
48 // Vector3 & Quaternion components are float anyway, so we can
49 // keep using the functions with 't' as float instead of double.
50 return new TransformSnapshot(
51 // interpolated snapshot is applied directly. don't need timestamps.
52 0, 0,
53 // lerp position/rotation/scale unclamped in case we ever need
54 // to extrapolate. atm SnapshotInterpolation never does.
55 Vector3.LerpUnclamped(from.position, to.position, (float)t),
56 // IMPORTANT: LerpUnclamped(0, 60, 1.5) extrapolates to ~86.
57 // SlerpUnclamped(0, 60, 1.5) extrapolates to 90!
58 // (0, 90, 1.5) is even worse. for Lerp.
59 // => Slerp works way better for our euler angles.
60 Quaternion.SlerpUnclamped(from.rotation, to.rotation, (float)t),
61 Vector3.LerpUnclamped(from.scale, to.scale, (float)t)
62 );
63 }
64
65 public override string ToString() =>
66 $"TransformSnapshot(remoteTime={remoteTime:F2}, localTime={localTime:F2}, pos={position}, rot={rotation}, scale={scale})";
67 }
