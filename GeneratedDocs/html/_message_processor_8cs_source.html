81 static int GetMessageLength(byte[] buffer, int offset, byte lenByte)
82 {
83 if (lenByte == Constants.UshortPayloadLength)
84 {
85 // header is 2 bytes
86 ushort value = 0;
87 value |= (ushort)(buffer[offset + 2] << 8);
88 value |= buffer[offset + 3];
89
90 return value;
91 }
92 else if (lenByte == Constants.UlongPayloadLength)
93 {
94 // header is 8 bytes
95 ulong value = 0;
96 value |= ((ulong)buffer[offset + 2] << 56);
97 value |= ((ulong)buffer[offset + 3] << 48);
98 value |= ((ulong)buffer[offset + 4] << 40);
99 value |= ((ulong)buffer[offset + 5] << 32);
100 value |= ((ulong)buffer[offset + 6] << 24);
101 value |= ((ulong)buffer[offset + 7] << 16);
102 value |= ((ulong)buffer[offset + 8] << 8);
103 value |= ((ulong)buffer[offset + 9] << 0);
104
105 if (value > int.MaxValue)
106 throw new NotSupportedException($"Can't receive payloads larger that int.max: {int.MaxValue}");
107
108 return (int)value;
109 }
110 else // is less than 126
111 {
112 // header is 2 bytes long
113 return lenByte;
114 }

125 static void ThrowIfBadOpCode(int opcode, bool finished, bool opCodeContinuation)
126 {
127 // 0 = continuation
128 // 2 = binary
129 // 8 = close
130
131 // do we expect Continuation?
132 if (opCodeContinuation)
133 {
134 // good it was Continuation

136 return;
137
138 // bad, wasn't Continuation
139 throw new InvalidDataException("Expected opcode to be Continuation");
140 }
141 else if (!finished)
142 {
143 // fragmented message, should be binary
144 if (opcode == 2)
145 return;
146
147 throw new InvalidDataException("Expected opcode to be binary");
148 }
149 else
150 {
151 // normal message, should be binary or close
152 if (opcode == 2 || opcode == 8)
153 return;
154
155 throw new InvalidDataException("Expected opcode to be binary or close");
156 }
