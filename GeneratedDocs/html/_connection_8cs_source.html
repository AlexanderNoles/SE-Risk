49 public void Dispose()
50 {
51 Log.Verbose($"[SWT-Connection]: Dispose {ToString()}");
52
53 // check hasDisposed first to stop ThreadInterruptedException on lock
54 if (hasDisposed) return;
55
56 Log.Verbose($"[SWT-Connection]: Connection Close: {ToString()}");
57
58 lock (disposedLock)
59 {
60 // check hasDisposed again inside lock to make sure no other object has called this
61 if (hasDisposed) return;
62
63 hasDisposed = true;
64
65 // stop threads first so they don't try to use disposed objects
66 receiveThread.Interrupt();
67 sendThread?.Interrupt();
68
69 try
70 {
71 // stream
72 stream?.Dispose();
73 stream = null;
74 client.Dispose();
75 client = null;
76 }
77 catch (Exception e)
78 {
79 Log.Exception(e);
80 }
81
82 sendPending.Dispose();
83
84 // release all buffers in send queue
85 while (sendQueue.TryDequeue(out ArrayBuffer buffer))
86 buffer.Release();
87
88 onDispose.Invoke(this);
89 }
