83 // helper function to simulate latency
84 float SimulateLatency(int channeldId)
85 {
86 // spike over perlin noise.
87 // no spikes isn't realistic.
88 // sin is too predictable / no realistic.
89 // perlin is still deterministic and random enough.
90#if !UNITY_2020_3_OR_NEWER
91 float spike = Noise((float)NetworkTime.localTime * jitterSpeed) * jitter;
92#else
93 float spike = Noise((float)Time.unscaledTimeAsDouble * jitterSpeed) * jitter;
94#endif
95
96 // base latency
97 switch (channeldId)
98 {
99 case Channels.Reliable:
100 return latency/1000 + spike;
101 case Channels.Unreliable:
102 return latency/1000 + spike;
103 default:
104 return 0;
105 }
106 }

108 // helper function to simulate a send with latency/loss/scramble
109 void SimulateSend(
110 int connectionId,
111 ArraySegment<byte> segment,
112 int channelId,
113 float latency,
114 List<QueuedMessage> reliableQueue,
115 List<QueuedMessage> unreliableQueue)
116 {
117 // segment is only valid after returning. copy it.
118 // (allocates for now. it's only for testing anyway.)
119 byte[] bytes = new byte[segment.Count];
120 Buffer.BlockCopy(segment.Array, segment.Offset, bytes, 0, segment.Count);
121
122 // enqueue message. send after latency interval.
123 QueuedMessage message = new QueuedMessage
124 {
125 connectionId = connectionId,
126 bytes = bytes,
127#if !UNITY_2020_3_OR_NEWER
128 time = NetworkTime.localTime + latency
129#else
130 time = Time.unscaledTimeAsDouble + latency
131#endif
132 };
133
134 switch (channelId)
135 {
136 case Channels.Reliable:
137 // simulate latency
138 reliableQueue.Add(message);
139 break;
140 case Channels.Unreliable:
141 // simulate packet loss
142 bool drop = random.NextDouble() < unreliableLoss/100;
143 if (!drop)
144 {
145 // simulate scramble (Random.Next is < max, so +1)
146 bool scramble = random.NextDouble() < unreliableScramble/100;
147 int last = unreliableQueue.Count;
148 int index = scramble ? random.Next(0, last + 1) : last;
149
150 // simulate latency
151 unreliableQueue.Insert(index, message);
152 }
153 break;
154 default:
155 Debug.LogError($"{nameof(LatencySimulation)} unexpected channelId: {channelId}");
156 break;
157 }
158 }

227 public override void ClientLateUpdate()
228 {
229 // flush reliable messages after latency.
230 // need to iterate all, since queue isn't a sortedlist.
231 for (int i = 0; i < reliableClientToServer.Count; ++i)
232 {
233 // message ready to be sent?
234 QueuedMessage message = reliableClientToServer[i];
235#if !UNITY_2020_3_OR_NEWER
236 if (message.time <= NetworkTime.localTime)
237#else
238 if (message.time <= Time.unscaledTimeAsDouble)
239#endif
240 {
241 // send and eat
242 wrap.ClientSend(new ArraySegment<byte>(message.bytes), Channels.Reliable);
243 reliableClientToServer.RemoveAt(i);
244 --i;
245 }
246 }
247
248 // flush unreliable messages after latency.
249 // need to iterate all, since queue isn't a sortedlist.
250 for (int i = 0; i < unreliableClientToServer.Count; ++i)
251 {
252 // message ready to be sent?
253 QueuedMessage message = unreliableClientToServer[i];
254#if !UNITY_2020_3_OR_NEWER
255 if (message.time <= NetworkTime.localTime)
256#else
257 if (message.time <= Time.unscaledTimeAsDouble)
258#endif
