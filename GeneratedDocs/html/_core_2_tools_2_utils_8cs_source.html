150 // pretty print seconds as hours:minutes:seconds(.milliseconds/100)s.
151 // double for long running servers.
152 public static string PrettySeconds(double seconds)
153 {
154 TimeSpan t = TimeSpan.FromSeconds(seconds);
155 string res = "";
156 if (t.Days > 0) res += $"{t.Days}d";
157 if (t.Hours > 0) res += $"{(res.Length > 0 ? " " : "")}{t.Hours}h";
158 if (t.Minutes > 0) res += $"{(res.Length > 0 ? " " : "")}{t.Minutes}m";
159 // 0.5s, 1.5s etc. if any milliseconds. 1s, 2s etc. if any seconds
160 if (t.Milliseconds > 0) res += $"{(res.Length > 0 ? " " : "")}{t.Seconds}.{(t.Milliseconds / 100)}s";
161 else if (t.Seconds > 0) res += $"{(res.Length > 0 ? " " : "")}{t.Seconds}s";
162 // if the string is still empty because the value was '0', then at least
163 // return the seconds instead of returning an empty string
164 return res != "" ? res : "0s";
165 }

188 // keep a GUI window in screen.

197 // ensure max
198 rect.x = Math.Min(rect.x, Screen.width - rect.width);
