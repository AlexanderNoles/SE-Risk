7 // ClientState OBJECT that can be handed to the ReceiveThread safely.
8 // => allows us to create a NEW OBJECT every time we connect and start a
9 // receive thread.
10 // => perfectly protects us against data races. fixes all the flaky tests
11 // where .Connecting or .client would still be used by a dieing thread
12 // while attempting to use it for a new connection attempt etc.
13 // => creating a fresh client state each time is the best solution against
14 // data races here!
15 class ClientConnectionState : ConnectionState
16 {
17 public Thread receiveThread;
18
19 // TcpClient.Connected doesn't check if socket != null, which
20 // results in NullReferenceExceptions if connection was closed.
21 // -> let's check it manually instead
22 public bool Connected => client != null &&
23 client.Client != null &&
24 client.Client.Connected;
25
26 // TcpClient has no 'connecting' state to check. We need to keep track
27 // of it manually.
28 // -> checking 'thread.IsAlive && !Connected' is not enough because the
29 // thread is alive and connected is false for a short moment after
30 // disconnecting, so this would cause race conditions.
31 // -> we use a threadsafe bool wrapper so that ThreadFunction can remain
32 // static (it needs a common lock)
33 // => Connecting is true from first Connect() call in here, through the
34 // thread start, until TcpClient.Connect() returns. Simple and clear.
35 // => bools are atomic according to
36 // https://docs.microsoft.com/en-us/dotnet/csharp/language-reference/language-specification/variables
37 // made volatile so the compiler does not reorder access to it
38 public volatile bool Connecting;
39
40 // thread safe pipe for received messages
41 // => inside client connection state so that we can create a new state
42 // each time we connect
43 // (unlike server which has one receive pipe for all connections)
44 public readonly MagnificentReceivePipe receivePipe;
45
46 // constructor always creates new TcpClient for client connection!
47 public ClientConnectionState(int MaxMessageSize) : base(new TcpClient(), MaxMessageSize)
48 {
49 // create receive pipe with max message size for pooling
50 receivePipe = new MagnificentReceivePipe(MaxMessageSize);
51 }
52
53 // dispose all the state safely
54 public void Dispose()
55 {
56 // close client
57 client.Close();
58
59 // wait until thread finished. this is the only way to guarantee
60 // that we can call Connect() again immediately after Disconnect
61 // -> calling .Join would sometimes wait forever, e.g. when
62 // calling Disconnect while trying to connect to a dead end
63 receiveThread?.Interrupt();
64
65 // we interrupted the receive Thread, so we can't guarantee that
66 // connecting was reset. let's do it manually.
67 Connecting = false;
68
69 // clear send pipe. no need to hold on to elements.
70 // (unlike receiveQueue, which is still needed to process the
71 // latest Disconnected message, etc.)
72 sendPipe.Clear();
73
74 // IMPORTANT: DO NOT CLEAR RECEIVE PIPE.
75 // we still want to process disconnect messages in Tick()!
76
77 // let go of this client completely. the thread ended, no one uses
78 // it anymore and this way Connected is false again immediately.
79 client = null;
80 }
81 }

92 // -> avoids ever growing queue memory if network is slower than input
93 // -> disconnecting is great for load balancing. better to disconnect
94 // one connection than risking every connection / the whole server

118 // the thread function
119 // STATIC to avoid sharing state!
120 // => pass ClientState object. a new one is created for each new thread!
121 // => avoids data races where an old dieing thread might still modify
122 // the current thread's state :/
123 static void ReceiveThreadFunction(ClientConnectionState state, string ip, int port, int MaxMessageSize, bool NoDelay, int SendTimeout, int ReceiveTimeout, int ReceiveQueueLimit)
124
125 {
126 Thread sendThread = null;
127
128 // absolutely must wrap with try/catch, otherwise thread
129 // exceptions are silent
130 try
131 {
132 // connect (blocking)
133 state.client.Connect(ip, port);
134 state.Connecting = false; // volatile!
135
136 // set socket options after the socket was created in Connect()
137 // (not after the constructor because we clear the socket there)
138 state.client.NoDelay = NoDelay;
139 state.client.SendTimeout = SendTimeout;
140 state.client.ReceiveTimeout = ReceiveTimeout;
141
142 // start send thread only after connected
143 // IMPORTANT: DO NOT SHARE STATE ACROSS MULTIPLE THREADS!
144 sendThread = new Thread(() => { ThreadFunctions.SendLoop(0, state.client, state.sendPipe, state.sendPending); });
145 sendThread.IsBackground = true;
146 sendThread.Start();
147
148 // run the receive loop
149 // (receive pipe is shared across all loops)
150 ThreadFunctions.ReceiveLoop(0, state.client, MaxMessageSize, state.receivePipe, ReceiveQueueLimit);
151 }
152 catch (SocketException exception)
153 {
154 // this happens if (for example) the ip address is correct
155 // but there is no server running on that ip/port
156 Log.Info("Client Recv: failed to connect to ip=" + ip + " port=" + port + " reason=" + exception);
157 }
158 catch (ThreadInterruptedException)
159 {
160 // expected if Disconnect() aborts it
161 }
162 catch (ThreadAbortException)
163 {
164 // expected if Disconnect() aborts it
165 }
166 catch (ObjectDisposedException)
167 {
168 // expected if Disconnect() aborts it and disposed the client
169 // while ReceiveThread is in a blocking Connect() call
170 }
171 catch (Exception exception)
172 {
173 // something went wrong. probably important.
174 Log.Error("Client Recv Exception: " + exception);
175 }
176 // add 'Disconnected' event to receive pipe so that the caller
177 // knows that the Connect failed. otherwise they will never know
178 state.receivePipe.Enqueue(0, EventType.Disconnected, default);
179
180 // sendthread might be waiting on ManualResetEvent,
181 // so let's make sure to end it if the connection
182 // closed.
183 // otherwise the send thread would only end if it's
184 // actually sending data while the connection is
185 // closed.
186 sendThread?.Interrupt();
187
188 // Connect might have failed. thread might have been closed.
189 // let's reset connecting state no matter what.
190 state.Connecting = false;
191
192 // if we got here then we are done. ReceiveLoop cleans up already,
193 // but we may never get there if connect fails. so let's clean up
194 // here too.
195 state.client?.Close();
196 }

198 public void Connect(string ip, int port)
199 {
200 // not if already started
201 if (Connecting || Connected)
202 {
203 Log.Warning("Telepathy Client can not create connection because an existing connection is connecting or connected");
204 return;
205 }
206
207 // overwrite old thread's state object. create a new one to avoid
208 // data races where an old dieing thread might still modify the
209 // current state! fixes all the flaky tests!
210 state = new ClientConnectionState(MaxMessageSize);
211
212 // We are connecting from now until Connect succeeds or fails
213 state.Connecting = true;
214
215 // create a TcpClient with perfect IPv4, IPv6 and hostname resolving
216 // support.
217 //
218 // * TcpClient(hostname, port): works but would connect (and block)
219 // already
220 // * TcpClient(AddressFamily.InterNetworkV6): takes Ipv4 and IPv6
221 // addresses but only connects to IPv6 servers (e.g. Telepathy).
222 // does NOT connect to IPv4 servers (e.g. Mirror Booster), even
223 // with DualMode enabled.
224 // * TcpClient(): creates IPv4 socket internally, which would force
225 // Connect() to only use IPv4 sockets.
226 //
227 // => the trick is to clear the internal IPv4 socket so that Connect
228 // resolves the hostname and creates either an IPv4 or an IPv6
229 // socket as needed (see TcpClient source)
230 state.client.Client = null; // clear internal IPv4 socket until Connect()
231
232 // client.Connect(ip, port) is blocking. let's call it in the thread
233 // and return immediately.
234 // -> this way the application doesn't hang for 30s if connect takes
235 // too long, which is especially good in games
236 // -> this way we don't async client.BeginConnect, which seems to
237 // fail sometimes if we connect too many clients too fast
238 state.receiveThread = new Thread(() => {
239 ReceiveThreadFunction(state, ip, port, MaxMessageSize, NoDelay, SendTimeout, ReceiveTimeout, ReceiveQueueLimit);
240 });
241 state.receiveThread.IsBackground = true;
242 state.receiveThread.Start();
243 }

258 // send message to server using socket connection.
259 // arraysegment for allocation free sends later.
260 // -> the segment's array is only used until Send() returns!
261 public bool Send(ArraySegment<byte> message)
262 {
263 if (Connected)
264 {
265 // respect max message size to avoid allocation attacks.
266 if (message.Count <= MaxMessageSize)
267 {
268 // check send pipe limit
269 if (state.sendPipe.Count < SendQueueLimit)
270 {
271 // add to thread safe send pipe and return immediately.
272 // calling Send here would be blocking (sometimes for long
273 // times if other side lags or wire was disconnected)
274 state.sendPipe.Enqueue(message);
275 state.sendPending.Set(); // interrupt SendThread WaitOne()
276 return true;
277 }
278 // disconnect if send queue gets too big.
279 // -> avoids ever growing queue memory if network is slower
280 // than input
281 // -> avoids ever growing latency as well
282 //
283 // note: while SendThread always grabs the WHOLE send queue
284 // immediately, it's still possible that the sending
285 // blocks for so long that the send queue just gets
286 // way too big. have a limit - better safe than sorry.
287 else
288 {
289 // log the reason
290 Log.Warning($"Client.Send: sendPipe reached limit of {SendQueueLimit}. This can happen if we call send faster than the network can process messages. Disconnecting to avoid ever growing memory & latency.");
291
292 // just close it. send thread will take care of the rest.
293 state.client.Close();
294 return false;
295 }
296 }
297 Log.Error("Client.Send: message too big: " + message.Count + ". Limit: " + MaxMessageSize);
298 return false;
299 }
300 Log.Warning("Client.Send: not connected!");
301 return false;
302 }

304 // tick: processes up to 'limit' messages
305 // => limit parameter to avoid deadlocks / too long freezes if server or
306 // client is too slow to process network load
307 // => Mirror & DOTSNET need to have a process limit anyway.
308 // might as well do it here and make life easier.
309 // => returns amount of remaining messages to process, so the caller
310 // can call tick again as many times as needed (or up to a limit)
311 //
312 // Tick() may process multiple messages, but Mirror needs a way to stop
313 // processing immediately if a scene change messages arrives. Mirror
314 // can't process any other messages during a scene change.
315 // (could be useful for others too)
316 // => make sure to allocate the lambda only once in transports
317 public int Tick(int processLimit, Func<bool> checkEnabled = null)
318 {
319 // only if state was created yet (after connect())
320 // note: we don't check 'only if connected' because we want to still
321 // process Disconnect messages afterwards too!
322 if (state == null)
323 return 0;
324
325 // process up to 'processLimit' messages
326 for (int i = 0; i < processLimit; ++i)
327 {
328 // check enabled in case a Mirror scene message arrived
329 if (checkEnabled != null && !checkEnabled())
330 break;
331
332 // peek first. allows us to process the first queued entry while
333 // still keeping the pooled byte[] alive by not removing anything.
334 if (state.receivePipe.TryPeek(out int _, out EventType eventType, out ArraySegment<byte> message))
335 {
336 switch (eventType)
337 {
338 case EventType.Connected:
339 OnConnected?.Invoke();
340 break;
341 case EventType.Data:
342 OnData?.Invoke(message);
343 break;
344 case EventType.Disconnected:
345 OnDisconnected?.Invoke();
346 break;
347 }
348
349 // IMPORTANT: now dequeue and return it to pool AFTER we are
350 // done processing the event.
351 state.receivePipe.TryDequeue();
352 }
353 // no more messages. stop the loop.
354 else break;
355 }
356
357 // return what's left to process for next time
358 return state.receivePipe.TotalCount;
359 }
