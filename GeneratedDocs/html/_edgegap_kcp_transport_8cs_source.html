36 protected override void Awake()
37 {
38 // logging
39 // Log.Info should use Debug.Log if enabled, or nothing otherwise
40 // (don't want to spam the console on headless servers)
41 if (debugLog)
42 Log.Info = Debug.Log;
43 else
44 Log.Info = _ => {};
45 Log.Warning = Debug.LogWarning;
46 Log.Error = Debug.LogError;
47
48 // create config from serialized settings.
49 // with MaxPayload as max size to respect relay overhead.
50 config = new KcpConfig(DualMode, RecvBufferSize, SendBufferSize, MaxPayload, NoDelay, Interval, FastResend, false, SendWindowSize, ReceiveWindowSize, Timeout, MaxRetransmit);
51
52 // client (NonAlloc version is not necessary anymore)
53 client = new EdgegapKcpClient(
54 () => OnClientConnected.Invoke(),
55 (message, channel) => OnClientDataReceived.Invoke(message, FromKcpChannel(channel)),
56 () => OnClientDisconnected.Invoke(),
57 (error, reason) => OnClientError.Invoke(ToTransportError(error), reason),
58 config
59 );
60
61 // server
62 server = new EdgegapKcpServer(
63 (connectionId) => OnServerConnected.Invoke(connectionId),
64 (connectionId, message, channel) => OnServerDataReceived.Invoke(connectionId, message, FromKcpChannel(channel)),
65 (connectionId) => OnServerDisconnected.Invoke(connectionId),
66 (connectionId, error, reason) => OnServerError.Invoke(connectionId, ToTransportError(error), reason),
67 config);
68
69 if (statisticsLog)
70 InvokeRepeating(nameof(OnLogStatistics), 1, 1);
71
72 Debug.Log("EdgegapTransport initialized!");
73 }
