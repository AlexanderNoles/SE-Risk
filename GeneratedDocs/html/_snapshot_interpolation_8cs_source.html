30 // calculate timescale for catch-up / slow-down
31 // note that negative threshold should be <0.
32 // caller should verify (i.e. Unity OnValidate).
33 // improves branch prediction.
34 public static double Timescale(
35 double drift, // how far we are off from bufferTime
36 double catchupSpeed, // in % [0,1]
37 double slowdownSpeed, // in % [0,1]
38 double absoluteCatchupNegativeThreshold, // in seconds (careful, we may run out of snapshots)
39 double absoluteCatchupPositiveThreshold) // in seconds
40 {
41 // if the drift time is too large, it means we are behind more time.
42 // so we need to speed up the timescale.
43 // note the threshold should be sendInterval * catchupThreshold.
44 if (drift > absoluteCatchupPositiveThreshold)
45 {
46 // localTimeline += 0.001; // too simple, this would ping pong
47 return 1 + catchupSpeed; // n% faster
48 }
49
50 // if the drift time is too small, it means we are ahead of time.
51 // so we need to slow down the timescale.
52 // note the threshold should be sendInterval * catchupThreshold.
53 if (drift < absoluteCatchupNegativeThreshold)
54 {
55 // localTimeline -= 0.001; // too simple, this would ping pong
56 return 1 - slowdownSpeed; // n% slower
57 }
58
59 // keep constant timescale while within threshold.
60 // this way we have perfectly smooth speed most of the time.
61 return 1;
62 }

64 // calculate dynamic buffer time adjustment
65 public static double DynamicAdjustment(
66 double sendInterval,
67 double jitterStandardDeviation,
68 double dynamicAdjustmentTolerance)
69 {
70 // jitter is equal to delivery time standard variation.
71 // delivery time is made up of 'sendInterval+jitter'.
72 // .Average would be dampened by the constant sendInterval
73 // .StandardDeviation is the changes in 'jitter' that we want
74 // so add it to send interval again.
75 double intervalWithJitter = sendInterval + jitterStandardDeviation;
76
77 // how many multiples of sendInterval is that?
78 // we want to convert to bufferTimeMultiplier later.
79 double multiples = intervalWithJitter / sendInterval;
80
81 // add the tolerance
82 double safezone = multiples + dynamicAdjustmentTolerance;
83 // UnityEngine.Debug.Log($"sendInterval={sendInterval:F3} jitter std={jitterStandardDeviation:F3} => that is ~{multiples:F1} x sendInterval + {dynamicAdjustmentTolerance} => dynamic bufferTimeMultiplier={safezone}");
84 return safezone;
85 }

87 // helper function to insert a snapshot if it doesn't exist yet.
88 // extra function so we can use it for both cases:
89 // NetworkClient global timeline insertions & adjustments via Insert<T>.
90 // NetworkBehaviour local insertion without any time adjustments.
91 public static bool InsertIfNotExists<T>(
92 SortedList<double, T> buffer, // snapshot buffer
93 int bufferLimit, // don't grow infinitely
94 T snapshot) // the newly received snapshot
95 where T : Snapshot
96 {
97 // slow clients may not be able to process incoming snapshots fast enough.
98 // infinitely growing snapshots would make it even worse.
99 // for example, run NetworkRigidbodyBenchmark while deep profiling client.
100 // the client just grows and reallocates the buffer forever.
101 if (buffer.Count >= bufferLimit) return false;
102
103 // SortedList does not allow duplicates.
104 // we don't need to check ContainsKey (which is expensive).
105 // simply add and compare count before/after for the return value.
106
107 //if (buffer.ContainsKey(snapshot.remoteTime)) return false; // too expensive
108 // buffer.Add(snapshot.remoteTime, snapshot); // throws if key exists
109
110 int before = buffer.Count;
111 buffer[snapshot.remoteTime] = snapshot; // overwrites if key exists
112 return buffer.Count > before;
113 }

115 // clamp timeline for cases where it gets too far behind.
116 // for example, a client app may go into the background and get updated
117 // with 1hz for a while. by the time it's back it's at least 30 frames
118 // behind, possibly more if the transport also queues up. In this
119 // scenario, at 1% catch up it took around 20+ seconds to finally catch
120 // up. For these kinds of scenarios it will be better to snap / clamp.
121 //
122 // to reproduce, try snapshot interpolation demo and press the button to
123 // simulate the client timeline at multiple seconds behind. it'll take
124 // a long time to catch up if the timeline is a long time behind.
125 public static double TimelineClamp(
126 double localTimeline,
127 double bufferTime,
128 double latestRemoteTime)
129 {
130 // we want local timeline to always be 'bufferTime' behind remote.
131 double targetTime = latestRemoteTime - bufferTime;
132
133 // we define a boundary of 'bufferTime' around the target time.
134 // this is where catchup / slowdown will happen.
135 // outside of the area, we clamp.
136 double lowerBound = targetTime - bufferTime; // how far behind we can get
137 double upperBound = targetTime + bufferTime; // how far ahead we can get
138 return Mathd.Clamp(localTimeline, lowerBound, upperBound);
139 }

141 // call this for every received snapshot.
142 // adds / inserts it to the list & initializes local time if needed.
143 public static void InsertAndAdjust<T>(
144 SortedList<double, T> buffer, // snapshot buffer
145 int bufferLimit, // don't grow infinitely
146 T snapshot, // the newly received snapshot
147 ref double localTimeline, // local interpolation time based on server time
148 ref double localTimescale, // timeline multiplier to apply catchup / slowdown over time
149 float sendInterval, // for debugging
150 double bufferTime, // offset for buffering
151 double catchupSpeed, // in % [0,1]
152 double slowdownSpeed, // in % [0,1]
153 ref ExponentialMovingAverage driftEma, // for catchup / slowdown
154 float catchupNegativeThreshold, // in % of sendInteral (careful, we may run out of snapshots)
155 float catchupPositiveThreshold, // in % of sendInterval
156 ref ExponentialMovingAverage deliveryTimeEma) // for dynamic buffer time adjustment
157 where T : Snapshot
158 {
159 // first snapshot?
160 // initialize local timeline.
161 // we want it to be behind by 'offset'.
162 //
163 // note that the first snapshot may be a lagging packet.
164 // so we would always be behind by that lag.
165 // this requires catchup later.
166 if (buffer.Count == 0)
167 localTimeline = snapshot.remoteTime - bufferTime;
168
169 // insert into the buffer.
170 //
171 // note that we might insert it between our current interpolation
172 // which is fine, it adds another data point for accuracy.
173 //
174 // note that insert may be called twice for the same key.
175 // by default, this would throw.
176 // need to handle it silently.
177 if (InsertIfNotExists(buffer, bufferLimit, snapshot))
178 {
179 // dynamic buffer adjustment needs delivery interval jitter
180 if (buffer.Count >= 2)
181 {
182 // note that this is not entirely accurate for scrambled inserts.
183 //
184 // we always use the last two, not what we just inserted
185 // even if we were to use the diff for what we just inserted,
186 // a scrambled insert would still not be 100% accurate:
187 // => assume a buffer of AC, with delivery time C-A
188 // => we then insert B, with delivery time B-A
189 // => but then technically the first C-A wasn't correct,
190 // as it would have to be C-B
191 //
192 // in practice, scramble is rare and won't make much difference
193 double previousLocalTime = buffer.Values[buffer.Count - 2].localTime;
194 double lastestLocalTime = buffer.Values[buffer.Count - 1].localTime;
195
196 // this is the delivery time since last snapshot
197 double localDeliveryTime = lastestLocalTime - previousLocalTime;
198
199 // feed the local delivery time to the EMA.
200 // this is what the original stream did too.
201 // our final dynamic buffer adjustment is different though.
202 // we use standard deviation instead of average.
203 deliveryTimeEma.Add(localDeliveryTime);
204 }
205
206 // adjust timescale to catch up / slow down after each insertion
207 // because that is when we add new values to our EMA.
208
209 // we want localTimeline to be about 'bufferTime' behind.
210 // for that, we need the delivery time EMA.
211 // snapshots may arrive out of order, we can not use last-timeline.
212 // we need to use the inserted snapshot's time - timeline.
213 double latestRemoteTime = snapshot.remoteTime;
214
215 // ensure timeline stays within a reasonable bound behind/ahead.
216 localTimeline = TimelineClamp(localTimeline, bufferTime, latestRemoteTime);
217
218 // calculate timediff after localTimeline override changes
219 double timeDiff = latestRemoteTime - localTimeline;
220
221 // next, calculate average of a few seconds worth of timediffs.
222 // this gives smoother results.
223 //
224 // to calculate the average, we could simply loop through the
225 // last 'n' seconds worth of timediffs, but:
226 // - our buffer may only store a few snapshots (bufferTime)
227 // - looping through seconds worth of snapshots every time is
228 // expensive
229 //
230 // to solve this, we use an exponential moving average.
231 // https://en.wikipedia.org/wiki/Moving_average#Exponential_moving_average
232 // which is basically fancy math to do the same but faster.
233 // additionally, it allows us to look at more timeDiff values
234 // than we sould have access to in our buffer :)
235 driftEma.Add(timeDiff);
236
237 // timescale depends on driftEma.
238 // driftEma only changes when inserting.
239 // therefore timescale only needs to be calculated when inserting.
240 // saves CPU cycles in Update.
241
242 // next up, calculate how far we are currently away from bufferTime
243 double drift = driftEma.Value - bufferTime;
244
245 // convert relative thresholds to absolute values based on sendInterval
246 double absoluteNegativeThreshold = sendInterval * catchupNegativeThreshold;
247 double absolutePositiveThreshold = sendInterval * catchupPositiveThreshold;
248
249 // next, set localTimescale to catchup consistently in Update().
250 // we quantize between default/catchup/slowdown,
251 // this way we have 'default' speed most of the time(!).
252 // and only catch up / slow down for a little bit occasionally.
253 // a consistent multiplier would never be exactly 1.0.
254 localTimescale = Timescale(drift, catchupSpeed, slowdownSpeed, absoluteNegativeThreshold, absolutePositiveThreshold);
255
256 // debug logging
257 // UnityEngine.Debug.Log($"sendInterval={sendInterval:F3} bufferTime={bufferTime:F3} drift={drift:F3} driftEma={driftEma.Value:F3} timescale={localTimescale:F3} deliveryIntervalEma={deliveryTimeEma.Value:F3}");
258 }

261 // sample snapshot buffer to find the pair around the given time.
262 // returns indices so we can use it with RemoveRange to clear old snaps.
263 // make sure to use use buffer.Values[from/to], not buffer[from/to].
264 // make sure to only call this is we have > 0 snapshots.
265 public static void Sample<T>(
266 SortedList<double, T> buffer, // snapshot buffer
267 double localTimeline, // local interpolation time based on server time
268 out int from, // the snapshot <= time
269 out int to, // the snapshot >= time
270 out double t) // interpolation factor
271 where T : Snapshot
272 {
273 from = -1;
274 to = -1;
275 t = 0;
276
277 // sample from [0,count-1] so we always have two at 'i' and 'i+1'.
278 for (int i = 0; i < buffer.Count - 1; ++i)
279 {
280 // is local time between these two?
281 T first = buffer.Values[i];
282 T second = buffer.Values[i + 1];
283 if (localTimeline >= first.remoteTime &&
284 localTimeline <= second.remoteTime)
285 {
286 // use these two snapshots
287 from = i;
288 to = i + 1;
289 t = Mathd.InverseLerp(first.remoteTime, second.remoteTime, localTimeline);
290 return;
291 }
292 }
293
294 // didn't find two snapshots around local time.
295 // so pick either the first or last, depending on which is closer.
296
297 // oldest snapshot ahead of local time?
298 if (buffer.Values[0].remoteTime > localTimeline)
299 {
300 from = to = 0;
301 t = 0;
302 }
303 // otherwise initialize both to the last one
304 else
305 {
306 from = to = buffer.Count - 1;
307 t = 0;
308 }
309 }

311 // progress local timeline every update.
312 //
313 // ONLY CALL IF SNAPSHOTS.COUNT > 0!
314 //
315 // decoupled from Step<T> for easier testing and so we can progress
316 // time only once in NetworkClient, while stepping for each component.
317 [MethodImpl(MethodImplOptions.AggressiveInlining)]
318 public static void StepTime(
319 double deltaTime, // engine delta time (unscaled)
320 ref double localTimeline, // local interpolation time based on server time
321 double localTimescale) // catchup / slowdown is applied to time every update)
322 {
323 // move local forward in time, scaled with catchup / slowdown applied
324 localTimeline += deltaTime * localTimescale;
325 }

327 // sample, clear old.
328 // call this every update.
329 //
330 // ONLY CALL IF SNAPSHOTS.COUNT > 0!
331 //
332 // returns true if there is anything to apply (requires at least 1 snap)
333 // from/to/t are out parameters instead of an interpolated 'computed'.
334 // this allows us to store from/to/t globally (i.e. in NetworkClient)
335 // and have each component apply the interpolation manually.
336 // besides, passing "Func Interpolate" would allocate anyway.
337 public static void StepInterpolation<T>(
338 SortedList<double, T> buffer, // snapshot buffer
339 double localTimeline, // local interpolation time based on server time
340 out T fromSnapshot, // we interpolate 'from' this snapshot
341 out T toSnapshot, // 'to' this snapshot
342 out double t) // at ratio 't' [0,1]
343 where T : Snapshot
344 {
345 // check this in caller:
346 // nothing to do if there are no snapshots at all yet
347 // if (buffer.Count == 0) return false;
348
349 // sample snapshot buffer at local interpolation time
350 Sample(buffer, localTimeline, out int from, out int to, out t);
351
352 // save from/to
353 fromSnapshot = buffer.Values[from];
354 toSnapshot = buffer.Values[to];
355
356 // remove older snapshots that we definitely don't need anymore.
357 // after(!) using the indices.
358 //
359 // if we have 3 snapshots and we are between 2nd and 3rd:
360 // from = 1, to = 2
361 // then we need to remove the first one, which is exactly 'from'.
362 // because 'from-1' = 0 would remove none.
363 buffer.RemoveRange(from);
364 }
365
366 // update time, sample, clear old.
367 // call this every update.
368 //
369 // ONLY CALL IF SNAPSHOTS.COUNT > 0!
370 //
371 // returns true if there is anything to apply (requires at least 1 snap)
372 // from/to/t are out parameters instead of an interpolated 'computed'.
373 // this allows us to store from/to/t globally (i.e. in NetworkClient)
374 // and have each component apply the interpolation manually.
375 // besides, passing "Func Interpolate" would allocate anyway.
376 public static void Step<T>(
377 SortedList<double, T> buffer, // snapshot buffer
378 double deltaTime, // engine delta time (unscaled)
379 ref double localTimeline, // local interpolation time based on server time
380 double localTimescale, // catchup / slowdown is applied to time every update
381 out T fromSnapshot, // we interpolate 'from' this snapshot
382 out T toSnapshot, // 'to' this snapshot
383 out double t) // at ratio 't' [0,1]
384 where T : Snapshot
385 {
386 StepTime(deltaTime, ref localTimeline, localTimescale);
387 StepInterpolation(buffer, localTimeline, out fromSnapshot, out toSnapshot, out t);
388 }
