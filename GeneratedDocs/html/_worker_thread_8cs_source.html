2// thread with proper exception handling, profling, init, cleanup, etc. for Unity.
3// use this from main thread.

10namespace Mirror
11{
12 public class WorkerThread

14 readonly Thread thread;

16 protected volatile bool active;
17
18 // stopwatch so we don't need to use Unity's Time (engine independent)
19 readonly Stopwatch watch = new Stopwatch();
20
21 // callbacks need to be set after constructor.
22 // inheriting classes can't pass their member funcs to base ctor.
23 // don't set them while the thread is running!
24 public Action Init;
25 public Action Tick;
26 public Action Cleanup;
27
28 public WorkerThread(string identifier)
29 {
30 // start the thread wrapped in safety guard
31 // if main application terminates, this thread needs to terminate too
32 thread = new Thread(
33 () => Guard(identifier)
34 );
35 thread.IsBackground = true;
36 }
37
38 public void Start()
39 {
40 // only if thread isn't already running
41 if (thread.IsAlive)
42 {
43 Debug.LogWarning("WorkerThread is still active, can't start it again.");
44 return;
45 }
46
47 active = true;
48 thread.Start();
49 }
50
51 // signal the thread to stop gracefully.
52 // returns immediately, but the thread may take a while to stop.
53 // may be overwritten to clear more flags like 'computing' etc.
54 public virtual void SignalStop() => active = false;
55
56 // wait for the thread to fully stop
57 public bool StopBlocking(float timeout)
58 {
59 // only if alive
60 if (!thread.IsAlive) return true;
61
62 // double precision for long running servers.
63 watch.Restart();
64
65 // signal to stop
66 SignalStop();
67
68 // wait while thread is still alive
69 while (IsAlive)
70 {
71 // simply wait..
72 Thread.Sleep(0);
73
74 // deadlock detection
75 if (watch.Elapsed.TotalSeconds >= timeout)
76 {
77 // force kill all threads as last resort to stop them.
78 // return false to indicate deadlock.
79 Interrupt();
80 return false;
81 }
82 }
83 return true;
84 }
85
86 public bool IsAlive => thread.IsAlive;
87
88 // signal an interrupt in the thread.
89 // this function is very safe to use.
90 // https://stackoverflow.com/questions/5950994/thread-abort-vs-thread-interrupt
91 //
92 // note this does not always kill the thread:
93 // "If this thread is not currently blocked in a wait, sleep, or join
94 // state, it will be interrupted when it next begins to block."
95 // https://docs.microsoft.com/en-us/dotnet/api/system.threading.thread.interrupt?view=net-6.0
96 //
97 // in other words, "while (true) {}" wouldn't throw an interrupt exception.
98 // and that's _okay_. using interrupt is safe & best practice.
99 // => Unity still aborts deadlocked threads on script reload.
100 // => and we catch + warn on AbortException.
101 public void Interrupt() => thread.Interrupt();
102
103 // thread constructor needs callbacks.
104 // always define them, and make them call actions.
105 // those can be set at any time.
106 void OnInit() => Init?.Invoke();
107 void OnTick() => Tick?.Invoke();
108 void OnCleanup() => Cleanup?.Invoke();
109
110 // guarded wrapper for thread code.
111 // catches exceptions which would otherwise be silent.
112 // shows in Unity profiler.
113 // etc.
114 public void Guard(string identifier)
115 {
116 try
117 {
118 // log when work begins = thread starts.
119 // very important for debugging threads.
120 Debug.Log($"{identifier}: started.");
121
122 // show this thread in Unity profiler
123 Profiler.BeginThreadProfiling("Mirror Worker Threads", $"{identifier}");
124
125 // run init once
126 OnInit();
127
128 // run thread func while active
129 while (active)
130 {
131 OnTick();
132 }
133 }
134 // Thread.Interrupt() will gracefully raise a InterruptedException.
135 catch (ThreadInterruptedException)
136 {
137 Debug.Log($"{identifier}: interrupted. That's okay.");
138 }
139 // Unity domain reload will cause a ThreadAbortException.
140 // for example, when saving a changed script while in play mode.
141 catch (ThreadAbortException)
142 {
143 Debug.LogWarning($"{identifier}: aborted. This may happen after domain reload. That's okay.");
144 }
145 catch (Exception e)
146 {
147 Debug.LogException(e);
148 }
149 finally
150 {
151 // run cleanup (if any)
152 active = false;
153 OnCleanup();
154
155 // remove this thread from Unity profiler
156 Profiler.EndThreadProfiling();
157
158 // log when work ends = thread terminates.
159 // very important for debugging threads.
160 // 'finally' to log no matter what (even if exceptions)
161 Debug.Log($"{identifier}: ended.");
162 }
163 }
164 }
