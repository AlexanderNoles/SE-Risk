112 public override void OnSerializeDelta(NetworkWriter writer)
113 {
114 // write all the queued up changes
115 writer.WriteUInt((uint)changes.Count);
116
117 for (int i = 0; i < changes.Count; i++)

119 Change change = changes[i];
120 writer.WriteByte((byte)change.operation);
121
122 switch (change.operation)

124 case Operation.OP_ADD:

128 case Operation.OP_CLEAR:
129 break;
130
131 case Operation.OP_REMOVEAT:
132 writer.WriteUInt((uint)change.index);
133 break;
134
135 case Operation.OP_INSERT:
136 case Operation.OP_SET:
137 writer.WriteUInt((uint)change.index);

164 public override void OnDeserializeDelta(NetworkReader reader)
165 {
166 int changesCount = (int)reader.ReadUInt();
167
168 for (int i = 0; i < changesCount; i++)
169 {
170 Operation operation = (Operation)reader.ReadByte();
171
172 // apply the operation only if it is a new change
173 // that we have not applied yet
174 bool apply = changesAhead == 0;
175 int index = 0;
176 T oldItem = default;
177 T newItem = default;
178
179 switch (operation)
180 {
181 case Operation.OP_ADD:
182 newItem = reader.Read<T>();
183 if (apply)
184 {
185 index = objects.Count;
186 objects.Add(newItem);
187 // add dirty + changes.
188 // ClientToServer needs to set dirty in server OnDeserialize.
189 // no access check: server OnDeserialize can always
190 // write, even for ClientToServer (for broadcasting).
191 AddOperation(Operation.OP_ADD, objects.Count - 1, default, newItem, false);
192 }
193 break;
194
195 case Operation.OP_CLEAR:
196 if (apply)
197 {
198 objects.Clear();
199 // add dirty + changes.
200 // ClientToServer needs to set dirty in server OnDeserialize.
201 // no access check: server OnDeserialize can always
202 // write, even for ClientToServer (for broadcasting).
203 AddOperation(Operation.OP_CLEAR, 0, default, default, false);
204 }
205 break;
206
207 case Operation.OP_INSERT:
208 index = (int)reader.ReadUInt();
209 newItem = reader.Read<T>();
210 if (apply)
211 {
212 objects.Insert(index, newItem);
213 // add dirty + changes.
214 // ClientToServer needs to set dirty in server OnDeserialize.
215 // no access check: server OnDeserialize can always
216 // write, even for ClientToServer (for broadcasting).
217 AddOperation(Operation.OP_INSERT, index, default, newItem, false);
218 }
219 break;
220
221 case Operation.OP_REMOVEAT:
222 index = (int)reader.ReadUInt();
223 if (apply)
224 {
225 oldItem = objects[index];
226 objects.RemoveAt(index);
227 // add dirty + changes.
228 // ClientToServer needs to set dirty in server OnDeserialize.
229 // no access check: server OnDeserialize can always
230 // write, even for ClientToServer (for broadcasting).
231 AddOperation(Operation.OP_REMOVEAT, index, oldItem, default, false);
232 }
233 break;
234
235 case Operation.OP_SET:
236 index = (int)reader.ReadUInt();
237 newItem = reader.Read<T>();
238 if (apply)
239 {
240 oldItem = objects[index];
241 objects[index] = newItem;
242 // add dirty + changes.
243 // ClientToServer needs to set dirty in server OnDeserialize.
244 // no access check: server OnDeserialize can always
245 // write, even for ClientToServer (for broadcasting).
246 AddOperation(Operation.OP_SET, index, oldItem, newItem, false);
247 }
248 break;
249 }
250
251 if (!apply)
252 {
253 // we just skipped this change
254 changesAhead--;
255 }
256 }
