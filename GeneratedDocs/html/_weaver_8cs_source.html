43 // -> so, append the signature to the generated method name,

45 // Example:
46 // RpcTeleport(Vector3 position) -> InvokeUserCode_RpcTeleport__Vector3()

49 public static string GenerateMethodName(string initialPrefix, MethodDefinition md)
50 {
51 initialPrefix += md.Name;
52
53 for (int i = 0; i < md.Parameters.Count; ++i)
54 {
55 // with __ so it's more obvious that this is the parameter suffix.
56 // otherwise RpcTest(int) => RpcTestInt(int) which is not obvious.
57 initialPrefix += $"__{md.Parameters[i].ParameterType.Name}";
58 }

69 bool WeaveNetworkBehavior(TypeDefinition td)
70 {
71 if (!td.IsClass)
72 return false;
73
74 if (!td.IsDerivedFrom<NetworkBehaviour>())
75 {
76 if (td.IsDerivedFrom<UnityEngine.MonoBehaviour>())
77 MonoBehaviourProcessor.Process(Log, td, ref WeavingFailed);
78 return false;
79 }
80
81 // process this and base classes from parent to child order
82
83 List<TypeDefinition> behaviourClasses = new List<TypeDefinition>();
84
85 TypeDefinition parent = td;
86 while (parent != null)
87 {
88 if (parent.Is<NetworkBehaviour>())
89 {
90 break;
91 }
92
93 try
94 {
95 behaviourClasses.Insert(0, parent);
96 parent = parent.BaseType.Resolve();
97 }
98 catch (AssemblyResolutionException)
99 {
100 // this can happen for plugins.
101 //Console.WriteLine("AssemblyResolutionException: "+ ex.ToString());
102 break;
103 }
104 }
105
106 bool modified = false;
107 foreach (TypeDefinition behaviour in behaviourClasses)
108 {
109 modified |= new NetworkBehaviourProcessor(CurrentAssembly, weaverTypes, syncVarAccessLists, writers, readers, Log, behaviour).Process(ref WeavingFailed);
110 }
111 return modified;
112 }

114 bool WeaveModule(ModuleDefinition moduleDefinition)
115 {
116 bool modified = false;
117
118 Stopwatch watch = Stopwatch.StartNew();
119 watch.Start();
120
121 // ModuleDefinition.Types only finds top level types.
122 // GetAllTypes recursively finds all nested types as well.
123 // fixes nested types not being weaved, for example:
124 // class Parent { // ModuleDefinition.Types finds this
125 // class Child { // .Types.NestedTypes finds this
126 // class GrandChild {} // only GetAllTypes finds this too
127 // }
128 // }
129 // note this is not about inheritance, only about type definitions.
130 // see test: NetworkBehaviourTests.DeeplyNested()
131 foreach (TypeDefinition td in moduleDefinition.GetAllTypes())
132 {
133 if (td.IsClass && td.BaseType.CanBeResolved())
134 {
135 modified |= WeaveNetworkBehavior(td);
136 modified |= ServerClientAttributeProcessor.Process(weaverTypes, Log, td, ref WeavingFailed);
137 }
138 }
139
140 watch.Stop();
141 Console.WriteLine($"Weave behaviours and messages took {watch.ElapsedMilliseconds} milliseconds");
142
143 return modified;
144 }

165 // Weave takes an AssemblyDefinition to be compatible with both old and
166 // new weavers:
167 // * old takes a filepath, new takes a in-memory byte[]
168 // * old uses DefaultAssemblyResolver with added dependencies paths,
169 // new uses ...?

172 // => resolver: useful in case we need to resolve any of the assembly's
173 // assembly.MainModule.AssemblyReferences.
174 // -> we can resolve ANY of them given that the resolver
175 // works properly (need custom one for ILPostProcessor)
176 // -> IMPORTANT: .Resolve() takes an AssemblyNameReference.
177 // those from assembly.MainModule.AssemblyReferences are
178 // guaranteed to be resolve-able.

180 // would not be guaranteed to be resolve-able because
181 // for ILPostProcessor we can't assume where Mirror.dll

183 public bool Weave(AssemblyDefinition assembly, IAssemblyResolver resolver, out bool modified)
184 {
185 WeavingFailed = false;
186 modified = false;
187 try
188 {
189 CurrentAssembly = assembly;
190
191 // fix "No writer found for ..." error
192 // https://github.com/vis2k/Mirror/issues/2579
193 // -> when restarting Unity, weaver would try to weave a DLL
194 // again
195 // -> resulting in two GeneratedNetworkCode classes (see ILSpy)
196 // -> the second one wouldn't have all the writer types setup
197 if (CurrentAssembly.MainModule.ContainsClass(GeneratedCodeNamespace, GeneratedCodeClassName))
198 {
199 //Log.Warning($"Weaver: skipping {CurrentAssembly.Name} because already weaved");
200 return true;
201 }
202
203 weaverTypes = new WeaverTypes(CurrentAssembly, Log, ref WeavingFailed);
204
205 // weaverTypes are needed for CreateGeneratedCodeClass
206 CreateGeneratedCodeClass();
207
208 // WeaverList depends on WeaverTypes setup because it uses Import
209 syncVarAccessLists = new SyncVarAccessLists();
210
211 // initialize readers & writers with this assembly.
212 // we need to do this in every Process() call.
213 // otherwise we would get
214 // "System.ArgumentException: Member ... is declared in another module and needs to be imported"
215 // errors when still using the previous module's reader/writer funcs.
216 writers = new Writers(CurrentAssembly, weaverTypes, GeneratedCodeClass, Log);
217 readers = new Readers(CurrentAssembly, weaverTypes, GeneratedCodeClass, Log);
218
219 Stopwatch rwstopwatch = Stopwatch.StartNew();
220 // Need to track modified from ReaderWriterProcessor too because it could find custom read/write functions or create functions for NetworkMessages
221 modified = ReaderWriterProcessor.Process(CurrentAssembly, resolver, Log, writers, readers, ref WeavingFailed);
222 rwstopwatch.Stop();
223 Console.WriteLine($"Find all reader and writers took {rwstopwatch.ElapsedMilliseconds} milliseconds");
224
225 ModuleDefinition moduleDefinition = CurrentAssembly.MainModule;
226 Console.WriteLine($"Script Module: {moduleDefinition.Name}");
227
228 modified |= WeaveModule(moduleDefinition);
229
230 if (WeavingFailed)
231 {
232 return false;
233 }
234
235 if (modified)
236 {
237 SyncVarAttributeAccessReplacer.Process(Log, moduleDefinition, syncVarAccessLists);
238
239 // add class that holds read/write functions
240 moduleDefinition.Types.Add(GeneratedCodeClass);

242 ReaderWriterProcessor.InitializeReaderAndWriters(CurrentAssembly, weaverTypes, writers, readers, GeneratedCodeClass);
243
244 // DO NOT WRITE here.
245 // CompilationFinishedHook writes to the file.
246 // ILPostProcessor writes to in-memory assembly.
247 // it depends on the caller.

258 }
