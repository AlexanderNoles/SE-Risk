1// our ideal update looks like this:
2// transport.process_incoming()
3// update_world()
4// transport.process_outgoing()
5//
6// this way we avoid unnecessary latency for low-ish server tick rates.
7// for example, if we were to use this tick:
8// transport.process_incoming/outgoing()
9// update_world()
10//
11// then anything sent in update_world wouldn't be actually sent out by the
12// transport until the next frame. if server runs at 60Hz, then this can add
13// 16ms latency for every single packet.
14//
15// => instead we process incoming, update world, process_outgoing in the same
16// frame. it's more clear (no race conditions) and lower latency.
17// => we need to add custom Update functions to the Unity engine:
18// NetworkEarlyUpdate before Update()/FixedUpdate()
19// NetworkLateUpdate after LateUpdate()
20// this way the user can update the world in Update/FixedUpdate/LateUpdate
21// and networking still runs before/after those functions no matter what!
22// => see also: https://docs.unity3d.com/Manual/ExecutionOrder.html
23// => update order:
24// * we add to the end of EarlyUpdate so it runs after any Unity initializations
25// * we add to the end of PreLateUpdate so it runs after LateUpdate(). adding
26// to the beginning of PostLateUpdate doesn't actually work.

54 internal static int FindPlayerLoopEntryIndex(PlayerLoopSystem.UpdateFunction function, PlayerLoopSystem playerLoop, Type playerLoopSystemType)
55 {
56 // did we find the type? e.g. EarlyUpdate/PreLateUpdate/etc.
57 if (playerLoop.type == playerLoopSystemType)
58 return Array.FindIndex(playerLoop.subSystemList, (elem => elem.updateDelegate == function));

72 // MODIFIED AddSystemToPlayerLoopList from Unity.Entities.ScriptBehaviourUpdateOrder (ECS)
73 //
74 // => adds an update function to the Unity internal update type.
75 // => Unity has different update loops:
76 // https://medium.com/@thebeardphantom/unity-2018-and-playerloop-5c46a12a677
77 // EarlyUpdate
78 // FixedUpdate
79 // PreUpdate
80 // Update
81 // PreLateUpdate
82 // PostLateUpdate
83 //
84 // function: the custom update function to add
85 // IMPORTANT: according to a comment in Unity.Entities.ScriptBehaviourUpdateOrder,
86 // the UpdateFunction can not be virtual because
87 // Mono 4.6 has problems invoking virtual methods
88 // as delegates from native!
89 // ownerType: the .type to fill in so it's obvious who the new function
90 // belongs to. seems to be mostly for debugging. pass any.
91 // addMode: prepend or append to update list
92 internal static bool AddToPlayerLoop(PlayerLoopSystem.UpdateFunction function, Type ownerType, ref PlayerLoopSystem playerLoop, Type playerLoopSystemType, AddMode addMode)
93 {
94 // did we find the type? e.g. EarlyUpdate/PreLateUpdate/etc.
95 if (playerLoop.type == playerLoopSystemType)
96 {
97 // debugging
98 //Debug.Log($"Found playerLoop of type {playerLoop.type} with {playerLoop.subSystemList.Length} Functions:");
99 //foreach (PlayerLoopSystem sys in playerLoop.subSystemList)
100 // Debug.Log($" ->{sys.type}");
101
102 // make sure the function wasn't added yet.
103 // with domain reload disabled, it would otherwise be added twice:
104 // fixes: https://github.com/MirrorNetworking/Mirror/issues/3392
105 if (Array.FindIndex(playerLoop.subSystemList, (s => s.updateDelegate == function)) != -1)
106 {
107 // loop contains the function, so return true.
108 return true;
109 }
110
111 // resize & expand subSystemList to fit one more entry
112 int oldListLength = (playerLoop.subSystemList != null) ? playerLoop.subSystemList.Length : 0;
113 Array.Resize(ref playerLoop.subSystemList, oldListLength + 1);
114
115 // IMPORTANT: always insert a FRESH PlayerLoopSystem!
116 // We CAN NOT resize and then OVERWRITE an entry's type/loop.
117 // => PlayerLoopSystem has native IntPtr loop members
118 // => forgetting to clear those would cause undefined behaviour!
119 // see also: https://github.com/vis2k/Mirror/pull/2652
120 PlayerLoopSystem system = new PlayerLoopSystem {
121 type = ownerType,
122 updateDelegate = function
123 };
124
125 // prepend our custom loop to the beginning
126 if (addMode == AddMode.Beginning)
127 {
128 // shift to the right, write into first array element
129 Array.Copy(playerLoop.subSystemList, 0, playerLoop.subSystemList, 1, playerLoop.subSystemList.Length - 1);
130 playerLoop.subSystemList[0] = system;
131 }
132 // append our custom loop to the end
133 else if (addMode == AddMode.End)
134 {
135 // simply write into last array element
136 playerLoop.subSystemList[oldListLength] = system;
137 }
138
139 // debugging
140 //Debug.Log($"New playerLoop of type {playerLoop.type} with {playerLoop.subSystemList.Length} Functions:");
141 //foreach (PlayerLoopSystem sys in playerLoop.subSystemList)
142 // Debug.Log($" ->{sys.type}");
143
144 return true;
145 }
146
147 // recursively keep looking
148 if (playerLoop.subSystemList != null)
149 {
150 for (int i = 0; i < playerLoop.subSystemList.Length; ++i)
151 {
152 if (AddToPlayerLoop(function, ownerType, ref playerLoop.subSystemList[i], playerLoopSystemType, addMode))
153 return true;
154 }
155 }
156 return false;
157 }

160 [RuntimeInitializeOnLoadMethod(RuntimeInitializeLoadType.BeforeSceneLoad)]
161 static void RuntimeInitializeOnLoad()
162 {
163 //Debug.Log("Mirror: adding Network[Early/Late]Update to Unity...");
164
165 // get loop
166 // 2019 has GetCURRENTPlayerLoop which is safe to use without
167 // breaking other custom system's custom loops.
168 // see also: https://github.com/vis2k/Mirror/pull/2627/files
169 PlayerLoopSystem playerLoop = PlayerLoop.GetCurrentPlayerLoop();
170
171 // add NetworkEarlyUpdate to the end of EarlyUpdate so it runs after
172 // any Unity initializations but before the first Update/FixedUpdate
173 AddToPlayerLoop(NetworkEarlyUpdate, typeof(NetworkLoop), ref playerLoop, typeof(EarlyUpdate), AddMode.End);
174
175 // add NetworkLateUpdate to the end of PreLateUpdate so it runs after
176 // LateUpdate(). adding to the beginning of PostLateUpdate doesn't
177 // actually work.
178 AddToPlayerLoop(NetworkLateUpdate, typeof(NetworkLoop), ref playerLoop, typeof(PreLateUpdate), AddMode.End);
179
180 // set the new loop
181 PlayerLoop.SetPlayerLoop(playerLoop);
182 }
