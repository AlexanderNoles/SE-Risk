222 // max message size
223 public override int GetMaxPacketSize(int channelId = Channels.Reliable)
224 {
225 // switch to kcp channel.
226 // unreliable or reliable.
227 // default to reliable just to be sure.
228 switch (channelId)
229 {
230 case Channels.Unreliable:
231 return KcpPeer.UnreliableMaxMessageSize(config.Mtu);
232 default:
233 return KcpPeer.ReliableMaxMessageSize(config.Mtu, ReceiveWindowSize);
234 }
235 }
236
237 // kcp reliable channel max packet size is MTU * WND_RCV
238 // this allows 144kb messages. but due to head of line blocking, all
239 // other messages would have to wait until the maxed size one is
240 // delivered. batching 144kb messages each time would be EXTREMELY slow
241 // and fill the send queue nearly immediately when using it over the
242 // network.
243 // => instead we always use MTU sized batches.
244 // => people can still send maxed size if needed.
245 public override int GetBatchThreshold(int channelId) =>
246 KcpPeer.UnreliableMaxMessageSize(config.Mtu);
