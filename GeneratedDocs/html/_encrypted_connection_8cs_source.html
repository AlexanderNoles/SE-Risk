169 public void OnReceiveRaw(ArraySegment<byte> data, int channel)
170 {
171 if (data.Count < 1)
172 {
173 _error(TransportError.Unexpected, "Received empty packet");
174 return;
175 }
176
177 using (NetworkReaderPooled reader = NetworkReaderPool.Get(data))
178 {
179 OpCodes opcode = (OpCodes)reader.ReadByte();
180 switch (opcode)
181 {
182 case OpCodes.Data:
183 // first sender ready is implicit when data is received
184 if (_sendsFirst && _state == State.WaitingHandshakeReply)
185 {
186 SetReady();
187 }
188 else if (!IsReady)
189 {
190 _error(TransportError.Unexpected, "Unexpected data while not ready.");
191 }
192
193 if (reader.Remaining < Overhead)
194 {
195 _error(TransportError.Unexpected, "received data packet smaller than metadata size");
196 return;
197 }
198
199 ArraySegment<byte> ciphertext = reader.ReadBytesSegment(reader.Remaining - NonceSize);
200 reader.ReadBytes(ReceiveNonce, NonceSize);
201
202 Profiler.BeginSample("EncryptedConnection.Decrypt");
203 ArraySegment<byte> plaintext = Decrypt(ciphertext);
204 Profiler.EndSample();
205 if (plaintext.Count == 0)
206 {
207 // error
208 return;
209 }
210 _receive(plaintext, channel);
211 break;
212 case OpCodes.HandshakeStart:
213 if (_sendsFirst)
214 {
215 _error(TransportError.Unexpected, "Received HandshakeStart packet, we don't expect this.");
216 return;
217 }
218
219 if (_state == State.WaitingHandshakeReply)
220 {
221 // this is fine, packets may arrive out of order
222 return;
223 }
224
225 _state = State.WaitingHandshakeReply;
226 ResetTimeouts();
227 CompleteExchange(reader.ReadBytesSegment(reader.Remaining), _hkdfSalt);
228 SendHandshakeAndPubKey(OpCodes.HandshakeAck);
229 break;
230 case OpCodes.HandshakeAck:
231 if (!_sendsFirst)
232 {
233 _error(TransportError.Unexpected, "Received HandshakeAck packet, we don't expect this.");
234 return;
235 }
236
237 if (IsReady)
238 {
239 // this is fine, packets may arrive out of order
240 return;
241 }
242
243 if (_state == State.WaitingHandshakeReply)
244 {
245 // this is fine, packets may arrive out of order
246 return;
247 }
248
249
250 _state = State.WaitingHandshakeReply;
251 ResetTimeouts();
252 reader.ReadBytes(_tmpRemoteSaltBuffer, HkdfSaltSize);
253 CompleteExchange(reader.ReadBytesSegment(reader.Remaining), _tmpRemoteSaltBuffer);
254 SendHandshakeFin();
255 break;
256 case OpCodes.HandshakeFin:
257 if (_sendsFirst)
258 {
259 _error(TransportError.Unexpected, "Received HandshakeFin packet, we don't expect this.");
260 return;
261 }
262
263 if (IsReady)
264 {
265 // this is fine, packets may arrive out of order
266 return;
267 }
268
269 if (_state != State.WaitingHandshakeReply)
270 {
271 _error(TransportError.Unexpected,
272 "Received HandshakeFin packet, we didn't expect this yet.");
273 return;
274 }
275
276 SetReady();
277
278 break;
279 default:
280 _error(TransportError.InvalidReceive, $"Unhandled opcode {(byte)opcode:x}");
281 break;
282 }
283 }

321 private ArraySegment<byte> Encrypt(ArraySegment<byte> plaintext)
322 {
323 if (plaintext.Count == 0)
324 {
325 // Invalid
326 return new ArraySegment<byte>();
327 }
328 // Need to make the nonce unique again before encrypting another message
329 UpdateNonce();
330 // Re-initialize the cipher with our cached parameters
331 Cipher.Init(true, _cipherParametersEncrypt);
332
333 // Calculate the expected output size, this should always be input size + mac size
334 int outSize = Cipher.GetOutputSize(plaintext.Count);
335#if UNITY_EDITOR
336 // expecting the outSize to be input size + MacSize
337 if (outSize != plaintext.Count + MacSizeBytes)
338 {
339 throw new Exception($"Encrypt: Unexpected output size (Expected {plaintext.Count + MacSizeBytes}, got {outSize}");
340 }
341#endif
342 // Resize the static buffer to fit
343 EnsureSize(ref _tmpCryptBuffer, outSize);
344 int resultLen;
345 try
346 {
347 // Run the plain text through the cipher, ProcessBytes will only process full blocks
348 resultLen =
349 Cipher.ProcessBytes(plaintext.Array, plaintext.Offset, plaintext.Count, _tmpCryptBuffer, 0);
350 // Then run any potentially remaining partial blocks through with DoFinal (and calculate the mac)
351 resultLen += Cipher.DoFinal(_tmpCryptBuffer, resultLen);
352 }
353 // catch all Exception's since BouncyCastle is fairly noisy with both standard and their own exception types
354 //
355 catch (Exception e)
356 {
357 _error(TransportError.Unexpected, $"Unexpected exception while encrypting {e.GetType()}: {e.Message}");
358 return new ArraySegment<byte>();
359 }
360#if UNITY_EDITOR
361 // expecting the result length to match the previously calculated input size + MacSize
362 if (resultLen != outSize)
363 {
364 throw new Exception($"Encrypt: resultLen did not match outSize (expected {outSize}, got {resultLen})");
365 }
366#endif
367 return new ArraySegment<byte>(_tmpCryptBuffer, 0, resultLen);
368 }

370 private ArraySegment<byte> Decrypt(ArraySegment<byte> ciphertext)
371 {
372 if (ciphertext.Count <= MacSizeBytes)
373 {
374 _error(TransportError.Unexpected, $"Received too short data packet (min {{MacSizeBytes + 1}}, got {ciphertext.Count})");
375 // Invalid
376 return new ArraySegment<byte>();
377 }
378 // Re-initialize the cipher with our cached parameters
379 Cipher.Init(false, _cipherParametersDecrypt);
380
381 // Calculate the expected output size, this should always be input size - mac size
382 int outSize = Cipher.GetOutputSize(ciphertext.Count);
383#if UNITY_EDITOR
384 // expecting the outSize to be input size - MacSize
385 if (outSize != ciphertext.Count - MacSizeBytes)
386 {
387 throw new Exception($"Decrypt: Unexpected output size (Expected {ciphertext.Count - MacSizeBytes}, got {outSize}");
388 }
389#endif
390 // Resize the static buffer to fit
391 EnsureSize(ref _tmpCryptBuffer, outSize);
392 int resultLen;
393 try
394 {
395 // Run the ciphertext through the cipher, ProcessBytes will only process full blocks
396 resultLen =
397 Cipher.ProcessBytes(ciphertext.Array, ciphertext.Offset, ciphertext.Count, _tmpCryptBuffer, 0);
398 // Then run any potentially remaining partial blocks through with DoFinal (and calculate/check the mac)
399 resultLen += Cipher.DoFinal(_tmpCryptBuffer, resultLen);
400 }
401 // catch all Exception's since BouncyCastle is fairly noisy with both standard and their own exception types
402 catch (Exception e)
403 {
404 _error(TransportError.Unexpected, $"Unexpected exception while decrypting {e.GetType()}: {e.Message}. This usually signifies corrupt data");
405 return new ArraySegment<byte>();
406 }
407#if UNITY_EDITOR
408 // expecting the result length to match the previously calculated input size + MacSize
409 if (resultLen != outSize)
410 {
411 throw new Exception($"Decrypt: resultLen did not match outSize (expected {outSize}, got {resultLen})");
412 }
413#endif
414 return new ArraySegment<byte>(_tmpCryptBuffer, 0, resultLen);

464 private void CompleteExchange(ArraySegment<byte> remotePubKeyRaw, byte[] salt)
465 {
466 AsymmetricKeyParameter remotePubKey;
467 try
468 {
469 remotePubKey = EncryptionCredentials.DeserializePublicKey(remotePubKeyRaw);
470 }
471 catch (Exception e)
472 {
473 _error(TransportError.Unexpected, $"Failed to deserialize public key of remote. {e.GetType()}: {e.Message}");
474 return;
475 }
476
477 if (_validateRemoteKey != null)
478 {
479 PubKeyInfo info = new PubKeyInfo
480 {
481 Fingerprint = EncryptionCredentials.PubKeyFingerprint(remotePubKeyRaw),
482 Serialized = remotePubKeyRaw,
483 Key = remotePubKey
484 };
485 if (!_validateRemoteKey(info))
486 {
487 _error(TransportError.Unexpected, $"Remote public key (fingerprint: {info.Fingerprint}) failed validation. ");
488 return;
489 }
490 }
491
492 // Calculate a common symmetric key from our private key and the remotes public key
493 // This gives us the same key on the other side, with our public key and their remote
494 // It's like magic, but with math!
495 ECDHBasicAgreement ecdh = new ECDHBasicAgreement();
496 ecdh.Init(_credentials.PrivateKey);
497 byte[] sharedSecret;
498 try
499 {
500 sharedSecret = ecdh.CalculateAgreement(remotePubKey).ToByteArrayUnsigned();
501 }
502 catch
503 (Exception e)
504 {
505 _error(TransportError.Unexpected, $"Failed to calculate the ECDH key exchange. {e.GetType()}: {e.Message}");
506 return;
507 }
508
509 if (salt.Length != HkdfSaltSize)
510 {
511 _error(TransportError.Unexpected, $"Salt is expected to be {HkdfSaltSize} bytes long, got {salt.Length}.");
512 return;
513 }
514
515 Hkdf.Init(new HkdfParameters(sharedSecret, salt, HkdfInfo));
516
517 // Allocate a buffer for the output key
518 byte[] keyRaw = new byte[KeyLength];
519
520 // Generate the output keying material
521 Hkdf.GenerateBytes(keyRaw, 0, keyRaw.Length);
522
523 KeyParameter key = new KeyParameter(keyRaw);
524
525 // generate a starting nonce
526 _nonce = GenerateSecureBytes(NonceSize);
527
528 // we pass in the nonce array once (as it's stored by reference) so we can cache the AeadParameters instance
529 // instead of creating a new one each encrypt/decrypt
530 _cipherParametersEncrypt = new AeadParameters(key, MacSizeBits, _nonce);
531 _cipherParametersDecrypt = new AeadParameters(key, MacSizeBits, ReceiveNonce);
532 }

537 public void TickNonReady(double time)
538 {
539 if (IsReady)
540 {
541 return;
542 }
543
544 // Timeout reset
545 if (_handshakeTimeout == 0)
546 {
547 _handshakeTimeout = time + DurationTimeout;
548 }
549 else if (time > _handshakeTimeout)
550 {
551 _error?.Invoke(TransportError.Timeout, $"Timed out during {_state}, this probably just means the other side went away which is fine.");
552 return;
553 }
554
555 // Timeout reset
556 if (_nextHandshakeResend < 0)
557 {
558 _nextHandshakeResend = time + DurationResend;
559 return;
560 }
561
562 if (time < _nextHandshakeResend)
563 {
564 // Resend isn't due yet
565 return;
566 }
567
568 _nextHandshakeResend = time + DurationResend;
569 switch (_state)
570 {
571 case State.WaitingHandshake:
572 if (_sendsFirst)
573 {
574 SendHandshakeAndPubKey(OpCodes.HandshakeStart);
575 }
576
577 break;
578 case State.WaitingHandshakeReply:
579 if (_sendsFirst)
580 {
581 SendHandshakeFin();
582 }
583 else
584 {
585 SendHandshakeAndPubKey(OpCodes.HandshakeAck);
586 }
587
588 break;
589 case State.Ready: // IsReady is checked above & early-returned
590 default:
591 throw new ArgumentOutOfRangeException();
592 }
593 }
