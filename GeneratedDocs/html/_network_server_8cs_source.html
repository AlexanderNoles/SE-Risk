297 static void OnCommandMessage(NetworkConnectionToClient conn, CommandMessage msg, int channelId)
298 {
299 if (!conn.isReady)
300 {
301 // Clients may be set NotReady due to scene change or other game logic by user, e.g. respawning.
302 // Ignore commands that may have been in flight before client received NotReadyMessage message.
303 // Unreliable messages may be out of order, so don't spam warnings for those.
304 if (channelId == Channels.Reliable)
305 {
306 // Attempt to identify the target object, component, and method to narrow down the cause of the error.
307 if (spawned.TryGetValue(msg.netId, out NetworkIdentity netIdentity))
308 if (msg.componentIndex < netIdentity.NetworkBehaviours.Length && netIdentity.NetworkBehaviours[msg.componentIndex] is NetworkBehaviour component)
309 if (RemoteProcedureCalls.GetFunctionMethodName(msg.functionHash, out string methodName))
310 {
311 Debug.LogWarning($"Command {methodName} received for {netIdentity.name} [netId={msg.netId}] component {component.name} [index={msg.componentIndex}] when client not ready.\nThis may be ignored if client intentionally set NotReady.");
312 return;
313 }
314
315 Debug.LogWarning("Command received while client is not ready.\nThis may be ignored if client intentionally set NotReady.");
316 }
317 return;
318 }
319
320 if (!spawned.TryGetValue(msg.netId, out NetworkIdentity identity))
321 {
322 // over reliable channel, commands should always come after spawn.
323 // over unreliable, they might come in before the object was spawned.
324 // for example, NetworkTransform.
325 // let's not spam the console for unreliable out of order messages.
326 if (channelId == Channels.Reliable)
327 Debug.LogWarning($"Spawned object not found when handling Command message {identity.name} netId={msg.netId}");
328 return;
329 }
330
331 // Commands can be for player objects, OR other objects with client-authority
332 // -> so if this connection's controller has a different netId then
333 // only allow the command if clientAuthorityOwner
334 bool requiresAuthority = RemoteProcedureCalls.CommandRequiresAuthority(msg.functionHash);
335 if (requiresAuthority && identity.connectionToClient != conn)
336 {
337 // Attempt to identify the component and method to narrow down the cause of the error.
338 if (msg.componentIndex < identity.NetworkBehaviours.Length && identity.NetworkBehaviours[msg.componentIndex] is NetworkBehaviour component)
339 if (RemoteProcedureCalls.GetFunctionMethodName(msg.functionHash, out string methodName))
340 {
341 Debug.LogWarning($"Command {methodName} received for {identity.name} [netId={msg.netId}] component {component.name} [index={msg.componentIndex}] without authority");
342 return;
343 }
344
345 Debug.LogWarning($"Command received for {identity.name} [netId={msg.netId}] without authority");
346 return;
347 }
348
349 // Debug.Log($"OnCommandMessage for netId:{msg.netId} conn:{conn}");
350
351 using (NetworkReaderPooled networkReader = NetworkReaderPool.Get(msg.payload))
352 identity.HandleRemoteCall(msg.componentIndex, msg.functionHash, RemoteCallType.Command, networkReader, conn);
353 }

478 public static void SendToAll<T>(T message, int channelId = Channels.Reliable, bool sendToReadyOnly = false)
479 where T : struct, NetworkMessage
480 {
481 if (!active)
482 {
483 Debug.LogWarning("Can not send using NetworkServer.SendToAll<T>(T msg) because NetworkServer is not active");
484 return;
485 }
486
487 // Debug.Log($"Server.SendToAll {typeof(T)}");
488 using (NetworkWriterPooled writer = NetworkWriterPool.Get())
489 {
490 // pack message only once
491 NetworkMessages.Pack(message, writer);
492 ArraySegment<byte> segment = writer.ToArraySegment();
493
494 // validate packet size immediately.
495 // we know how much can fit into one batch at max.
496 // if it's larger, log an error immediately with the type <T>.
497 // previously we only logged in Update() when processing batches,
498 // but there we don't have type information anymore.
499 int max = NetworkMessages.MaxMessageSize(channelId);
500 if (writer.Position > max)
501 {
502 Debug.LogError($"NetworkServer.SendToAll: message of type {typeof(T)} with a size of {writer.Position} bytes is larger than the max allowed message size in one batch: {max}.\nThe message was dropped, please make it smaller.");
503 return;
504 }
505
506 // filter and then send to all internet connections at once
507 // -> makes code more complicated, but is HIGHLY worth it to
508 // avoid allocations, allow for multicast, etc.
509 int count = 0;
510 foreach (NetworkConnectionToClient conn in connections.Values)
511 {
512 if (sendToReadyOnly && !conn.isReady)
513 continue;
514
515 count++;
516 conn.Send(segment, channelId);
517 }
518
519 NetworkDiagnostics.OnSend(message, channelId, segment.Count, count);
520 }
521 }

675 static bool UnpackAndInvoke(NetworkConnectionToClient connection, NetworkReader reader, int channelId)
676 {
677 if (NetworkMessages.UnpackId(reader, out ushort msgType))
678 {
679 // try to invoke the handler for that message
680 if (handlers.TryGetValue(msgType, out NetworkMessageDelegate handler))
681 {
682 handler.Invoke(connection, reader, channelId);
683 connection.lastMessageTime = Time.time;
684 return true;
685 }
686 else
687 {
688 // message in a batch are NOT length prefixed to save bandwidth.
689 // every message needs to be handled and read until the end.
690 // otherwise it would overlap into the next message.
691 // => need to warn and disconnect to avoid undefined behaviour.
692 // => WARNING, not error. can happen if attacker sends random data.
693 Debug.LogWarning($"Unknown message id: {msgType} for connection: {connection}. This can happen if no handler was registered for this message.");
694 // simply return false. caller is responsible for disconnecting.
695 //connection.Disconnect();
696 return false;
697 }
698 }
699 else
700 {
701 // => WARNING, not error. can happen if attacker sends random data.
702 Debug.LogWarning($"Invalid message header for connection: {connection}.");
703 // simply return false. caller is responsible for disconnecting.
704 //connection.Disconnect();
705 return false;
706 }

709 // called by transport
710 internal static void OnTransportData(int connectionId, ArraySegment<byte> data, int channelId)
711 {
712 if (connections.TryGetValue(connectionId, out NetworkConnectionToClient connection))
713 {
714 // client might batch multiple messages into one packet.
715 // feed it to the Unbatcher.
716 // NOTE: we don't need to associate a channelId because we
717 // always process all messages in the batch.
718 if (!connection.unbatcher.AddBatch(data))
719 {
720 if (exceptionsDisconnect)
721 {
722 Debug.LogError($"NetworkServer: received message from connectionId:{connectionId} was too short (messages should start with message id). Disconnecting.");
723 connection.Disconnect();
724 }
725 else
726 Debug.LogWarning($"NetworkServer: received message from connectionId:{connectionId} was too short (messages should start with message id).");
727
728 return;
729 }
730
731 // process all messages in the batch.
732 // only while NOT loading a scene.
733 // if we get a scene change message, then we need to stop
734 // processing. otherwise we might apply them to the old scene.
735 // => fixes https://github.com/vis2k/Mirror/issues/2651
736 //
737 // NOTE: if scene starts loading, then the rest of the batch
738 // would only be processed when OnTransportData is called
739 // the next time.
740 // => consider moving processing to NetworkEarlyUpdate.
741 while (!isLoadingScene &&
742 connection.unbatcher.GetNextMessage(out ArraySegment<byte> message, out double remoteTimestamp))
743 {
744 using (NetworkReaderPooled reader = NetworkReaderPool.Get(message))
745 {
746 // enough to read at least header size?
747 if (reader.Remaining >= NetworkMessages.IdSize)
748 {
749 // make remoteTimeStamp available to the user
750 connection.remoteTimeStamp = remoteTimestamp;
751
752 // handle message
753 if (!UnpackAndInvoke(connection, reader, channelId))
754 {

756 // -> warning because attackers might send random data
757 // -> messages in a batch aren't length prefixed.

759 // behaviour for every message afterwards.

761 // -> return to avoid the below unbatches.count error.
762 // we already disconnected and handled it.

766 connection.Disconnect();
767 }
768 else
769 Debug.LogWarning($"NetworkServer: failed to unpack and invoke message from connectionId:{connectionId}.");
770
771 return;
772 }
773 }
774 // otherwise disconnect
775 else
776 {
777 if (exceptionsDisconnect)
778 {
779 Debug.LogError($"NetworkServer: received message from connectionId:{connectionId} was too short (messages should start with message id). Disconnecting.");

781 }
782 else
783 Debug.LogWarning($"NetworkServer: received message from connectionId:{connectionId} was too short (messages should start with message id).");

1306 static ArraySegment<byte> CreateSpawnMessagePayload(bool isOwner, NetworkIdentity identity, NetworkWriterPooled ownerWriter, NetworkWriterPooled observersWriter)
1307 {
1308 // Only call SerializeAll if there are NetworkBehaviours
1309 if (identity.NetworkBehaviours.Length == 0)
1310 {
1311 return default;
1312 }
1313
1314 // serialize all components with initialState = true
1315 // (can be null if has none)
1316 identity.SerializeServer(true, ownerWriter, observersWriter);
1317
1318 // convert to ArraySegment to avoid reader allocations
1319 // if nothing was written, .ToArraySegment returns an empty segment.
1320 ArraySegment<byte> ownerSegment = ownerWriter.ToArraySegment();

1322
1323 // use owner segment if 'conn' owns this identity, otherwise
1324 // use observers segment
1325 ArraySegment<byte> payload = isOwner ? ownerSegment : observersSegment;
1326
1327 return payload;
