11 // FakeByte: gather bounds in smaller buckets.
12 // for example, bucket(t0,t1,t2), bucket(t3,t4,t5), ...
13 // instead of removing old bounds t0, t1, ...
14 // we remove a whole bucket every 3 times: bucket(t0,t1,t2)
15 // and when building total bounds, we encapsulate a few larger buckets
16 // instead of many smaller bounds.
17 //
18 // => a bucket is encapsulate(bounds0, bounds1, bounds2) so we don't
19 // need a custom struct, simply reuse bounds but remember that each
20 // entry includes N timestamps.
21 //
22 // => note that simply reducing capture interval is _not_ the same.
23 // we want to capture in detail in case players run in zig-zag.
24 // but still grow larger buckets internally.
25 public class HistoryBounds
26 {
27 // mischa: use MinMaxBounds to avoid Unity Bounds.Encapsulate conversions.
28 readonly int boundsPerBucket;
29 readonly Queue<MinMaxBounds> fullBuckets;
30
31 // full bucket limit. older ones will be removed.
32 readonly int bucketLimit;
33
34 // bucket in progress, contains 0..boundsPerBucket bounds encapsulated.
35 MinMaxBounds? currentBucket;
36 int currentBucketSize;
37
38 // amount of total bounds, including bounds in full buckets + current
39 public int boundsCount { get; private set; }
40
41 // total bounds encapsulating all of the bounds history.
42 // totalMinMax is used for internal calculations.
43 // public total is used for Unity representation.
44 MinMaxBounds totalMinMax;
45 public Bounds total
46 {
47 get
48 {
49 Bounds bounds = new Bounds();
50 bounds.SetMinMax(totalMinMax.min, totalMinMax.max);
51 return bounds;
52 }

55 public HistoryBounds(int boundsLimit, int boundsPerBucket)
56 {
57 // bucketLimit via '/' cuts off remainder.
58 // that's what we want, since we always have a 'currentBucket'.
59 this.boundsPerBucket = boundsPerBucket;
60 this.bucketLimit = (boundsLimit / boundsPerBucket);
61
62 // initialize queue with maximum capacity to avoid runtime resizing
63 // capacity +1 because it makes the code easier if we insert first, and then remove.
64 fullBuckets = new Queue<MinMaxBounds>(bucketLimit + 1);
65 }
66
67 // insert new bounds into history. calculates new total bounds.
68 // Queue.Dequeue() always has the oldest bounds.
69 public void Insert(Bounds bounds)
70 {
71 // convert to MinMax representation for faster .Encapsulate()
72 MinMaxBounds minmax = new MinMaxBounds
73 {
74 min = bounds.min,
75 max = bounds.max
76 };
77
78 // initialize 'total' if not initialized yet.
79 // we don't want to call (0,0).Encapsulate(bounds).
80 if (boundsCount == 0)
81 {
82 totalMinMax = minmax;
83 }
84
85 // add to current bucket:
86 // either initialize new one, or encapsulate into existing one
87 if (currentBucket == null)
88 {
89 currentBucket = minmax;
90 }
91 else
92 {
93 currentBucket.Value.Encapsulate(minmax);
94 }
95
96 // current bucket has one more bounds.
97 // total bounds increased as well.
98 currentBucketSize += 1;
99 boundsCount += 1;
100
101 // always encapsulate into total immediately.
102 // this is free.
103 totalMinMax.Encapsulate(minmax);
104
105 // current bucket full?
106 if (currentBucketSize == boundsPerBucket)
107 {
108 // move it to full buckets
109 fullBuckets.Enqueue(currentBucket.Value);
110 currentBucket = null;
111 currentBucketSize = 0;
112
113 // full bucket capacity reached?
114 if (fullBuckets.Count > bucketLimit)
115 {
116 // remove oldest bucket
117 fullBuckets.Dequeue();
118 boundsCount -= boundsPerBucket;
119
120 // recompute total bounds
121 // instead of iterating N buckets, we iterate N / boundsPerBucket buckets.
122 // TODO technically we could reuse 'currentBucket' before clearing instead of encapsulating again
123 totalMinMax = minmax;
124 foreach (MinMaxBounds bucket in fullBuckets)
125 totalMinMax.Encapsulate(bucket);
126 }
127 }
128 }
129
130 public void Reset()
131 {
132 fullBuckets.Clear();
133 currentBucket = null;
134 currentBucketSize = 0;
135 boundsCount = 0;
136 totalMinMax = new MinMaxBounds();
137 }
138 }
