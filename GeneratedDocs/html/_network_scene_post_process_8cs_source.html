12 public static void OnPostProcessScene()
13 {
14 // find all NetworkIdentities in all scenes
15 // => can't limit it to GetActiveScene() because that wouldn't work
16 // for additive scene loads (the additively loaded scene is never
17 // the active scene)
18 // => ignore DontDestroyOnLoad scene! this avoids weird situations
19 // like in NetworkZones when we destroy the local player and
20 // load another scene afterwards, yet the local player is still
21 // in the FindObjectsOfType result with scene=DontDestroyOnLoad
22 // for some reason
23 // => OfTypeAll so disabled objects are included too
24 // => Unity 2019 returns prefabs here too, so filter them out.
25 IEnumerable<NetworkIdentity> identities = Resources.FindObjectsOfTypeAll<NetworkIdentity>()
26 .Where(identity => identity.gameObject.hideFlags != HideFlags.NotEditable &&
27 identity.gameObject.hideFlags != HideFlags.HideAndDontSave &&
28 identity.gameObject.scene.name != "DontDestroyOnLoad" &&
29 !Utils.IsPrefab(identity.gameObject));
30
31 foreach (NetworkIdentity identity in identities)
32 {
33 // if we had a [ConflictComponent] attribute that would be better than this check.
34 // also there is no context about which scene this is in.
35 if (identity.GetComponent<NetworkManager>() != null)
36 Debug.LogError("NetworkManager has a NetworkIdentity component. This will cause the NetworkManager object to be disabled, so it is not recommended.");
37
38 // not spawned before?
39 // OnPostProcessScene is called after additive scene loads too,
40 // and we don't want to set main scene's objects inactive again
41 if (!identity.isClient && !identity.isServer)
42 {
43 // valid scene object?
44 // otherwise it might be an unopened scene that still has null
45 // sceneIds. builds are interrupted if they contain 0 sceneIds,
46 // but it's still possible that we call LoadScene in Editor
47 // for a previously unopened scene.
48 // (and only do SetActive if this was actually a scene object)
49 if (identity.sceneId != 0)
50 {
51 PrepareSceneObject(identity);
52 }
53 // throwing an exception would only show it for one object
54 // because this function would return afterwards.
55 else
56 {
57 // there are two cases where sceneId == 0:
58 // * if we have a prefab open in the prefab scene
59 // * if an unopened scene needs resaving
60 // show a proper error message in both cases so the user
61 // knows what to do.
62 string path = identity.gameObject.scene.path;
63 if (string.IsNullOrWhiteSpace(path))
64 Debug.LogError($"{identity.name} is currently open in Prefab Edit Mode. Please open the actual scene before launching Mirror.");
65 else
66 Debug.LogError($"Scene {path} needs to be opened and resaved, because the scene object {identity.name} has no valid sceneId yet.");
67
68 // either way we shouldn't continue. nothing good will
69 // happen when trying to launch with invalid sceneIds.
70 EditorApplication.isPlaying = false;
71 }
72 }
73 }
