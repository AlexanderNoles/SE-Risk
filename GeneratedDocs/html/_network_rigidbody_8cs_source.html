154 [Server]
155 void SyncToClients()
156 {
157 // only update if they have changed more than Sensitivity
158
159 Vector3 currentVelocity = syncVelocity ? target.velocity : default;
160 Vector3 currentAngularVelocity = syncAngularVelocity ? target.angularVelocity : default;
161
162 bool velocityChanged = syncVelocity && ((previousValue.velocity - currentVelocity).sqrMagnitude > velocitySensitivity * velocitySensitivity);
163 bool angularVelocityChanged = syncAngularVelocity && ((previousValue.angularVelocity - currentAngularVelocity).sqrMagnitude > angularVelocitySensitivity * angularVelocitySensitivity);
164
165 if (velocityChanged)
166 {
167 velocity = currentVelocity;
168 previousValue.velocity = currentVelocity;
169 }
170
171 if (angularVelocityChanged)
172 {
173 angularVelocity = currentAngularVelocity;
174 previousValue.angularVelocity = currentAngularVelocity;
175 }
176
177 // other rigidbody settings
178 isKinematic = target.isKinematic;
179 useGravity = target.useGravity;
180 drag = target.drag;
181 angularDrag = target.angularDrag;
182 }

200 [Client]
201 void SendVelocity()
202 {
203 double now = NetworkTime.localTime; // Unity 2019 doesn't have Time.timeAsDouble yet
204 if (now < previousValue.nextSyncTime)
205 return;
206
207 Vector3 currentVelocity = syncVelocity ? target.velocity : default;
208 Vector3 currentAngularVelocity = syncAngularVelocity ? target.angularVelocity : default;
209
210 bool velocityChanged = syncVelocity && ((previousValue.velocity - currentVelocity).sqrMagnitude > velocitySensitivity * velocitySensitivity);
211 bool angularVelocityChanged = syncAngularVelocity && ((previousValue.angularVelocity - currentAngularVelocity).sqrMagnitude > angularVelocitySensitivity * angularVelocitySensitivity);
212
213 // if angularVelocity has changed it is likely that velocity has also changed so just sync both values
214 // however if only velocity has changed just send velocity
215 if (angularVelocityChanged)
216 {
217 CmdSendVelocityAndAngular(currentVelocity, currentAngularVelocity);
218 previousValue.velocity = currentVelocity;
219 previousValue.angularVelocity = currentAngularVelocity;
220 }
221 else if (velocityChanged)
222 {
223 CmdSendVelocity(currentVelocity);
224 previousValue.velocity = currentVelocity;
225 }
226
227
228 // only update syncTime if either has changed
229 if (angularVelocityChanged || velocityChanged)
230 previousValue.nextSyncTime = now + syncInterval;
231 }
