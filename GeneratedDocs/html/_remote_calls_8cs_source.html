35 // one lookup for all remote calls.
36 // allows us to easily add more remote call types without duplicating code.
37 // note: do not clear those with [RuntimeInitializeOnLoad]
38 //
39 // IMPORTANT: cmd/rpc functions are identified via **HASHES**.
40 // an index would requires half the bandwidth, but introduces issues
41 // where static constructors are lazily called, so index order isn't
42 // guaranteed. keep hashes to avoid:
43 // https://github.com/vis2k/Mirror/pull/3135
44 // https://github.com/vis2k/Mirror/issues/3138
45 // BUT: 2 byte hash is enough if we check for collisions. that's what we
46 // do for NetworkMessage as well.
47 static readonly Dictionary<ushort, Invoker> remoteCallDelegates = new Dictionary<ushort, Invoker>();
48
49 static bool CheckIfDelegateExists(Type componentType, RemoteCallType remoteCallType, RemoteCallDelegate func, ushort functionHash)
50 {
51 if (remoteCallDelegates.ContainsKey(functionHash))
52 {
53 // something already registered this hash.
54 // it's okay if it was the same function.
55 Invoker oldInvoker = remoteCallDelegates[functionHash];
56 if (oldInvoker.AreEqual(componentType, remoteCallType, func))
57 {
58 return true;
59 }
60
61 // otherwise notify user. there is a rare chance of string
62 // hash collisions.
63 Debug.LogError($"Function {oldInvoker.componentType}.{oldInvoker.function.GetMethodName()} and {componentType}.{func.GetMethodName()} have the same hash. Please rename one of them. To save bandwidth, we only use 2 bytes for the hash, which has a small chance of collisions.");
64 }
65
66 return false;
67 }

69 // pass full function name to avoid ClassA.Func & ClassB.Func collisions
70 internal static ushort RegisterDelegate(Type componentType, string functionFullName, RemoteCallType remoteCallType, RemoteCallDelegate func, bool cmdRequiresAuthority = true)
71 {
72 // type+func so Inventory.RpcUse != Equipment.RpcUse
73 ushort hash = (ushort)(functionFullName.GetStableHashCode() & 0xFFFF);
74
75 if (CheckIfDelegateExists(componentType, remoteCallType, func, hash))
76 return hash;
77
78 remoteCallDelegates[hash] = new Invoker
79 {
80 callType = remoteCallType,
81 componentType = componentType,
82 function = func,
83 cmdRequiresAuthority = cmdRequiresAuthority
84 };
85 return hash;
86 }

115 // note: no need to throw an error if not found.
116 // an attacker might just try to call a cmd with an rpc's hash etc.
117 // returning false is enough.
118 static bool GetInvokerForHash(ushort functionHash, RemoteCallType remoteCallType, out Invoker invoker) =>
119 remoteCallDelegates.TryGetValue(functionHash, out invoker) &&
120 invoker != null &&
121 invoker.callType == remoteCallType;
122
123 // InvokeCmd/Rpc Delegate can all use the same function here
124 internal static bool Invoke(ushort functionHash, RemoteCallType remoteCallType, NetworkReader reader, NetworkBehaviour component, NetworkConnectionToClient senderConnection = null)
125 {
126 // IMPORTANT: we check if the message's componentIndex component is
127 // actually of the right type. prevents attackers trying
128 // to invoke remote calls on wrong components.
129 if (GetInvokerForHash(functionHash, remoteCallType, out Invoker invoker) &&
130 invoker.componentType.IsInstanceOfType(component))
131 {
132 // invoke function on this component
133 invoker.function(component, reader, senderConnection);
134 return true;
135 }
136 return false;
137 }
138
139 // check if the command 'requiresAuthority' which is set in the attribute
140 internal static bool CommandRequiresAuthority(ushort cmdHash) =>
141 GetInvokerForHash(cmdHash, RemoteCallType.Command, out Invoker invoker) &&
142 invoker.cmdRequiresAuthority;

145 public static RemoteCallDelegate GetDelegate(ushort functionHash) =>
