176 public void CheckReadyToBegin()
177 {
178 if (!Utils.IsSceneActive(RoomScene))
179 return;
180
181 int numberOfReadyPlayers = NetworkServer.connections.Count(conn =>
182 conn.Value != null &&
183 conn.Value.identity != null &&
184 conn.Value.identity.TryGetComponent(out NetworkRoomPlayer nrp) &&
185 nrp.readyToBegin);
186
187 bool enoughReadyPlayers = minPlayers <= 0 || numberOfReadyPlayers >= minPlayers;
188 if (enoughReadyPlayers)
189 {
190 pendingPlayers.Clear();
191 allPlayersReady = true;
192 }
193 else
194 allPlayersReady = false;
195 }

223 public override void OnServerDisconnect(NetworkConnectionToClient conn)
224 {
225 if (conn.identity != null)
226 {
227 NetworkRoomPlayer roomPlayer = conn.identity.GetComponent<NetworkRoomPlayer>();
228
229 if (roomPlayer != null)
230 roomSlots.Remove(roomPlayer);
231
232 foreach (NetworkIdentity clientOwnedObject in conn.owned)
233 {
234 roomPlayer = clientOwnedObject.GetComponent<NetworkRoomPlayer>();
235 if (roomPlayer != null)
236 roomSlots.Remove(roomPlayer);
237 }
238 }
239
240 allPlayersReady = false;
241
242 foreach (NetworkRoomPlayer player in roomSlots)
243 {
244 if (player != null)
245 player.GetComponent<NetworkRoomPlayer>().readyToBegin = false;
246 }
247
248 if (Utils.IsSceneActive(RoomScene))
249 RecalculateRoomPlayerIndices();
250
251 OnRoomServerDisconnect(conn);
252 base.OnServerDisconnect(conn);
253
254 if (Utils.IsHeadless())
255 {
256 if (numPlayers < 1)
257 StopServer();
258 }
259 }
