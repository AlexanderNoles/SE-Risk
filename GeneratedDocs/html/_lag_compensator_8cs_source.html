88 // sampling ////////////////////////////////////////////////////////////
89 // sample the sub-tick (=interpolated) history of this object for a hit test.
90 // 'viewer' needs to be the player who fired!
91 // for example, if A fires at B, then call B.Sample(viewer, point, tolerance).
92 [Server]
93 public virtual bool Sample(NetworkConnectionToClient viewer, out Capture3D sample)
94 {
95 // never trust the client: estimate client time instead.
96 // https://developer.valvesoftware.com/wiki/Source_Multiplayer_Networking
97 // the estimation is very good. the error is as low as ~6ms for the demo.
98 // note that passing 'rtt' is fine: EstimateTime halves it to latency.
99 double estimatedTime = LagCompensation.EstimateTime(NetworkTime.localTime, viewer.rtt, NetworkClient.bufferTime);
100
101 // sample the history to get the nearest snapshots around 'timestamp'
102 if (LagCompensation.Sample(history, estimatedTime, lagCompensationSettings.captureInterval, out Capture3D resultBefore, out Capture3D resultAfter, out double t))
103 {
104 // interpolate to get a decent estimation at exactly 'timestamp'
105 sample = Capture3D.Interpolate(resultBefore, resultAfter, t);
106 return true;
107 }
108 else Debug.Log($"CmdClicked: history doesn't contain {estimatedTime:F3}");
109
110 sample = default;
111 return false;
112 }

114 // convenience tests ///////////////////////////////////////////////////
115 // there are multiple different ways to check a hit against the sample:
116 // - raycasting
117 // - bounds.contains
118 // - increasing bounds by tolerance and checking contains
119 // - threshold to bounds.closestpoint
120 // let's offer a few solutions directly and see which users prefer.

122 // bounds check: checks distance to closest point on bounds in history @ -rtt.
123 // 'viewer' needs to be the player who fired!
124 // for example, if A fires at B, then call B.Sample(viewer, point, tolerance).
125 // this is super simple and fast, but not 100% physically accurate since we don't raycast.
126 [Server]
127 public virtual bool BoundsCheck(
128 NetworkConnectionToClient viewer,
129 Vector3 hitPoint,
130 float toleranceDistance,
131 out float distance,
132 out Vector3 nearest)
133 {
134 // first, sample the history at -rtt of the viewer.
135 if (Sample(viewer, out Capture3D capture))
136 {
137 // now that we know where the other player was at that time,
138 // we can see if the hit point was within tolerance of it.
139 // TODO consider rotations???
140 // TODO consider original collider shape??
141 Bounds bounds = new Bounds(capture.position, capture.size);
142 nearest = bounds.ClosestPoint(hitPoint);
143 distance = Vector3.Distance(nearest, hitPoint);
144 return distance <= toleranceDistance;
145 }
146 nearest = hitPoint;
147 distance = 0;
148 return false;
149 }

151 // raycast check: creates a collider the sampled position and raycasts to hitPoint.
152 // 'viewer' needs to be the player who fired!
153 // for example, if A fires at B, then call B.Sample(viewer, point, tolerance).
154 // this is physically accurate (checks against walls etc.), with the cost
155 // of a runtime instantiation.
156 //
157 // originPoint: where the player fired the weapon.
158 // hitPoint: where the player's local raycast hit.
159 // tolerance: scale up the sampled collider by % in order to have a bit of a tolerance.
160 // 0 means no extra tolerance, 0.05 means 5% extra tolerance.
161 // layerMask: the layer mask to use for the raycast.
162 [Server]
163 public virtual bool RaycastCheck(
164 NetworkConnectionToClient viewer,
165 Vector3 originPoint,
166 Vector3 hitPoint,
167 float tolerancePercent,
168 int layerMask,
169 out RaycastHit hit)
170 {
171 // first, sample the history at -rtt of the viewer.
172 if (Sample(viewer, out Capture3D capture))
173 {
174 // instantiate a real physics collider on demand.
175 // TODO rotation??
176 // TODO different collier types??
177 GameObject temp = new GameObject("LagCompensatorTest");
178 temp.transform.position = capture.position;
179 BoxCollider tempCollider = temp.AddComponent<BoxCollider>();
180 tempCollider.size = capture.size * (1 + tolerancePercent);
181
182 // raycast
183 Vector3 direction = hitPoint - originPoint;
184 float maxDistance = direction.magnitude * 2;
185 bool result = Physics.Raycast(originPoint, direction, out hit, maxDistance, layerMask);
186
187 // cleanup
188 Destroy(temp);
189 return result;
190 }
191
192 hit = default;
193 return false;
194 }
