1// batching functionality encapsulated into one class.
2// -> less complexity
3// -> easy to test
4//
5// IMPORTANT: we use THRESHOLD batching, not MAXED SIZE batching.
6// see threshold comments below.
7//
8// includes timestamp for tick batching.
9// -> allows NetworkTransform etc. to use timestamp without including it in
10// every single message
11using System;
12using System.Collections.Generic;
13
14namespace Mirror
15{
16 public class Batcher
17 {
18 // batching threshold instead of max size.
19 // -> small messages are fit into threshold sized batches
20 // -> messages larger than threshold are single batches
21 //
22 // in other words, we fit up to 'threshold' but still allow larger ones
23 // for two reasons:
24 // 1.) data races: skipping batching for larger messages would send a
25 // large spawn message immediately, while others are batched and
26 // only flushed at the end of the frame
27 // 2) timestamp batching: if each batch is expected to contain a
28 // timestamp, then large messages have to be a batch too. otherwise
29 // they would not contain a timestamp
30 readonly int threshold;
31
32 // TimeStamp header size. each batch has one.
33 public const int TimestampSize = sizeof(double);
34
35 // Message header size. each message has one.
36 public static int MessageHeaderSize(int messageSize) =>
37 Compression.VarUIntSize((ulong)messageSize);
38
39 // maximum overhead for a single message.
40 // useful for the outside to calculate max message sizes.
41 public static int MaxMessageOverhead(int messageSize) =>
42 TimestampSize + MessageHeaderSize(messageSize);
43
44 // full batches ready to be sent.
45 // DO NOT queue NetworkMessage, it would box.
46 // DO NOT queue each serialization separately.
47 // it would allocate too many writers.
48 // https://github.com/vis2k/Mirror/pull/3127
49 // => best to build batches on the fly.
50 readonly Queue<NetworkWriterPooled> batches = new Queue<NetworkWriterPooled>();
51
52 // current batch in progress
53 NetworkWriterPooled batch;
54
55 public Batcher(int threshold)
56 {
57 this.threshold = threshold;
58 }
59
60 // add a message for batching
61 // we allow any sized messages.
62 // caller needs to make sure they are within max packet size.
63 public void AddMessage(ArraySegment<byte> message, double timeStamp)
64 {
65 // predict the needed size, which is varint(size) + content
66 int headerSize = Compression.VarUIntSize((ulong)message.Count);
67 int neededSize = headerSize + message.Count;
68
69 // when appending to a batch in progress, check final size.
70 // if it expands beyond threshold, then we should finalize it first.
71 // => less than or exactly threshold is fine.
72 // GetBatch() will finalize it.
73 // => see unit tests.
74 if (batch != null &&
75 batch.Position + neededSize > threshold)
76 {
77 batches.Enqueue(batch);
78 batch = null;
79 }
80
81 // initialize a new batch if necessary
82 if (batch == null)
83 {
84 // borrow from pool. we return it in GetBatch.
85 batch = NetworkWriterPool.Get();
86
87 // write timestamp first.
88 // -> double precision for accuracy over long periods of time
89 // -> batches are per-frame, it doesn't matter which message's
90 // timestamp we use.
91 batch.WriteDouble(timeStamp);
92 }
93
94 // add serialization to current batch. even if > threshold.
95 // -> we do allow > threshold sized messages as single batch
96 // -> WriteBytes instead of WriteSegment because the latter
97 // would add a size header. we want to write directly.
98 //
99 // include size prefix as varint!
100 // -> fixes NetworkMessage serialization mismatch corrupting the
101 // next message in a batch.
102 // -> a _lot_ of time was wasted debugging corrupt batches.
103 // no easy way to figure out which NetworkMessage has a mismatch.
104 // -> this is worth everyone's sanity.
105 // -> varint means we prefix with 1 byte most of the time.
106 // -> the same issue in NetworkIdentity was why Mirror started!
107 Compression.CompressVarUInt(batch, (ulong)message.Count);
108 batch.WriteBytes(message.Array, message.Offset, message.Count);
109 }
110
111 // helper function to copy a batch to writer and return it to pool
112 static void CopyAndReturn(NetworkWriterPooled batch, NetworkWriter writer)
113 {
114 // make sure the writer is fresh to avoid uncertain situations
115 if (writer.Position != 0)
116 throw new ArgumentException($"GetBatch needs a fresh writer!");
117
118 // copy to the target writer
119 ArraySegment<byte> segment = batch.ToArraySegment();
120 writer.WriteBytes(segment.Array, segment.Offset, segment.Count);
121
122 // return batch to pool for reuse
123 NetworkWriterPool.Return(batch);
124 }
125
126 // get the next batch which is available for sending (if any).
127 // TODO safely get & return a batch instead of copying to writer?
128 // TODO could return pooled writer & use GetBatch in a 'using' statement!
129 public bool GetBatch(NetworkWriter writer)
130 {
131 // get first batch from queue (if any)
132 if (batches.TryDequeue(out NetworkWriterPooled first))
133 {
134 CopyAndReturn(first, writer);
135 return true;
136 }
137
138 // if queue was empty, we can send the batch in progress.
139 if (batch != null)
140 {
141 CopyAndReturn(batch, writer);
142 batch = null;
143 return true;
144 }
145
146 // nothing was written
147 return false;
148 }
149 }
150}
