1// standalone lag compensation algorithm
2// based on the Valve Networking Model:
3// https://developer.valvesoftware.com/wiki/Source_Multiplayer_Networking
4using System.Collections.Generic;
5
6namespace Mirror
7{
8 public static class LagCompensation
9 {
10 // history is of <timestamp, capture>.
11 // Queue allows for fast 'remove first' and 'append last'.
12 //
13 // make sure to always insert in order.
14 // inserting out of order like [1,2,4,3] would cause issues.
15 // can't safeguard this because Queue doesn't have .Last access.
16 public static void Insert<T>(
17 Queue<KeyValuePair<double, T>> history,
18 int historyLimit,
19 double timestamp,
20 T capture)
21 where T : Capture
22 {
23 // make space according to history limit.
24 // do this before inserting, to avoid resizing past capacity.
25 if (history.Count >= historyLimit)
26 history.Dequeue();
27
28 // insert
29 history.Enqueue(new KeyValuePair<double, T>(timestamp, capture));
30 }
31
32 // get the two snapshots closest to a given timestamp.
33 // those can be used to interpolate the exact snapshot at that time.
34 // if timestamp is newer than the newest history entry, then we extrapolate.
35 // 't' will be between 1 and 2, before is second last, after is last.
36 // callers should Lerp(before, after, t=1.5) to extrapolate the hit.
37 // see comments below for extrapolation.
38 public static bool Sample<T>(
39 Queue<KeyValuePair<double, T>> history,
40 double timestamp, // current server time
41 double interval, // capture interval
42 out T before,
43 out T after,
44 out double t) // interpolation factor
45 where T : Capture
46 {
47 before = default;
48 after = default;
49 t = 0;
50
51 // can't sample an empty history
52 // interpolation needs at least one entry.
53 // extrapolation needs at least two entries.
54 // can't Lerp(A, A, 1.5). dist(A, A) * 1.5 is always 0.
55 if(history.Count < 2) {
56 return false;
57 }
58
59 // older than oldest
60 if (timestamp < history.Peek().Key) {
61 return false;
62 }
63
64 // iterate through the history
65 // TODO faster version: guess start index by how many 'intervals' we are behind.
66 // search around that area.
67 // should be O(1) most of the time, unless sampling was off.
68 KeyValuePair<double, T> prev = new KeyValuePair<double, T>();
69 KeyValuePair<double, T> prevPrev = new KeyValuePair<double, T>();
70 foreach(KeyValuePair<double, T> entry in history) {
71 // exact match?
72 if (timestamp == entry.Key) {
73 before = entry.Value;
74 after = entry.Value;
75 t = Mathd.InverseLerp(before.timestamp, after.timestamp, timestamp);
76 return true;
77 }
78
79 // did we check beyond timestamp? then return the previous two.
80 if (entry.Key > timestamp) {
81 before = prev.Value;
82 after = entry.Value;
83 t = Mathd.InverseLerp(before.timestamp, after.timestamp, timestamp);
84 return true;
85 }
86
87 // remember the last two for extrapolation.
88 // Queue doesn't have access to .Last.
89 prevPrev = prev;
90 prev = entry;
91 }
92
93 // newer than newest: extrapolate up to one interval.
94 // let's say we capture every 100 ms:
95 // 100, 200, 300, 400
96 // and the server is at 499
97 // if a client sends CmdFire at time 480, then there's no history entry.
98 // => adding the current entry every time would be too expensive.
99 // worst case we would capture at 401, 402, 403, 404, ... 100 times
100 // => not extrapolating isn't great. low latency clients would be
101 // punished by missing their targets since no entry at 'time' was found.
102 // => extrapolation is the best solution. make sure this works as
103 // expected and within limits.
104 if (prev.Key < timestamp && timestamp <= prev.Key + interval) {
105 // return the last two valid snapshots.
106 // can't just return (after, after) because we can't extrapolate
107 // if their distance is 0.
108 before = prevPrev.Value;
109 after = prev.Value;
110
111 // InverseLerp will give [after, after+interval].
112 // but we return [before, after, t].
113 // so add +1 for the distance from before->after
114 t = 1 + Mathd.InverseLerp(after.timestamp, after.timestamp + interval, timestamp);
115 return true;
116 }
117
118 return false;
119 }
120
121 // never trust the client.
122 // we estimate when a message was sent.
123 // don't trust the client to tell us the time.
124 // https://developer.valvesoftware.com/wiki/Source_Multiplayer_Networking
125 // Command Execution Time = Current Server Time - Packet Latency - Client View Interpolation
126 // => lag compensation demo estimation is off by only ~6ms
127 public static double EstimateTime(double serverTime, double rtt, double bufferTime)
128 {
129 // packet latency is one trip from client to server, so rtt / 2
130 // client view interpolation is the snapshot interpolation buffer time
131 double latency = rtt / 2;
132 return serverTime - latency - bufferTime;
133 }
134
135 // convenience function to draw all history gizmos.
136 // this should be called from OnDrawGizmos.
137 public static void DrawGizmos<T>(Queue<KeyValuePair<double, T>> history)
138 where T : Capture
139 {
140 foreach (KeyValuePair<double, T> entry in history)
141 entry.Value.DrawGizmo();
142 }
143 }
