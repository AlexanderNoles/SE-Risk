58 // push hook from GetHookMethod() onto the stack as a new Action<T,T>.
59 // allows for reuse without handling static/virtual cases every time.
60 // perf warning: it is recommended to use this method only when generating IL to create a new Action<T, T>() in order to store it into a field
61 // avoid using this to emit IL to instantiate a new action instance every single time one is needed for the same method
62 public void GenerateNewActionFromHookMethod(FieldDefinition syncVar, ILProcessor worker, MethodDefinition hookMethod)
63 {
64 // IL_000a: ldarg.0
65 // IL_000b: ldftn instance void Mirror.Examples.Tanks.Tank::ExampleHook(int32, int32)
66 // IL_0011: newobj instance void class [netstandard]System.Action`2<int32, int32>::.ctor(object, native int)
67
68 // we support static hooks and instance hooks.
69 if (hookMethod.IsStatic)
70 {
71 // for static hooks, we need to push 'null' first.
72 // we can't just push nothing.
73 // stack would get out of balance because we already pushed
74 // other stuff above.
75 worker.Emit(OpCodes.Ldnull);
76 }
77 else
78 {
79 // for instance hooks, we need to push 'this.' first.
80 worker.Emit(OpCodes.Ldarg_0);
81 }
82
83 MethodReference hookMethodReference;
84 // if the network behaviour class is generic, we need to make the method reference generic for correct IL
85 if (hookMethod.DeclaringType.HasGenericParameters)
86 {
87 hookMethodReference = hookMethod.MakeHostInstanceGeneric(hookMethod.Module, hookMethod.DeclaringType.MakeGenericInstanceType(hookMethod.DeclaringType.GenericParameters.ToArray()));
88 }
89 else
90 {
91 hookMethodReference = hookMethod;
92 }
93
94 // we support regular and virtual hook functions.
95 if (hookMethod.IsVirtual)
96 {
97 // for virtual / overwritten hooks, we need different IL.
98 // this is from simply testing Action = VirtualHook; in C#.
99 worker.Emit(OpCodes.Dup);
100 worker.Emit(OpCodes.Ldvirtftn, hookMethodReference);
101 }
102 else
103 {
104 worker.Emit(OpCodes.Ldftn, hookMethodReference);
105 }
106
107 // call 'new Action<T,T>()' constructor to convert the function to an action
108 // we need to make an instance of the generic Action<T,T>.
109 TypeReference actionRef = assembly.MainModule.ImportReference(typeof(Action<,>));
110 GenericInstanceType genericInstance = actionRef.MakeGenericInstanceType(syncVar.FieldType, syncVar.FieldType);
111 worker.Emit(OpCodes.Newobj, weaverTypes.ActionT_T.MakeHostInstanceGeneric(assembly.MainModule, genericInstance));
112 }

126 MethodDefinition FindHookMethod(TypeDefinition td, FieldDefinition syncVar, string hookFunctionName, ref bool WeavingFailed)
127 {
128 List<MethodDefinition> methods = td.GetMethods(hookFunctionName);
129
130 List<MethodDefinition> methodsWith2Param = new List<MethodDefinition>(methods.Where(m => m.Parameters.Count == 2));
131
132 if (methodsWith2Param.Count == 0)
133 {
134 Log.Error($"Could not find hook for '{syncVar.Name}', hook name '{hookFunctionName}'. " +
135 $"Method signature should be {HookParameterMessage(hookFunctionName, syncVar.FieldType)}",
136 syncVar);
137 WeavingFailed = true;
138
139 return null;
140 }
141
142 foreach (MethodDefinition method in methodsWith2Param)
143 {
144 if (MatchesParameters(syncVar, method))
145 {
146 return method;
147 }
148 }
149
150 Log.Error($"Wrong type for Parameter in hook for '{syncVar.Name}', hook name '{hookFunctionName}'. " +
151 $"Method signature should be {HookParameterMessage(hookFunctionName, syncVar.FieldType)}",
152 syncVar);
153 WeavingFailed = true;
154
155 return null;
156 }

158 bool MatchesParameters(FieldDefinition syncVar, MethodDefinition method)
159 {
160 // matches void onValueChange(T oldValue, T newValue)
161 return method.Parameters[0].ParameterType.FullName == syncVar.FieldType.FullName &&
162 method.Parameters[1].ParameterType.FullName == syncVar.FieldType.FullName;
163 }

166 {

168 MethodDefinition get = new MethodDefinition(

171 MethodAttributes.HideBySig,

173
174 ILProcessor worker = get.Body.GetILProcessor();
175
176 FieldReference fr;

178 {
179 fr = fd.MakeHostInstanceGeneric();
180 }
181 else

183 fr = fd;

185
186 FieldReference netIdFieldReference = null;
187 if (netFieldId != null)
188 {
189 if (netFieldId.DeclaringType.HasGenericParameters)

191 netIdFieldReference = netFieldId.MakeHostInstanceGeneric();

193 else
194 {

196 }
197 }

209 worker.Emit(OpCodes.Call, weaverTypes.getSyncVarGameObjectReference);
210 worker.Emit(OpCodes.Ret);

218 worker.Emit(OpCodes.Ldarg_0);
219 worker.Emit(OpCodes.Ldfld, netIdFieldReference);
220 worker.Emit(OpCodes.Ldarg_0);
221 worker.Emit(OpCodes.Ldflda, fr);
222 worker.Emit(OpCodes.Call, weaverTypes.getSyncVarNetworkIdentityReference);
223 worker.Emit(OpCodes.Ret);
224 }
225 // handle both NetworkBehaviour and inheritors.
226 // fixes: https://github.com/MirrorNetworking/Mirror/issues/2939
227 else if (fd.FieldType.IsDerivedFrom<NetworkBehaviour>() || fd.FieldType.Is<NetworkBehaviour>())
228 {
229 // return this.GetSyncVarNetworkBehaviour<T>(ref field, uint netId);
230 // this.
231 worker.Emit(OpCodes.Ldarg_0);
232 worker.Emit(OpCodes.Ldarg_0);
233 worker.Emit(OpCodes.Ldfld, netIdFieldReference);
234 worker.Emit(OpCodes.Ldarg_0);
235 worker.Emit(OpCodes.Ldflda, fr);

237 worker.Emit(OpCodes.Call, getFunc);
238 worker.Emit(OpCodes.Ret);
239 }
240 // [SyncVar] int, string, etc.
241 else
242 {
243 worker.Emit(OpCodes.Ldarg_0);
244 worker.Emit(OpCodes.Ldfld, fr);
245 worker.Emit(OpCodes.Ret);
246 }

264 public MethodDefinition GenerateSyncVarSetter(TypeDefinition td, FieldDefinition fd, string originalName, long dirtyBit, FieldDefinition netFieldId, Dictionary<FieldDefinition, (FieldDefinition hookDelegateField, MethodDefinition hookMethod)> syncVarHookDelegates, ref bool WeavingFailed)
265 {
266 //Create the set method
267 MethodDefinition set = new MethodDefinition($"set_Network{originalName}", MethodAttributes.Public |
268 MethodAttributes.SpecialName |
269 MethodAttributes.HideBySig,
270 weaverTypes.Import(typeof(void)));
271
272 ILProcessor worker = set.Body.GetILProcessor();
273 FieldReference fr;
274 if (fd.DeclaringType.HasGenericParameters)
275 {
276 fr = fd.MakeHostInstanceGeneric();
277 }
278 else
279 {
280 fr = fd;
281 }
282
283 FieldReference netIdFieldReference = null;
284 if (netFieldId != null)
285 {
286 if (netFieldId.DeclaringType.HasGenericParameters)
287 {
288 netIdFieldReference = netFieldId.MakeHostInstanceGeneric();
289 }
290 else
291 {
292 netIdFieldReference = netFieldId;
293 }
294 }
295
296 // if (!SyncVarEqual(value, ref playerData))
297 Instruction endOfMethod = worker.Create(OpCodes.Nop);
298
299 // NOTE: SyncVar...Equal functions are static.
300 // don't Emit Ldarg_0 aka 'this'.
301
302 // call WeaverSyncVarSetter<T>(T value, ref T field, ulong dirtyBit, Action<T, T> OnChanged = null)
303 // IL_0000: ldarg.0
304 // IL_0001: ldarg.1
305 // IL_0002: ldarg.0
306 // IL_0003: ldflda int32 Mirror.Examples.Tanks.Tank::health
307 // IL_0008: ldc.i4.1
308 // IL_0009: conv.i8
309 // IL_000a: ldnull
310 // IL_000b: call instance void [Mirror]Mirror.NetworkBehaviour::GeneratedSyncVarSetter<int32>(!!0, !!0&, uint64, class [netstandard]System.Action`2<!!0, !!0>)
311 // IL_0010: ret
312
313 // 'this.' for the call
314 worker.Emit(OpCodes.Ldarg_0);
315
316 // first push 'value'
317 worker.Emit(OpCodes.Ldarg_1);
318
319 // push 'ref T this.field'
320 worker.Emit(OpCodes.Ldarg_0);
321 worker.Emit(OpCodes.Ldflda, fr);
322
323 // push the dirty bit for this SyncVar
324 worker.Emit(OpCodes.Ldc_I8, dirtyBit);
325
326 // hook? then push 'this.HookDelegate' onto stack
327 MethodDefinition hookMethod = GetHookMethod(td, fd, ref WeavingFailed);
328 if (hookMethod != null)
329 {
330 // Create the field that will store a single instance of the hook as a delegate (field will be set in constructor)
331 FieldDefinition hookActionDelegateField = CreateNewActionFieldDefinitionFromHookMethod(fd);
332 syncVarHookDelegates[fd] = (hookActionDelegateField, hookMethod);
333
334 // push this.hookActionDelegateField
335 worker.Emit(OpCodes.Ldarg_0);
336 worker.Emit(OpCodes.Ldfld, hookActionDelegateField);
337 }
338 // otherwise push 'null' as hook
339 else
340 {
341 worker.Emit(OpCodes.Ldnull);
342 }
343
344 // call GeneratedSyncVarSetter<T>.
345 // special cases for GameObject/NetworkIdentity/NetworkBehaviour
346 // passing netId too for persistence.
347 if (fd.FieldType.Is<UnityEngine.GameObject>())
348 {
349 // GameObject setter needs one more parameter: netId field ref
350 worker.Emit(OpCodes.Ldarg_0);
351 worker.Emit(OpCodes.Ldflda, netIdFieldReference);
352 worker.Emit(OpCodes.Call, weaverTypes.generatedSyncVarSetter_GameObject);
353 }
354 else if (fd.FieldType.Is<NetworkIdentity>())
355 {
356 // NetworkIdentity setter needs one more parameter: netId field ref
357 worker.Emit(OpCodes.Ldarg_0);
358 worker.Emit(OpCodes.Ldflda, netIdFieldReference);
359 worker.Emit(OpCodes.Call, weaverTypes.generatedSyncVarSetter_NetworkIdentity);
360 }
361 // handle both NetworkBehaviour and inheritors.
362 // fixes: https://github.com/MirrorNetworking/Mirror/issues/2939
363 else if (fd.FieldType.IsDerivedFrom<NetworkBehaviour>() || fd.FieldType.Is<NetworkBehaviour>())
364 {
365 // NetworkIdentity setter needs one more parameter: netId field ref
366 // (actually its a NetworkBehaviourSyncVar type)
367 worker.Emit(OpCodes.Ldarg_0);
368 worker.Emit(OpCodes.Ldflda, netIdFieldReference);
369 // make generic version of GeneratedSyncVarSetter_NetworkBehaviour<T>
370 MethodReference getFunc = weaverTypes.generatedSyncVarSetter_NetworkBehaviour_T.MakeGeneric(assembly.MainModule, fd.FieldType);
371 worker.Emit(OpCodes.Call, getFunc);
372 }
373 else
374 {
375 // make generic version of GeneratedSyncVarSetter<T>
376 MethodReference generic = weaverTypes.generatedSyncVarSetter.MakeGeneric(assembly.MainModule, fd.FieldType);
377 worker.Emit(OpCodes.Call, generic);
378 }
379
380 worker.Append(endOfMethod);
381
382 worker.Emit(OpCodes.Ret);
383
384 set.Parameters.Add(new ParameterDefinition("value", ParameterAttributes.In, fd.FieldType));
385 set.SemanticsAttributes = MethodSemanticsAttributes.Setter;
386
387 return set;
388 }

390 public void ProcessSyncVar(TypeDefinition td, FieldDefinition fd, Dictionary<FieldDefinition, FieldDefinition> syncVarNetIds, Dictionary<FieldDefinition, (FieldDefinition hookDelegateField, MethodDefinition hookMethod)> syncVarHookDelegates, long dirtyBit, ref bool WeavingFailed)
391 {
392 string originalName = fd.Name;
393
394 // GameObject/NetworkIdentity SyncVars have a new field for netId
395 FieldDefinition netIdField = null;
396 // NetworkBehaviour has different field type than other NetworkIdentityFields
397 // handle both NetworkBehaviour and inheritors.
398 // fixes: https://github.com/MirrorNetworking/Mirror/issues/2939
399 if (fd.FieldType.IsDerivedFrom<NetworkBehaviour>() || fd.FieldType.Is<NetworkBehaviour>())
400 {
401 netIdField = new FieldDefinition($"___{fd.Name}NetId",
402 FieldAttributes.Family, // needs to be protected for generic classes, otherwise access isn't allowed
403 weaverTypes.Import<NetworkBehaviourSyncVar>());
404 netIdField.DeclaringType = td;
405
406 syncVarNetIds[fd] = netIdField;
407 }
408 else if (fd.FieldType.IsNetworkIdentityField())
409 {
410 netIdField = new FieldDefinition($"___{fd.Name}NetId",
411 FieldAttributes.Family, // needs to be protected for generic classes, otherwise access isn't allowed
412 weaverTypes.Import<uint>());
413 netIdField.DeclaringType = td;
414
415 syncVarNetIds[fd] = netIdField;
416 }
417
418 MethodDefinition get = GenerateSyncVarGetter(fd, originalName, netIdField);
419 MethodDefinition set = GenerateSyncVarSetter(td, fd, originalName, dirtyBit, netIdField, syncVarHookDelegates, ref WeavingFailed);
420
421 //NOTE: is property even needed? Could just use a setter function?
422 //create the property
423 PropertyDefinition propertyDefinition = new PropertyDefinition($"Network{originalName}", PropertyAttributes.None, fd.FieldType)
424 {
425 GetMethod = get,
426 SetMethod = set
427 };
428
429 //add the methods and property to the type.
430 td.Methods.Add(get);
431 td.Methods.Add(set);
432 td.Properties.Add(propertyDefinition);
433 syncVarAccessLists.replacementSetterProperties[fd] = set;
434
435 // replace getter field if GameObject/NetworkIdentity so it uses
436 // netId instead
437 // -> only for GameObjects, otherwise an int syncvar's getter would
438 // end up in recursion.
439 if (fd.FieldType.IsNetworkIdentityField())
440 {
441 syncVarAccessLists.replacementGetterProperties[fd] = get;
442 }
443 }

445 public (List<FieldDefinition> syncVars, Dictionary<FieldDefinition, FieldDefinition> syncVarNetIds, Dictionary<FieldDefinition, (FieldDefinition hookDelegateField, MethodDefinition hookMethod)> syncVarHookDelegates) ProcessSyncVars(TypeDefinition td, ref bool WeavingFailed)
446 {
447 List<FieldDefinition> syncVars = new List<FieldDefinition>();
448 Dictionary<FieldDefinition, FieldDefinition> syncVarNetIds = new Dictionary<FieldDefinition, FieldDefinition>();
449 Dictionary<FieldDefinition, (FieldDefinition hookDelegateField, MethodDefinition hookMethod)> syncVarHookDelegates = new Dictionary<FieldDefinition, (FieldDefinition hookDelegateField, MethodDefinition hookMethod)>();
450
451 // the mapping of dirtybits to sync-vars is implicit in the order of the fields here. this order is recorded in m_replacementProperties.
452 // start assigning syncvars at the place the base class stopped, if any
453 int dirtyBitCounter = syncVarAccessLists.GetSyncVarStart(td.BaseType.FullName);
454
455 // find syncvars
456 foreach (FieldDefinition fd in td.Fields)
457 {
458 if (fd.HasCustomAttribute<SyncVarAttribute>())
459 {
460 if ((fd.Attributes & FieldAttributes.Static) != 0)
461 {
462 Log.Error($"{fd.Name} cannot be static", fd);
463 WeavingFailed = true;
464 continue;
465 }
466
467 if (fd.FieldType.IsGenericParameter)
468 {
469 Log.Error($"{fd.Name} has generic type. Generic SyncVars are not supported", fd);
470 WeavingFailed = true;
471 continue;
472 }
473
474 if (SyncObjectInitializer.ImplementsSyncObject(fd.FieldType))
475 {
476 Log.Warning($"{fd.Name} has [SyncVar] attribute. SyncLists should not be marked with SyncVar", fd);
477 }
478 else
479 {
480 syncVars.Add(fd);
481
482 ProcessSyncVar(td, fd, syncVarNetIds, syncVarHookDelegates, 1L << dirtyBitCounter, ref WeavingFailed);
483 dirtyBitCounter += 1;
484
485 if (dirtyBitCounter > SyncVarLimit)
486 {
487 Log.Error($"{td.Name} has > {SyncVarLimit} SyncVars. Consider refactoring your class into multiple components", td);
488 WeavingFailed = true;
489 continue;
490 }
491 }
492 }
493 }
494
495 // add all the new SyncVar __netId fields
496 foreach (FieldDefinition fd in syncVarNetIds.Values)
497 {
498 td.Fields.Add(fd);
499 }
500
501 // add all of the new SyncVar Action<T,T> fields
502 foreach((FieldDefinition hookDelegateInstanceField, MethodDefinition) entry in syncVarHookDelegates.Values)
503 {
504 td.Fields.Add(entry.hookDelegateInstanceField);
505 }
506
507 // include parent class syncvars
508 // fixes: https://github.com/MirrorNetworking/Mirror/issues/3457
509 int parentSyncVarCount = syncVarAccessLists.GetSyncVarStart(td.BaseType.FullName);
510 syncVarAccessLists.SetNumSyncVars(td.FullName, parentSyncVarCount + syncVars.Count);
511
512 return (syncVars, syncVarNetIds, syncVarHookDelegates);
513 }
