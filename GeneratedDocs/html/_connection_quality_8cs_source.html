1// standalone, Unity-independent connection-quality algorithm & enum.

5namespace Mirror
6{
7 public enum ConnectionQuality : byte
8 {
9 ESTIMATING, // still estimating
10 POOR, // unplayable
11 FAIR, // very noticeable latency, not very enjoyable anymore
12 GOOD, // very playable for everyone but high level competitors
13 EXCELLENT // ideal experience for high level competitors
14 }

16 public enum ConnectionQualityMethod : byte
17 {
18 Simple, // simple estimation based on rtt and jitter
19 Pragmatic // based on snapshot interpolation adjustment
20 }

22 // provide different heuristics for users to choose from.
23 // simple heuristics to get started.
24 // this will be iterated on over time based on user feedback.
25 public static class ConnectionQualityHeuristics
26 {
27 // convenience extension to color code Connection Quality
28 public static Color ColorCode(this ConnectionQuality quality)
29 {
30 switch (quality)
31 {
32 case ConnectionQuality.POOR: return Color.red;
33 case ConnectionQuality.FAIR: return new Color(1.0f, 0.647f, 0.0f);
34 case ConnectionQuality.GOOD: return Color.yellow;
35 case ConnectionQuality.EXCELLENT: return Color.green;
36 default: return Color.gray; // ESTIMATING
37 }
38 }
39
40 // straight forward estimation
41 // rtt: average round trip time in seconds.
42 // jitter: average latency variance.
43 public static ConnectionQuality Simple(double rtt, double jitter)
44 {
45 if (rtt <= 0.100 && jitter <= 0.10) return ConnectionQuality.EXCELLENT;
46 if (rtt <= 0.200 && jitter <= 0.20) return ConnectionQuality.GOOD;
47 if (rtt <= 0.400 && jitter <= 0.50) return ConnectionQuality.FAIR;
48 return ConnectionQuality.POOR;
49 }
50
51 // snapshot interpolation based estimation.
52 // snap. interp. adjusts buffer time based on connection quality.
53 // based on this, we can measure how far away we are from the ideal.
54 // the returned quality will always directly correlate with gameplay.
55 // => requires SnapshotInterpolation dynamicAdjustment to be enabled!
56 public static ConnectionQuality Pragmatic(double targetBufferTime, double currentBufferTime)
57 {
58 // buffer time is set by the game developer.
59 // estimating in multiples is a great way to be game independent.
60 // for example, a fast paced shooter and a slow paced RTS will both
61 // have poor connection if the multiplier is >10.
62 double multiplier = currentBufferTime / targetBufferTime;
63
64 // empirically measured with Tanks demo + LatencySimulation.
65 // it's not obvious to estimate on paper.
66 if (multiplier <= 1.15) return ConnectionQuality.EXCELLENT;
67 if (multiplier <= 1.25) return ConnectionQuality.GOOD;
68 if (multiplier <= 1.50) return ConnectionQuality.FAIR;
69
70 // anything else is poor
71 return ConnectionQuality.POOR;
72 }
73 }
