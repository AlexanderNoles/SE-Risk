81 // we need to subtract the channel and cookie bytes from every
82 // MaxMessageSize calculation.
83 // we also need to tell kcp to use MTU-1 to leave space for the byte.
84 public const int CHANNEL_HEADER_SIZE = 1;
85 public const int COOKIE_HEADER_SIZE = 4;
86 public const int METADATA_SIZE = CHANNEL_HEADER_SIZE + COOKIE_HEADER_SIZE;
87
88 // reliable channel (= kcp) MaxMessageSize so the outside knows largest
89 // allowed message to send. the calculation in Send() is not obvious at
90 // all, so let's provide the helper here.
91 //
92 // kcp does fragmentation, so max message is way larger than MTU.
93 //
94 // -> runtime MTU changes are disabled: mss is always MTU_DEF-OVERHEAD
95 // -> Send() checks if fragment count < rcv_wnd, so we use rcv_wnd - 1.
96 // NOTE that original kcp has a bug where WND_RCV default is used
97 // instead of configured rcv_wnd, limiting max message size to 144 KB
98 // https://github.com/skywind3000/kcp/pull/291
99 // we fixed this in kcp2k.
100 // -> we add 1 byte KcpHeader enum to each message, so -1
101 //
102 // IMPORTANT: max message is MTU * rcv_wnd, in other words it completely
103 // fills the receive window! due to head of line blocking,
104 // all other messages have to wait while a maxed size message
105 // is being delivered.
106 // => in other words, DO NOT use max size all the time like
107 // for batching.
108 // => sending UNRELIABLE max message size most of the time is
109 // best for performance (use that one for batching!)
110 static int ReliableMaxMessageSize_Unconstrained(int mtu, uint rcv_wnd) =>
111 (mtu - Kcp.OVERHEAD - METADATA_SIZE) * ((int)rcv_wnd - 1) - 1;
112
113 // kcp encodes 'frg' as 1 byte.
114 // max message size can only ever allow up to 255 fragments.
115 // WND_RCV gives 127 fragments.
116 // WND_RCV * 2 gives 255 fragments.
117 // so we can limit max message size by limiting rcv_wnd parameter.
118 public static int ReliableMaxMessageSize(int mtu, uint rcv_wnd) =>
119 ReliableMaxMessageSize_Unconstrained(mtu, Math.Min(rcv_wnd, Kcp.FRG_MAX));
120
121 // unreliable max message size is simply MTU - channel header - kcp header
122 public static int UnreliableMaxMessageSize(int mtu) =>
123 mtu - METADATA_SIZE - 1;
124
125 // maximum send rate per second can be calculated from kcp parameters
126 // source: https://translate.google.com/translate?sl=auto&tl=en&u=https://wetest.qq.com/lab/view/391.html
127 //
128 // KCP can send/receive a maximum of WND*MTU per interval.
129 // multiple by 1000ms / interval to get the per-second rate.
130 //
131 // example:
132 // WND(32) * MTU(1400) = 43.75KB
133 // => 43.75KB * 1000 / INTERVAL(10) = 4375KB/s
134 //
135 // returns bytes/second!
136 public uint MaxSendRate => kcp.snd_wnd * kcp.mtu * 1000 / kcp.interval;
137 public uint MaxReceiveRate => kcp.rcv_wnd * kcp.mtu * 1000 / kcp.interval;
138

143 // SetupKcp creates and configures a new KCP instance.
144 // => useful to start from a fresh state every time the client connects
145 // => NoDelay, interval, wnd size are the most important configurations.
146 // let's force require the parameters so we don't forget it anywhere.
147 protected KcpPeer(KcpConfig config, uint cookie)
148 {
149 // initialize variable state in extra function so we can reuse it
150 // when reconnecting to reset state
151 Reset(config);
152
153 // set the cookie after resetting state so it's not overwritten again.
154 // with log message for debugging in case of cookie issues.
155 this.cookie = cookie;
156 Log.Info($"[KCP] {GetType()}: created with cookie={cookie}");
157
158 // create mtu sized send buffer
159 rawSendBuffer = new byte[config.Mtu];
160
161 // calculate max message sizes once
162 unreliableMax = UnreliableMaxMessageSize(config.Mtu);
163 reliableMax = ReliableMaxMessageSize(config.Mtu, config.ReceiveWindowSize);
164
165 // create message buffers AFTER window size is set
166 // see comments on buffer definition for the "+1" part
167 kcpMessageBuffer = new byte[1 + reliableMax];
168 kcpSendBuffer = new byte[1 + reliableMax];
169 }
170
171 // Reset all state once.
172 // useful for KcpClient to reconned with a fresh kcp state.
173 protected void Reset(KcpConfig config)
174 {
175 // reset state
176 cookie = 0;
177 state = KcpState.Connected;
178 lastReceiveTime = 0;
179 lastPingTime = 0;
180 watch.Restart(); // start at 0 each time
181
182 // set up kcp over reliable channel (that's what kcp is for)
183 kcp = new Kcp(0, RawSendReliable);
184
185 // set nodelay.
186 // note that kcp uses 'nocwnd' internally so we negate the parameter
187 kcp.SetNoDelay(config.NoDelay ? 1u : 0u, config.Interval, config.FastResend, !config.CongestionWindow);
188 kcp.SetWindowSize(config.SendWindowSize, config.ReceiveWindowSize);
189
190 // IMPORTANT: high level needs to add 1 channel byte to each raw
191 // message. so while Kcp.MTU_DEF is perfect, we actually need to
192 // tell kcp to use MTU-1 so we can still put the header into the
193 // message afterwards.
194 kcp.SetMtu((uint)config.Mtu - METADATA_SIZE);
195
196 // set maximum retransmits (aka dead_link)
197 kcp.dead_link = config.MaxRetransmits;
198 timeout = config.Timeout;
199 }

217 void HandleTimeout(uint time)
218 {
219 // note: we are also sending a ping regularly, so timeout should
220 // only ever happen if the connection is truly gone.
221 if (time >= lastReceiveTime + timeout)
222 {
223 // pass error to user callback. no need to log it manually.
224 // GetType() shows Server/ClientConn instead of just Connection.
225 OnError(ErrorCode.Timeout, $"{GetType()}: Connection timed out after not receiving any message for {timeout}ms. Disconnecting.");
226 Disconnect();
227 }
228 }

230 void HandleDeadLink()
231 {
232 // kcp has 'dead_link' detection. might as well use it.
233 if (kcp.state == -1)
234 {
235 // pass error to user callback. no need to log it manually.
236 // GetType() shows Server/ClientConn instead of just Connection.
237 OnError(ErrorCode.Timeout, $"{GetType()}: dead_link detected: a message was retransmitted {kcp.dead_link} times without ack. Disconnecting.");
238 Disconnect();
239 }

242 // send a ping occasionally in order to not time out on the other end.
243 void HandlePing(uint time)
244 {
245 // enough time elapsed since last ping?
246 if (time >= lastPingTime + PING_INTERVAL)
247 {
248 // ping again and reset time
249 //Log.Debug("[KCP] sending ping...");
250 SendPing();
251 lastPingTime = time;
252 }

254
255 void HandleChoked()
256 {
257 // disconnect connections that can't process the load.
258 // see QueueSizeDisconnect comments.
259 // => include all of kcp's buffers and the unreliable queue!
260 int total = kcp.rcv_queue.Count + kcp.snd_queue.Count +
261 kcp.rcv_buf.Count + kcp.snd_buf.Count;
262 if (total >= QueueDisconnectThreshold)
263 {
264 // pass error to user callback. no need to log it manually.
265 // GetType() shows Server/ClientConn instead of just Connection.
266 OnError(ErrorCode.Congestion,
267 $"{GetType()}: disconnecting connection because it can't process data fast enough.\n" +
268 $"Queue total {total}>{QueueDisconnectThreshold}. rcv_queue={kcp.rcv_queue.Count} snd_queue={kcp.snd_queue.Count} rcv_buf={kcp.rcv_buf.Count} snd_buf={kcp.snd_buf.Count}\n" +
269 $"* Try to Enable NoDelay, decrease INTERVAL, disable Congestion Window (= enable NOCWND!), increase SEND/RECV WINDOW or compress data.\n" +
270 $"* Or perhaps the network is simply too slow on our end, or on the other end.");
271
272 // let's clear all pending sends before disconnting with 'Bye'.
273 // otherwise a single Flush in Disconnect() won't be enough to
274 // flush thousands of messages to finally deliver 'Bye'.
275 // this is just faster and more robust.
276 kcp.snd_queue.Clear();
277
278 Disconnect();
279 }

282 // reads the next reliable message type & content from kcp.
283 // -> to avoid buffering, unreliable messages call OnData directly.
284 bool ReceiveNextReliable(out KcpHeaderReliable header, out ArraySegment<byte> message)
285 {
286 message = default;
287 header = KcpHeaderReliable.Ping;
288
289 int msgSize = kcp.PeekSize();
290 if (msgSize <= 0) return false;
291
292 // only allow receiving up to buffer sized messages.
293 // otherwise we would get BlockCopy ArgumentException anyway.
294 if (msgSize > kcpMessageBuffer.Length)
295 {
296 // we don't allow sending messages > Max, so this must be an
297 // attacker. let's disconnect to avoid allocation attacks etc.
298 // pass error to user callback. no need to log it manually.
299 OnError(ErrorCode.InvalidReceive, $"{GetType()}: possible allocation attack for msgSize {msgSize} > buffer {kcpMessageBuffer.Length}. Disconnecting the connection.");
300 Disconnect();
301 return false;
302 }
303
304 // receive from kcp
305 int received = kcp.Receive(kcpMessageBuffer, msgSize);
306 if (received < 0)
307 {
308 // if receive failed, close everything
309 // pass error to user callback. no need to log it manually.
310 // GetType() shows Server/ClientConn instead of just Connection.
311 OnError(ErrorCode.InvalidReceive, $"{GetType()}: Receive failed with error={received}. closing connection.");
312 Disconnect();
313 return false;
314 }
315
316 // extract header & content without header
317 header = (KcpHeaderReliable)kcpMessageBuffer[0];
318 message = new ArraySegment<byte>(kcpMessageBuffer, 1, msgSize - 1);
319 lastReceiveTime = (uint)watch.ElapsedMilliseconds;
320 return true;
321 }

323 void TickIncoming_Connected(uint time)
324 {
325 // detect common events & ping
326 HandleTimeout(time);
327 HandleDeadLink();
328 HandlePing(time);
329 HandleChoked();
330
331 // any reliable kcp message received?
332 if (ReceiveNextReliable(out KcpHeaderReliable header, out ArraySegment<byte> message))
333 {
334 // message type FSM. no default so we never miss a case.
335 switch (header)
336 {
337 case KcpHeaderReliable.Hello:
338 {
339 // we were waiting for a Hello message.
340 // it proves that the other end speaks our protocol.
341
342 // log with previously parsed cookie
343 Log.Info($"[KCP] {GetType()}: received hello with cookie={cookie}");
344 state = KcpState.Authenticated;
345 OnAuthenticated();
346 break;
347 }
348 case KcpHeaderReliable.Ping:
349 {
350 // ping keeps kcp from timing out. do nothing.
351 break;
352 }
353 case KcpHeaderReliable.Data:
354 {
355 // everything else is not allowed during handshake!
356 // pass error to user callback. no need to log it manually.
357 // GetType() shows Server/ClientConn instead of just Connection.
358 OnError(ErrorCode.InvalidReceive, $"[KCP] {GetType()}: received invalid header {header} while Connected. Disconnecting the connection.");
359 Disconnect();
360 break;
361 }
362 }
363 }
364 }

366 void TickIncoming_Authenticated(uint time)
367 {
368 // detect common events & ping
369 HandleTimeout(time);
370 HandleDeadLink();
371 HandlePing(time);
372 HandleChoked();
373
374 // process all received messages
375 while (ReceiveNextReliable(out KcpHeaderReliable header, out ArraySegment<byte> message))
376 {
377 // message type FSM. no default so we never miss a case.
378 switch (header)
379 {
380 case KcpHeaderReliable.Hello:
381 {
382 // should never receive another hello after auth
383 // GetType() shows Server/ClientConn instead of just Connection.
384 Log.Warning($"{GetType()}: received invalid header {header} while Authenticated. Disconnecting the connection.");
385 Disconnect();
386 break;
387 }
388 case KcpHeaderReliable.Data:
389 {
390 // call OnData IF the message contained actual data
391 if (message.Count > 0)
392 {
393 //Log.Warning($"Kcp recv msg: {BitConverter.ToString(message.Array, message.Offset, message.Count)}");
394 OnData(message, KcpChannel.Reliable);
395 }
396 // empty data = attacker, or something went wrong
397 else
398 {
399 // pass error to user callback. no need to log it manually.
400 // GetType() shows Server/ClientConn instead of just Connection.
401 OnError(ErrorCode.InvalidReceive, $"{GetType()}: received empty Data message while Authenticated. Disconnecting the connection.");
402 Disconnect();
403 }
404 break;
405 }
406 case KcpHeaderReliable.Ping:
407 {
408 // ping keeps kcp from timing out. do nothing.
409 break;
410 }
411 }
412 }

416 {
417 uint time = (uint)watch.ElapsedMilliseconds;
418
419 try
420 {
421 switch (state)
422 {
423 case KcpState.Connected:

425 TickIncoming_Connected(time);

432 }

436 break;
437 }
438 }
439 }

447 Disconnect();
448 }
449 catch (ObjectDisposedException exception)
450 {
451 // fine, socket was closed
452 // pass error to user callback. no need to log it manually.
453 // GetType() shows Server/ClientConn instead of just Connection.

527 protected void OnRawInputUnreliable(ArraySegment<byte> message)
528 {
529 // need at least one byte for the KcpHeader enum
530 if (message.Count < 1) return;
531
532 // parse header and subtract it from message content.
533 // (above we already ensure it's at least 1 byte long)
534 KcpHeaderUnreliable header = (KcpHeaderUnreliable)message.Array[message.Offset + 0];
535 message = new ArraySegment<byte>(message.Array, message.Offset + 1, message.Count - 1);
536
537 switch (header)
538 {
539 case KcpHeaderUnreliable.Data:
540 {
541 // ideally we would queue all unreliable messages and
542 // then process them in ReceiveNext() together with the
543 // reliable messages, but:
544 // -> queues/allocations/pools are slow and complex.
545 // -> DOTSNET 10k is actually slower if we use pooled
546 // unreliable messages for transform messages.
547 //
548 // DOTSNET 10k benchmark:
549 // reliable-only: 170 FPS
550 // unreliable queued: 130-150 FPS
551 // unreliable direct: 183 FPS(!)
552 //
553 // DOTSNET 50k benchmark:
554 // reliable-only: FAILS (queues keep growing)
555 // unreliable direct: 18-22 FPS(!)
556 //
557 // -> all unreliable messages are DATA messages anyway.
558 // -> let's skip the magic and call OnData directly if
559 // the current state allows it.
560 if (state == KcpState.Authenticated)
561 {
562 OnData(message, KcpChannel.Unreliable);
563
564 // set last receive time to avoid timeout.
565 // -> we do this in ANY case even if not enabled.
566 // a message is a message.
567 // -> we set last receive time for both reliable and
568 // unreliable messages. both count.
569 // otherwise a connection might time out even
570 // though unreliable were received, but no
571 // reliable was received.
572 lastReceiveTime = (uint)watch.ElapsedMilliseconds;
573 }
574 else
575 {
576 // it's common to receive unreliable messages before being
577 // authenticated, for example:
578 // - random internet noise
579 // - game server may send an unreliable message after authenticating,
580 // and the unreliable message arrives on the client before the
581 // 'auth_ok' message. this can be avoided by sending a final
582 // 'ready' message after being authenticated, but this would
583 // add another 'round trip time' of latency to the handshake.
584 //
585 // it's best to simply ignore invalid unreliable messages here.
586 // Log.Info($"{GetType()}: received unreliable message while not authenticated.");
587 }
588 break;
589 }
590 case KcpHeaderUnreliable.Disconnect:
591 {
592 // GetType() shows Server/ClientConn instead of just Connection.
593 Log.Info($"[KCP] {GetType()}: received disconnect message");
594 Disconnect();
595 break;
596 }
597 }

600 // raw send called by kcp
601 void RawSendReliable(byte[] data, int length)
602 {
603 // write channel header
604 // from 0, with 1 byte
605 rawSendBuffer[0] = (byte)KcpChannel.Reliable;
606
607 // write handshake cookie to protect against UDP spoofing.
608 // from 1, with 4 bytes
609 Utils.Encode32U(rawSendBuffer, 1, cookie); // allocation free
610
611 // write data
612 // from 5, with N bytes
613 Buffer.BlockCopy(data, 0, rawSendBuffer, 1+4, length);
614
615 // IO send
616 ArraySegment<byte> segment = new ArraySegment<byte>(rawSendBuffer, 0, length + 1+4);
617 RawSend(segment);
618 }

620 void SendReliable(KcpHeaderReliable header, ArraySegment<byte> content)
621 {
622 // 1 byte header + content needs to fit into send buffer
623 if (1 + content.Count > kcpSendBuffer.Length) // TODO
624 {
625 // otherwise content is larger than MaxMessageSize. let user know!
626 // GetType() shows Server/ClientConn instead of just Connection.
627 OnError(ErrorCode.InvalidSend, $"{GetType()}: Failed to send reliable message of size {content.Count} because it's larger than ReliableMaxMessageSize={reliableMax}");
628 return;
629 }
630
631 // write channel header
632 kcpSendBuffer[0] = (byte)header;
633
634 // write data (if any)
635 if (content.Count > 0)
636 Buffer.BlockCopy(content.Array, content.Offset, kcpSendBuffer, 1, content.Count);
637
638 // send to kcp for processing
639 int sent = kcp.Send(kcpSendBuffer, 0, 1 + content.Count);
640 if (sent < 0)
641 {
642 // GetType() shows Server/ClientConn instead of just Connection.
643 OnError(ErrorCode.InvalidSend, $"{GetType()}: Send failed with error={sent} for content with length={content.Count}");
644 }

695 public void SendData(ArraySegment<byte> data, KcpChannel channel)
696 {
697 // sending empty segments is not allowed.
698 // nobody should ever try to send empty data.
699 // it means that something went wrong, e.g. in Mirror/DOTSNET.
700 // let's make it obvious so it's easy to debug.
701 if (data.Count == 0)
702 {
703 // pass error to user callback. no need to log it manually.
704 // GetType() shows Server/ClientConn instead of just Connection.
705 OnError(ErrorCode.InvalidSend, $"{GetType()}: tried sending empty message. This should never happen. Disconnecting.");
706 Disconnect();
707 return;
708 }
709
710 switch (channel)
711 {
712 case KcpChannel.Reliable:
713 SendReliable(KcpHeaderReliable.Data, data);
714 break;
715 case KcpChannel.Unreliable:
716 SendUnreliable(KcpHeaderUnreliable.Data, data);
717 break;
718 }
719 }
720
721 // ping goes through kcp to keep it from timing out, so it goes over the
722 // reliable channel.
723 void SendPing() => SendReliable(KcpHeaderReliable.Ping, default);
724
725 // send disconnect message
726 void SendDisconnect()
727 {
728 // sending over reliable to ensure delivery seems like a good idea:
729 // but if we close the connection immediately, it often doesn't get
730 // fully delivered: https://github.com/MirrorNetworking/Mirror/issues/3591
731 // SendReliable(KcpHeader.Disconnect, default);
732 //
733 // instead, rapid fire a few unreliable messages.
734 // they are sent immediately even if we close the connection after.
735 // this way we don't need to keep the connection alive for a while.
736 // (glenn fiedler method)
737 for (int i = 0; i < 5; ++i)
738 SendUnreliable(KcpHeaderUnreliable.Disconnect, default);
739 }
740
741 // disconnect this connection
742 public virtual void Disconnect()
743 {
744 // only if not disconnected yet
745 if (state == KcpState.Disconnected)
746 return;
747
748 // send a disconnect message
749 try
750 {
751 SendDisconnect();
752 }
753 // TODO KcpConnection is IO agnostic. move this to outside later.
754 catch (SocketException)
755 {
756 // this is ok, the connection was already closed
757 }
758 catch (ObjectDisposedException)
759 {
760 // this is normal when we stop the server
761 // the socket is stopped so we can't send anything anymore
762 // to the clients
763
764 // the clients will eventually timeout and realize they
765 // were disconnected
766 }
767
768 // set as Disconnected, call event
769 // GetType() shows Server/ClientConn instead of just Connection.
770 Log.Info($"[KCP] {GetType()}: Disconnected.");
771 state = KcpState.Disconnected;
772 OnDisconnected();
773 }
