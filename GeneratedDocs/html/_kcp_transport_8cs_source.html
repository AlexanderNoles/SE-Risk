84 public static TransportError ToTransportError(ErrorCode error)
85 {
86 switch(error)
87 {
88 case ErrorCode.DnsResolve: return TransportError.DnsResolve;
89 case ErrorCode.Timeout: return TransportError.Timeout;
90 case ErrorCode.Congestion: return TransportError.Congestion;
91 case ErrorCode.InvalidReceive: return TransportError.InvalidReceive;
92 case ErrorCode.InvalidSend: return TransportError.InvalidSend;
93 case ErrorCode.ConnectionClosed: return TransportError.ConnectionClosed;
94 case ErrorCode.Unexpected: return TransportError.Unexpected;
95 default: throw new InvalidCastException($"KCP: missing error translation for {error}");
96 }
97 }

233 // max message size
234 public override int GetMaxPacketSize(int channelId = Channels.Reliable)
235 {
236 // switch to kcp channel.
237 // unreliable or reliable.
238 // default to reliable just to be sure.
239 switch (channelId)
240 {
241 case Channels.Unreliable:
242 return KcpPeer.UnreliableMaxMessageSize(config.Mtu);
243 default:
244 return KcpPeer.ReliableMaxMessageSize(config.Mtu, ReceiveWindowSize);
245 }
246 }
247
248 // kcp reliable channel max packet size is MTU * WND_RCV
249 // this allows 144kb messages. but due to head of line blocking, all
250 // other messages would have to wait until the maxed size one is
251 // delivered. batching 144kb messages each time would be EXTREMELY slow
252 // and fill the send queue nearly immediately when using it over the
253 // network.
254 // => instead we always use MTU sized batches.
255 // => people can still send maxed size if needed.
256 public override int GetBatchThreshold(int channelId) =>
257 KcpPeer.UnreliableMaxMessageSize(config.Mtu);

279 // PrettyBytes function from DOTSNET
280 // pretty prints bytes as KB/MB/GB/etc.
281 // long to support > 2GB
282 // divides by floats to return "2.5MB" etc.
283 public static string PrettyBytes(long bytes)
284 {
285 // bytes
286 if (bytes < 1024)
287 return $"{bytes} B";
288 // kilobytes
289 else if (bytes < 1024L * 1024L)
290 return $"{(bytes / 1024f):F2} KB";
291 // megabytes
292 else if (bytes < 1024 * 1024L * 1024L)
293 return $"{(bytes / (1024f * 1024f)):F2} MB";
294 // gigabytes
295 return $"{(bytes / (1024f * 1024f * 1024f)):F2} GB";
296 }
