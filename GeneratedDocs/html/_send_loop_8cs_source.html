38 public static void Loop(Config config)
39 {
40 (Connection conn, int bufferSize, bool setMask) = config;
41
42 Profiler.BeginThreadProfiling("SimpleWeb", $"SendLoop {conn.connId}");
43
44 // create write buffer for this thread
45 byte[] writeBuffer = new byte[bufferSize];
46 MaskHelper maskHelper = setMask ? new MaskHelper() : null;
47 try
48 {
49 TcpClient client = conn.client;
50 Stream stream = conn.stream;
51
52 // null check in case disconnect while send thread is starting
53 if (client == null)
54 return;
55
56 while (client.Connected)
57 {
58 // wait for message
59 conn.sendPending.Wait();
60 // wait for 1ms for mirror to send other messages
61 if (SendLoopConfig.sleepBeforeSend)
62 Thread.Sleep(1);
63
64 conn.sendPending.Reset();
65
66 if (SendLoopConfig.batchSend)
67 {
68 int offset = 0;
69 while (conn.sendQueue.TryDequeue(out ArrayBuffer msg))
70 {
71 // check if connected before sending message
72 if (!client.Connected)
73 {
74 Log.Verbose($"[SWT-SendLoop]: SendLoop {conn} not connected");
75 msg.Release();
76 return;
77 }
78
79 int maxLength = msg.count + Constants.HeaderSize + Constants.MaskSize;
80
81 // if next writer could overflow, write to stream and clear buffer
82 if (offset + maxLength > bufferSize)
83 {
84 stream.Write(writeBuffer, 0, offset);
85 offset = 0;
86 }
87
88 offset = SendMessage(writeBuffer, offset, msg, setMask, maskHelper);
89 msg.Release();
90 }
91
92 // after no message in queue, send remaining messages
93 // don't need to check offset > 0 because last message in queue will always be sent here
94
95 stream.Write(writeBuffer, 0, offset);
96 }
97 else
98 {
99 while (conn.sendQueue.TryDequeue(out ArrayBuffer msg))
100 {
101 // check if connected before sending message
102 if (!client.Connected)
103 {
104 Log.Verbose($"[SWT-SendLoop]: SendLoop {conn} not connected");
105 msg.Release();
106 return;
107 }
108
109 int length = SendMessage(writeBuffer, 0, msg, setMask, maskHelper);
110 stream.Write(writeBuffer, 0, length);
111 msg.Release();
112 }
113 }
114 }
115
116 Log.Verbose($"[SWT-SendLoop]: {conn} Not Connected");
117 }
118 catch (ThreadInterruptedException e) { Log.InfoException(e); }
119 catch (ThreadAbortException e) { Log.InfoException(e); }
120 catch (Exception e)
121 {
122 Log.Exception(e);
123 }
124 finally
125 {
126 Profiler.EndThreadProfiling();
127 conn.Dispose();
128 maskHelper?.Dispose();
129 }

133 static int SendMessage(byte[] buffer, int startOffset, ArrayBuffer msg, bool setMask, MaskHelper maskHelper)
134 {
135 int msgLength = msg.count;
136 int offset = WriteHeader(buffer, startOffset, msgLength, setMask);
137
138 if (setMask)
139 {
140 offset = maskHelper.WriteMask(buffer, offset);
141 }
142
143 msg.CopyTo(buffer, offset);
144 offset += msgLength;
145

153 }

156 }

158 public static int WriteHeader(byte[] buffer, int startOffset, int msgLength, bool setMask)
159 {
160 int sendLength = 0;
161 const byte finished = 128;
162 const byte byteOpCode = 2;
163
164 buffer[startOffset + 0] = finished | byteOpCode;
165 sendLength++;
166
167 if (msgLength <= Constants.BytePayloadLength)
168 {
169 buffer[startOffset + 1] = (byte)msgLength;
170 sendLength++;
171 }
172 else if (msgLength <= ushort.MaxValue)
173 {
174 buffer[startOffset + 1] = 126;
175 buffer[startOffset + 2] = (byte)(msgLength >> 8);
176 buffer[startOffset + 3] = (byte)msgLength;
177 sendLength += 3;
178 }
179 else
180 {
181 buffer[startOffset + 1] = 127;
182 // must be 64 bytes, but we only have 32 bit length, so first 4 bits are 0
183 buffer[startOffset + 2] = 0;
184 buffer[startOffset + 3] = 0;
185 buffer[startOffset + 4] = 0;
186 buffer[startOffset + 5] = 0;
187 buffer[startOffset + 6] = (byte)(msgLength >> 24);
188 buffer[startOffset + 7] = (byte)(msgLength >> 16);
189 buffer[startOffset + 8] = (byte)(msgLength >> 8);
190 buffer[startOffset + 9] = (byte)msgLength;
191
192 sendLength += 9;
193 }
194
195 if (setMask)
196 buffer[startOffset + 1] |= 0b1000_0000;
197
198 return sendLength + startOffset;
199 }
