324 private void UpdatePlayerUI()
325 {

328 startButton.gameObject.SetActive(true);

342 for (int i = 0; i < extraPlayerSlots.Count; i++)

344 yPos -= 120;
345 Vector2 newPos = new Vector2(0, yPos);
346
347 if (i < playOptions.numberOfNetworkPlayers)

349 extraPlayerSlots[i].gameObject.SetActive(true);
350 extraPlayerSlots[i].anchoredPosition = newPos;

352 string labelText = "LAN player";

354 color.a = 0.8f;
355
356 if (i == 0)

359 {
360 labelText = "You";

362 else if (NetworkManagement.GetClientState() == NetworkManagement.ClientState.Host)
363 {
364 labelText = "You (host)";
365 }
366 else
367 {
368 labelText = "Host";
369 }
370 }
371 else if (i == 1 && NetworkManagement.GetClientState() == NetworkManagement.ClientState.Client)
372 {
373 //Temp, all clients see first player as them
374 labelText = "You";
375 }
376
377 playerSlotRefrences[i].Item1.color = color;
378 playerSlotRefrences[i].Item2.text = labelText;
379 playerSlotRefrences[i].Item3.SetActive(false);
380 }
381 else if (i-playOptions.numberOfNetworkPlayers < playOptions.numberOfAIPlayers)
382 {
383 extraPlayerSlots[i].gameObject.SetActive(true);
384 extraPlayerSlots[i].anchoredPosition = newPos;
385
386 playerSlotRefrences[i].Item1.color = new Color(0.9f, 0.9f, 0.9f, 0.8f);
387 playerSlotRefrences[i].Item2.text = "AI PLAYER";

389 }
390 else
391 {
392 if (i == GetTotalNumberOfPlayers())
393 {
394 setAddButtonActive = NetworkManagement.GetClientState() != NetworkManagement.ClientState.Client;
395 addSlotButton.anchoredPosition = new Vector2(0, yPos);
