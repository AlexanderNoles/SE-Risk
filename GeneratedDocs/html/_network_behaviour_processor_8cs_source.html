65 // return true if modified
66 public bool Process(ref bool WeavingFailed)
67 {
68 // only process once
69 if (WasProcessed(netBehaviourSubclass))
70 {
71 return false;
72 }
73
74 MarkAsProcessed(netBehaviourSubclass);
75
76 // deconstruct tuple and set fields
77 (syncVars, syncVarNetIds, syncVarHookDelegates) = syncVarAttributeProcessor.ProcessSyncVars(netBehaviourSubclass, ref WeavingFailed);
78
79 syncObjects = SyncObjectProcessor.FindSyncObjectsFields(writers, readers, Log, netBehaviourSubclass, ref WeavingFailed);
80
81 ProcessMethods(ref WeavingFailed);
82 if (WeavingFailed)
83 {
84 // originally Process returned true in every case, except if already processed.
85 // maybe return false here in the future.
86 return true;
87 }
88
89 // inject initializations into static & instance constructor
90 InjectIntoStaticConstructor(ref WeavingFailed);
91 InjectIntoInstanceConstructor(ref WeavingFailed);
92
93 GenerateSerialization(ref WeavingFailed);
94 if (WeavingFailed)
95 {
96 // originally Process returned true in every case, except if already processed.
97 // maybe return false here in the future.
98 return true;
99 }
100
101 GenerateDeSerialization(ref WeavingFailed);
102 return true;
103 }

160 public static bool WriteArguments(ILProcessor worker, Writers writers, Logger Log, MethodDefinition method, RemoteCallType callType, ref bool WeavingFailed)
161 {
162 // write each argument
163 // example result
164 /*
165 writer.WritePackedInt32(someNumber);
166 writer.WriteNetworkIdentity(someTarget);
167 */
168
169 bool skipFirst = callType == RemoteCallType.TargetRpc
170 && TargetRpcProcessor.HasNetworkConnectionParameter(method);
171
172 // arg of calling function, arg 0 is "this" so start counting at 1
173 int argNum = 1;
174 foreach (ParameterDefinition param in method.Parameters)
175 {
176 // NetworkConnection is not sent via the NetworkWriter so skip it here
177 // skip first for NetworkConnection in TargetRpc
178 if (argNum == 1 && skipFirst)
179 {
180 argNum += 1;
181 continue;
182 }
183 // skip SenderConnection in Command
184 if (IsSenderConnection(param, callType))
185 {
186 argNum += 1;
187 continue;
188 }
189
190 MethodReference writeFunc = writers.GetWriteFunc(param.ParameterType, ref WeavingFailed);
191 if (writeFunc == null)
192 {
193 Log.Error($"{method.Name} has invalid parameter {param}", method);
194 WeavingFailed = true;
195 return false;
196 }
197
198 // use built-in writer func on writer object
199 // NetworkWriter object
200 worker.Emit(OpCodes.Ldloc_0);
201 // add argument to call
202 worker.Emit(OpCodes.Ldarg, argNum);
203 // call writer extension method
204 worker.Emit(OpCodes.Call, writeFunc);
205 argNum += 1;
206 }
207 return true;
208 }

240 // helper function to remove 'Ret' from the end of the method if 'Ret'
241 // is the last instruction.
242 // returns false if there was an issue
243 static bool RemoveFinalRetInstruction(MethodDefinition method)
244 {
245 // remove the return opcode from end of function. will add our own later.
246 if (method.Body.Instructions.Count != 0)
247 {
248 Instruction retInstr = method.Body.Instructions[method.Body.Instructions.Count - 1];
249 if (retInstr.OpCode == OpCodes.Ret)
250 {
251 method.Body.Instructions.RemoveAt(method.Body.Instructions.Count - 1);
252 return true;
253 }
254 return false;
255 }
256
257 // we did nothing, but there was no error.
258 return true;
259 }

261 // we need to inject several initializations into NetworkBehaviour cctor
262 void InjectIntoStaticConstructor(ref bool WeavingFailed)
263 {
264 if (commands.Count == 0 && clientRpcs.Count == 0 && targetRpcs.Count == 0)
265 return;
266
267 // find static constructor
268 MethodDefinition cctor = netBehaviourSubclass.GetMethod(".cctor");
269 bool cctorFound = cctor != null;
270 if (cctor != null)
271 {
272 // remove the return opcode from end of function. will add our own later.
273 if (!RemoveFinalRetInstruction(cctor))
274 {
275 Log.Error($"{netBehaviourSubclass.Name} has invalid class constructor", cctor);
276 WeavingFailed = true;
277 return;
278 }
279 }
280 else
281 {
282 // make one!
283 cctor = new MethodDefinition(".cctor", MethodAttributes.Private |
284 MethodAttributes.HideBySig |
285 MethodAttributes.SpecialName |
286 MethodAttributes.RTSpecialName |
287 MethodAttributes.Static,
288 weaverTypes.Import(typeof(void)));
289 }
290
291 ILProcessor cctorWorker = cctor.Body.GetILProcessor();
292
293 // register all commands in cctor
294 for (int i = 0; i < commands.Count; ++i)
295 {
296 CmdResult cmdResult = commands[i];
297 GenerateRegisterCommandDelegate(cctorWorker, weaverTypes.registerCommandReference, commandInvocationFuncs[i], cmdResult);
298 }
299
300 // register all client rpcs in cctor
301 for (int i = 0; i < clientRpcs.Count; ++i)
302 {
303 ClientRpcResult clientRpcResult = clientRpcs[i];
304 GenerateRegisterRemoteDelegate(cctorWorker, weaverTypes.registerRpcReference, clientRpcInvocationFuncs[i], clientRpcResult.method.FullName);
305 }
306
307 // register all target rpcs in cctor
308 for (int i = 0; i < targetRpcs.Count; ++i)
309 {
310 GenerateRegisterRemoteDelegate(cctorWorker, weaverTypes.registerRpcReference, targetRpcInvocationFuncs[i], targetRpcs[i].FullName);
311 }
312
313 // add final 'Ret' instruction to cctor
314 cctorWorker.Append(cctorWorker.Create(OpCodes.Ret));
315 if (!cctorFound)
316 {
317 netBehaviourSubclass.Methods.Add(cctor);
318 }
319
320 // in case class had no cctor, it might have BeforeFieldInit, so injected cctor would be called too late
321 netBehaviourSubclass.Attributes &= ~TypeAttributes.BeforeFieldInit;
322 }

324 // we need to inject several initializations into NetworkBehaviour ctor
325 void InjectIntoInstanceConstructor(ref bool WeavingFailed)
326 {
327 if ((syncObjects.Count == 0) && (syncVarHookDelegates.Count == 0))
328 return;
329
330 // find instance constructor
331 MethodDefinition ctor = netBehaviourSubclass.GetMethod(".ctor");
332 if (ctor == null)
333 {
334 Log.Error($"{netBehaviourSubclass.Name} has invalid constructor", netBehaviourSubclass);
335 WeavingFailed = true;
336 return;
337 }
338
339 // remove the return opcode from end of function. will add our own later.
340 if (!RemoveFinalRetInstruction(ctor))
341 {
342 Log.Error($"{netBehaviourSubclass.Name} has invalid constructor", ctor);
343 WeavingFailed = true;
344 return;
345 }
346
347 ILProcessor ctorWorker = ctor.Body.GetILProcessor();
348
349 // initialize all sync objects in ctor
350 foreach (FieldDefinition fd in syncObjects)
351 {
352 SyncObjectInitializer.GenerateSyncObjectInitializer(ctorWorker, weaverTypes, fd);
353 }
354
355 // initialize all delegate fields in ctor
356 foreach(KeyValuePair<FieldDefinition, (FieldDefinition, MethodDefinition)> entry in syncVarHookDelegates)
357 {
358 FieldDefinition syncVarField = entry.Key;
359 (FieldDefinition hookDelegate, MethodDefinition hookMethod) = entry.Value;
360 syncVarAttributeProcessor.GenerateSyncVarHookDelegateInitializer(ctorWorker, syncVarField, hookDelegate, hookMethod);
361 }
362
363 // add final 'Ret' instruction to ctor
364 ctorWorker.Append(ctorWorker.Create(OpCodes.Ret));
365 }

406 void GenerateSerialization(ref bool WeavingFailed)
407 {
408 const string SerializeMethodName = "SerializeSyncVars";
409 if (netBehaviourSubclass.GetMethod(SerializeMethodName) != null)
410 return;
411
412 if (syncVars.Count == 0)
413 {
414 // no synvars, no need for custom OnSerialize
415 return;
416 }
417
418 MethodDefinition serialize = new MethodDefinition(SerializeMethodName,
419 MethodAttributes.Public | MethodAttributes.Virtual | MethodAttributes.HideBySig,
420 weaverTypes.Import(typeof(void)));
421
422 serialize.Parameters.Add(new ParameterDefinition("writer", ParameterAttributes.None, weaverTypes.Import<NetworkWriter>()));
423 serialize.Parameters.Add(new ParameterDefinition("forceAll", ParameterAttributes.None, weaverTypes.Import<bool>()));
424 ILProcessor worker = serialize.Body.GetILProcessor();
425
426 serialize.Body.InitLocals = true;
427
428 // base.SerializeSyncVars(writer, forceAll);
429 MethodReference baseSerialize = Resolvers.TryResolveMethodInParents(netBehaviourSubclass.BaseType, assembly, SerializeMethodName);
430 if (baseSerialize != null)
431 {
432 // base
433 worker.Emit(OpCodes.Ldarg_0);
434 // writer
435 worker.Emit(OpCodes.Ldarg_1);
436 // forceAll
437 worker.Emit(OpCodes.Ldarg_2);
438 worker.Emit(OpCodes.Call, baseSerialize);
439 }
440
441 // Generates:
442 // if (forceAll)
443 // {
444 // writer.WriteInt(health);
445 // ...
446 // }
447 Instruction initialStateLabel = worker.Create(OpCodes.Nop);
448 // forceAll
449 worker.Emit(OpCodes.Ldarg_2); // load 'forceAll' flag
450 worker.Emit(OpCodes.Brfalse, initialStateLabel); // start the 'if forceAll' branch
451
452 // generates write.Write(syncVar) for each SyncVar in forceAll case
453 foreach (FieldDefinition syncVarDef in syncVars)
454 {
455 FieldReference syncVar = syncVarDef;
456 if (netBehaviourSubclass.HasGenericParameters)
457 {
458 syncVar = syncVarDef.MakeHostInstanceGeneric();
459 }
460 // Generates a writer call for each sync variable
461 // writer
462 worker.Emit(OpCodes.Ldarg_1);
463 // this
464 worker.Emit(OpCodes.Ldarg_0);
465 worker.Emit(OpCodes.Ldfld, syncVar);
466 MethodReference writeFunc;
467 // For NBs we always need to use the default NetworkBehaviour write func
468 // since the reader counter part uses that exact layout which is not easy to change
469 // without introducing more edge cases
470 // effectively this disallows custom NB-type writers/readers on SyncVars
471 // see: https://github.com/MirrorNetworking/Mirror/issues/2680
472 if (syncVar.FieldType.IsDerivedFrom<NetworkBehaviour>())
473 {
474 writeFunc = writers.GetWriteFunc(weaverTypes.Import<NetworkBehaviour>(), ref WeavingFailed);
475 }
476 else
477 {
478 writeFunc = writers.GetWriteFunc(syncVar.FieldType, ref WeavingFailed);
479 }
480
481 if (writeFunc != null)
482 {
483 worker.Emit(OpCodes.Call, writeFunc);
484 }
485 else
486 {
487 Log.Error($"{syncVar.Name} has unsupported type. Use a supported Mirror type instead", syncVar);
488 WeavingFailed = true;
489 return;
490 }
491 }
492
493 // if (forceAll) then always return at the end of the 'if' case
494 worker.Emit(OpCodes.Ret);
495
496 // end the 'if' case for "if (forceAll)"
497 worker.Append(initialStateLabel);
498
500
501 // write dirty bits before the data fields
502 // Generates: writer.WritePackedUInt64 (base.get_syncVarDirtyBits ());
503 // writer
504 worker.Emit(OpCodes.Ldarg_1);
505 // base
506 worker.Emit(OpCodes.Ldarg_0);
507 worker.Emit(OpCodes.Ldfld, weaverTypes.NetworkBehaviourDirtyBitsReference);
508 MethodReference writeUint64Func = writers.GetWriteFunc(weaverTypes.Import<ulong>(), ref WeavingFailed);
509 worker.Emit(OpCodes.Call, writeUint64Func);
510
511 // generate a writer call for any dirty variable in this class
512
513 // start at number of syncvars in parent
514 int dirtyBit = syncVarAccessLists.GetSyncVarStart(netBehaviourSubclass.BaseType.FullName);
515 foreach (FieldDefinition syncVarDef in syncVars)
516 {
517 FieldReference syncVar = syncVarDef;
518 if (netBehaviourSubclass.HasGenericParameters)
519 {
520 syncVar = syncVarDef.MakeHostInstanceGeneric();
521 }
522 Instruction varLabel = worker.Create(OpCodes.Nop);
523
524 // Generates: if ((base.get_syncVarDirtyBits() & 1uL) != 0uL)
525 // base
526 worker.Emit(OpCodes.Ldarg_0);
527 worker.Emit(OpCodes.Ldfld, weaverTypes.NetworkBehaviourDirtyBitsReference);
528 // 8 bytes = long
529 worker.Emit(OpCodes.Ldc_I8, 1L << dirtyBit);
530 worker.Emit(OpCodes.And);
531 worker.Emit(OpCodes.Brfalse, varLabel);
532
533 // Generates a call to the writer for that field
534 // writer
535 worker.Emit(OpCodes.Ldarg_1);
536 // base
537 worker.Emit(OpCodes.Ldarg_0);
538 worker.Emit(OpCodes.Ldfld, syncVar);
539
540 MethodReference writeFunc;
541 // For NBs we always need to use the default NetworkBehaviour write func
542 // since the reader counter part uses that exact layout which is not easy to change
543 // without introducing more edge cases
544 // effectively this disallows custom NB-type writers/readers on SyncVars
545 // see: https://github.com/MirrorNetworking/Mirror/issues/2680
546 if (syncVar.FieldType.IsDerivedFrom<NetworkBehaviour>())
547 {
548 writeFunc = writers.GetWriteFunc(weaverTypes.Import<NetworkBehaviour>(), ref WeavingFailed);
549 }
550 else
551 {
552 writeFunc = writers.GetWriteFunc(syncVar.FieldType, ref WeavingFailed);
553 }
554
555 if (writeFunc != null)
556 {
557 worker.Emit(OpCodes.Call, writeFunc);
558 }
559 else
560 {
561 Log.Error($"{syncVar.Name} has unsupported type. Use a supported Mirror type instead", syncVar);
562 WeavingFailed = true;
563 return;
564 }
565
566 worker.Append(varLabel);
567 dirtyBit += 1;
568 }
569
570 // add a log message if needed for debugging
571 //worker.Emit(OpCodes.Ldstr, $"Injected Serialize {netBehaviourSubclass.Name}");
572 //worker.Emit(OpCodes.Call, WeaverTypes.logErrorReference);
573
574 // generate: return
575 worker.Emit(OpCodes.Ret);
576 netBehaviourSubclass.Methods.Add(serialize);
577 }

579 void DeserializeField(FieldDefinition syncVar, ILProcessor worker, ref bool WeavingFailed)
580 {
581 // put 'this.' onto stack for 'this.syncvar' below
582 worker.Append(worker.Create(OpCodes.Ldarg_0));
583
584 // push 'ref T this.field'
585 worker.Emit(OpCodes.Ldarg_0);
586 // if the netbehaviour class is generic, we need to make the field reference generic as well for correct IL
587 if (netBehaviourSubclass.HasGenericParameters)
588 {
589 worker.Emit(OpCodes.Ldflda, syncVar.MakeHostInstanceGeneric());
590 }
591 else
592 {
593 worker.Emit(OpCodes.Ldflda, syncVar);
594 }
595
596 // If a hook exists, then we need to load the hook delegate on the stack
597 // The hook delegate is created once in the constructor and stored in an instance field
598 // We load the delegate from this instance field to avoid instantiating a new delegate instance every time (drastically reduces allocations)
599 if(syncVarHookDelegates.TryGetValue(syncVar, out (FieldDefinition hookDelegateField, MethodDefinition) value))
600 {
601 // A hook exists. Push this.hookDelegateField onto the stack
602 worker.Emit(OpCodes.Ldarg_0);
603 worker.Emit(OpCodes.Ldfld, value.hookDelegateField);
604 }
605 else
606 {
607 // No hook exists. Push 'null' as hook
608 worker.Emit(OpCodes.Ldnull);
609 }
610
611 // call GeneratedSyncVarDeserialize<T>.
612 // special cases for GameObject/NetworkIdentity/NetworkBehaviour
613 // passing netId too for persistence.
614 if (syncVar.FieldType.Is<UnityEngine.GameObject>())
615 {
616 // reader
617 worker.Emit(OpCodes.Ldarg_1);
618
619 // GameObject setter needs one more parameter: netId field ref
620 FieldDefinition netIdField = syncVarNetIds[syncVar];
621 worker.Emit(OpCodes.Ldarg_0);
622 worker.Emit(OpCodes.Ldflda, netIdField);
623 worker.Emit(OpCodes.Call, weaverTypes.generatedSyncVarDeserialize_GameObject);
624 }
625 else if (syncVar.FieldType.Is<NetworkIdentity>())
626 {
627 // reader
628 worker.Emit(OpCodes.Ldarg_1);
629
630 // NetworkIdentity deserialize needs one more parameter: netId field ref
631 FieldDefinition netIdField = syncVarNetIds[syncVar];
632 worker.Emit(OpCodes.Ldarg_0);
633 worker.Emit(OpCodes.Ldflda, netIdField);
634 worker.Emit(OpCodes.Call, weaverTypes.generatedSyncVarDeserialize_NetworkIdentity);
635 }
636 // handle both NetworkBehaviour and inheritors.
637 // fixes: https://github.com/MirrorNetworking/Mirror/issues/2939
638 else if (syncVar.FieldType.IsDerivedFrom<NetworkBehaviour>() || syncVar.FieldType.Is<NetworkBehaviour>())
639 {
640 // reader
641 worker.Emit(OpCodes.Ldarg_1);
642
643 // NetworkIdentity deserialize needs one more parameter: netId field ref
644 // (actually its a NetworkBehaviourSyncVar type)
645 FieldDefinition netIdField = syncVarNetIds[syncVar];
646 worker.Emit(OpCodes.Ldarg_0);
647 worker.Emit(OpCodes.Ldflda, netIdField);
648 // make generic version of GeneratedSyncVarSetter_NetworkBehaviour<T>
649 MethodReference getFunc = weaverTypes.generatedSyncVarDeserialize_NetworkBehaviour_T.MakeGeneric(assembly.MainModule, syncVar.FieldType);
650 worker.Emit(OpCodes.Call, getFunc);
651 }
652 else
653 {
654 // T value = reader.ReadT();
655 // this is still in IL because otherwise weaver generated
656 // readers/writers don't seem to work in tests.
657 // besides, this also avoids reader.Read<T> overhead.
658 MethodReference readFunc = readers.GetReadFunc(syncVar.FieldType, ref WeavingFailed);
659 if (readFunc == null)
660 {
661 Log.Error($"{syncVar.Name} has unsupported type. Use a supported Mirror type instead", syncVar);
662 WeavingFailed = true;
663 return;
664 }
665 // reader. for 'reader.Read()' below
666 worker.Emit(OpCodes.Ldarg_1);
667 // reader.Read()
668 worker.Emit(OpCodes.Call, readFunc);
669
670 // make generic version of GeneratedSyncVarDeserialize<T>
671 MethodReference generic = weaverTypes.generatedSyncVarDeserialize.MakeGeneric(assembly.MainModule, syncVar.FieldType);
672 worker.Emit(OpCodes.Call, generic);
673 }

676 void GenerateDeSerialization(ref bool WeavingFailed)
677 {
678 const string DeserializeMethodName = "DeserializeSyncVars";
679 if (netBehaviourSubclass.GetMethod(DeserializeMethodName) != null)
680 return;
681
682 if (syncVars.Count == 0)
683 {
684 // no synvars, no need for custom OnDeserialize
685 return;
686 }
687
688 MethodDefinition serialize = new MethodDefinition(DeserializeMethodName,
689 MethodAttributes.Public | MethodAttributes.Virtual | MethodAttributes.HideBySig,
690 weaverTypes.Import(typeof(void)));
691
692 serialize.Parameters.Add(new ParameterDefinition("reader", ParameterAttributes.None, weaverTypes.Import<NetworkReader>()));
693 serialize.Parameters.Add(new ParameterDefinition("initialState", ParameterAttributes.None, weaverTypes.Import<bool>()));
694 ILProcessor serWorker = serialize.Body.GetILProcessor();
695 // setup local for dirty bits
696 serialize.Body.InitLocals = true;
697 VariableDefinition dirtyBitsLocal = new VariableDefinition(weaverTypes.Import<long>());
698 serialize.Body.Variables.Add(dirtyBitsLocal);
699
700 MethodReference baseDeserialize = Resolvers.TryResolveMethodInParents(netBehaviourSubclass.BaseType, assembly, DeserializeMethodName);
701 if (baseDeserialize != null)
702 {
703 // base
704 serWorker.Append(serWorker.Create(OpCodes.Ldarg_0));
705 // reader
706 serWorker.Append(serWorker.Create(OpCodes.Ldarg_1));
707 // initialState
708 serWorker.Append(serWorker.Create(OpCodes.Ldarg_2));
709 serWorker.Append(serWorker.Create(OpCodes.Call, baseDeserialize));
710 }
711
712 // Generates: if (initialState);
713 Instruction initialStateLabel = serWorker.Create(OpCodes.Nop);
714
715 serWorker.Append(serWorker.Create(OpCodes.Ldarg_2));
716 serWorker.Append(serWorker.Create(OpCodes.Brfalse, initialStateLabel));
717
718 foreach (FieldDefinition syncVar in syncVars)
719 {
720 DeserializeField(syncVar, serWorker, ref WeavingFailed);
721 }
722
723 serWorker.Append(serWorker.Create(OpCodes.Ret));
724
725 // Generates: end if (initialState);
726 serWorker.Append(initialStateLabel);
727
728 // get dirty bits
729 serWorker.Append(serWorker.Create(OpCodes.Ldarg_1));
730 serWorker.Append(serWorker.Create(OpCodes.Call, readers.GetReadFunc(weaverTypes.Import<ulong>(), ref WeavingFailed)));
731 serWorker.Append(serWorker.Create(OpCodes.Stloc_0));
732
733 // conditionally read each syncvar
734 // start at number of syncvars in parent
735 int dirtyBit = syncVarAccessLists.GetSyncVarStart(netBehaviourSubclass.BaseType.FullName);
736 foreach (FieldDefinition syncVar in syncVars)
737 {
738 Instruction varLabel = serWorker.Create(OpCodes.Nop);
739
740 // check if dirty bit is set
741 serWorker.Append(serWorker.Create(OpCodes.Ldloc_0));
742 serWorker.Append(serWorker.Create(OpCodes.Ldc_I8, 1L << dirtyBit));
743 serWorker.Append(serWorker.Create(OpCodes.And));
744 serWorker.Append(serWorker.Create(OpCodes.Brfalse, varLabel));
745
746 DeserializeField(syncVar, serWorker, ref WeavingFailed);
747
748 serWorker.Append(varLabel);
749 dirtyBit += 1;
750 }
751
752 // add a log message if needed for debugging
753 //serWorker.Append(serWorker.Create(OpCodes.Ldstr, $"Injected Deserialize {netBehaviourSubclass.Name}"));
754 //serWorker.Append(serWorker.Create(OpCodes.Call, WeaverTypes.logErrorReference));
755
756 serWorker.Append(serWorker.Create(OpCodes.Ret));
757 netBehaviourSubclass.Methods.Add(serialize);
758 }

760 public static bool ReadArguments(MethodDefinition method, Readers readers, Logger Log, ILProcessor worker, RemoteCallType callType, ref bool WeavingFailed)
761 {
762 // read each argument
763 // example result
764 /*
765 CallCmdDoSomething(reader.ReadPackedInt32(), reader.ReadNetworkIdentity());
766 */
767
768 bool skipFirst = callType == RemoteCallType.TargetRpc
769 && TargetRpcProcessor.HasNetworkConnectionParameter(method);
770
771 // arg of calling function, arg 0 is "this" so start counting at 1
772 int argNum = 1;
773 foreach (ParameterDefinition param in method.Parameters)
774 {
775 // NetworkConnection is not sent via the NetworkWriter so skip it here
776 // skip first for NetworkConnection in TargetRpc
777 if (argNum == 1 && skipFirst)
778 {
779 argNum += 1;
780 continue;
781 }
782 // skip SenderConnection in Command
783 if (IsSenderConnection(param, callType))
784 {
785 argNum += 1;
786 continue;
787 }
788
789
790 MethodReference readFunc = readers.GetReadFunc(param.ParameterType, ref WeavingFailed);
791
792 if (readFunc == null)
793 {
794 Log.Error($"{method.Name} has invalid parameter {param}. Unsupported type {param.ParameterType}, use a supported Mirror type instead", method);
795 WeavingFailed = true;
796 return false;
797 }
798
799 worker.Emit(OpCodes.Ldarg_1);
800 worker.Emit(OpCodes.Call, readFunc);
801
802 // conversion.. is this needed?
803 if (param.ParameterType.Is<float>())
804 {
805 worker.Emit(OpCodes.Conv_R4);
806 }
807 else if (param.ParameterType.Is<double>())
808 {
809 worker.Emit(OpCodes.Conv_R8);
810 }
811 }
812 return true;
813 }

876 bool ValidateParameter(MethodReference method, ParameterDefinition param, RemoteCallType callType, bool firstParam, ref bool WeavingFailed)
877 {
878 // need to check this before any type lookups since those will fail since generic types don't resolve
879 if (param.ParameterType.IsGenericParameter)
880 {
881 Log.Error($"{method.Name} cannot have generic parameters", method);
882 WeavingFailed = true;
883 return false;

888
889 if (param.IsOut)

892 WeavingFailed = true;
893 return false;
894 }
895
896 // if not SenderConnection And not TargetRpc NetworkConnection first param
897 if (!isSenderConnection && isNetworkConnection && !(callType == RemoteCallType.TargetRpc && firstParam))
898 {
899 if (callType == RemoteCallType.Command)
900 {
901 Log.Error($"{method.Name} has invalid parameter {param}, Cannot pass NetworkConnections. Instead use 'NetworkConnectionToClient conn = null' to get the sender's connection on the server", method);
902 }
903 else
904 {
905 Log.Error($"{method.Name} has invalid parameter {param}. Cannot pass NetworkConnections", method);
906 }
907 WeavingFailed = true;
908 return false;
909 }
910
911 // sender connection can be optional
912 if (param.IsOptional && !isSenderConnection)
913 {
914 Log.Error($"{method.Name} cannot have optional parameters", method);
915 WeavingFailed = true;
916 return false;
917 }
918
919 return true;
