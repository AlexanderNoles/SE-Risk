49 public override void OnStartClient()
50 {
51 if (isLocalPlayer)
52 {
53 networkID = GetComponent<NetworkIdentity>().netId;
54 intentionallyDisconnecting = false;
55 }
56
57 NetworkManagement.AddPlayerObject(GetComponent<NetworkIdentity>());
58 if (!isLocalPlayer || isOnHost)
59 {
60 return;
61 }
62 else
63 {
64 instance = this;
65 Debug.Log("CLIENT CONNECTED: " + PlayScreenNetworkDataCommunicator.GetTotalNumberOfPlayers() + " in lobby");
66 touchedServer = true;
67
68 //Check if the host has any avaliable slots
69 //if not then disconnect
70 if (PlayScreenNetworkDataCommunicator.GetTotalNumberOfPlayers() > 5)
71 {
72 //Go offline immediately
73 //Notify the playoptions management so it doesn't run the swipe out transition
74 //as the join button is already doing those transitions
75 NetworkManagement.ResetPlayerObjects();
76 PlayOptionsManagement.DontRunDisconnectTransitions();
77 NetworkManagement.UpdateClientNetworkState(NetworkManagement.ClientState.Offline);
78 }
79 else
80 {
81 //Update our local play screen ui
82 //a.k.a remove ability to add AI and change mode
83 //and add number of AI set by host
84 PlayScreenNetworkDataCommunicator.UpdatePlayUI(-1, -1);
85 }
86 }

92 public override void OnStopClient()
93 {
94 if (isLocalPlayer)
95 {
96 instance = null;
97 NetworkManagement.ResetPlayerObjects();
98 }
99
100 if (!isLocalPlayer)
101 {
102 if(isOnHost)
103 {
104 Debug.Log("Connection Lost");
105
106 PlayOptionsManagement.NotifyHostOfLostConnection();
107 }
108
109 NetworkManagement.RemovePlayerObject(GetComponent<NetworkIdentity>());
110 }
111
112 //Someone disconnected mid game so we return to the title screen
113 if (SceneManager.GetActiveScene().buildIndex == 1 && !intentionallyDisconnecting) //Play scene
114 {
115 MenuManagement.SetDefaultMenu(MenuManagement.Menu.Main);
116 NetworkManagement.UpdateClientNetworkState(NetworkManagement.ClientState.Offline);
117 SceneManager.LoadScene(0);
118 }
