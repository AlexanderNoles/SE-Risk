9 // for performance, we (ab)use c# generics to cache the message id in a static field
10 // this is significantly faster than doing the computation at runtime or looking up cached results via Dictionary
11 // generic classes have separate static fields per type specification
12 public static class NetworkMessageId<T> where T : struct, NetworkMessage
13 {
14 // automated message id from type hash.
15 // platform independent via stable hashcode.
16 // => convenient so we don't need to track messageIds across projects
17 // => addons can work with each other without knowing their ids before
18 // => 2 bytes is enough to avoid collisions.
19 // registering a messageId twice will log a warning anyway.
20 public static readonly ushort Id = CalculateId();
21
22 // Gets the 32bit fnv1a hash
23 // To get it down to 16bit but still reduce hash collisions we cant just cast it to ushort
24 // Instead we take the highest 16bits of the 32bit hash and fold them with xor into the lower 16bits
25 // This will create a more uniform 16bit hash, the method is described in:
26 // http://www.isthe.com/chongo/tech/comp/fnv/ in section "Changing the FNV hash size - xor-folding"
27 static ushort CalculateId() => typeof(T).FullName.GetStableHashCode16();
28 }

57 // max message content size (without header) calculation for convenience
58 // -> Transport.GetMaxPacketSize is the raw maximum
59 // -> Every message gets serialized into <<id, content>>
60 // -> Every serialized message get put into a batch with one timestamp per batch
61 // -> Every message in a batch has a varuint size header.
62 // use the worst case VarUInt size for the largest possible
63 // message size = int.max.
64 public static int MaxContentSize(int channelId)
65 {
66 // calculate the max possible size that can fit in a batch
67 int transportMax = Transport.active.GetMaxPacketSize(channelId);
68 return transportMax - IdSize - Batcher.MaxMessageOverhead(transportMax);
69 }

71 // max message size which includes header + content.
72 public static int MaxMessageSize(int channelId) =>
73 MaxContentSize(channelId) + IdSize;

116 internal static NetworkMessageDelegate WrapHandler<T, C>(Action<C, T, int> handler, bool requireAuthentication, bool exceptionsDisconnect)
117 where T : struct, NetworkMessage
118 where C : NetworkConnection
119 => (conn, reader, channelId) =>
120 {
121 // protect against DOS attacks if attackers try to send invalid
122 // data packets to crash the server/client. there are a thousand
123 // ways to cause an exception in data handling:
124 // - invalid headers
125 // - invalid message ids
126 // - invalid data causing exceptions
127 // - negative ReadBytesAndSize prefixes
128 // - invalid utf8 strings
129 // - etc.
130 //
131 // let's catch them all and then disconnect that connection to avoid
132 // further attacks.
133 T message = default;
134 // record start position for NetworkDiagnostics because reader might contain multiple messages if using batching
135 int startPos = reader.Position;
136 try
137 {
138 if (requireAuthentication && !conn.isAuthenticated)
139 {
140 // message requires authentication, but the connection was not authenticated
141 Debug.LogWarning($"Disconnecting connection: {conn}. Received message {typeof(T)} that required authentication, but the user has not authenticated yet");
142 conn.Disconnect();
143 return;
144 }
145
146 //Debug.Log($"ConnectionRecv {conn} msgType:{typeof(T)} content:{BitConverter.ToString(reader.buffer.Array, reader.buffer.Offset, reader.buffer.Count)}");
147
148 // if it is a value type, just use default(T)
149 // otherwise allocate a new instance
150 message = reader.Read<T>();
151 }
152 catch (Exception exception)
153 {
154 // should we disconnect on exceptions?
155 if (exceptionsDisconnect)
156 {
157 Debug.LogError($"Disconnecting connection: {conn} because reading a message of type {typeof(T)} caused an Exception. This can happen if the other side accidentally (or an attacker intentionally) sent invalid data. Reason: {exception}");
158 conn.Disconnect();
159 return;
160 }
161 // otherwise log it but allow the connection to keep playing
162 else
163 {
164 Debug.LogError($"Caught an Exception when reading a message from: {conn} of type {typeof(T)}. Reason: {exception}");
165 return;
166 }
167 }
168 finally
169 {
170 int endPos = reader.Position;
171 // TODO: Figure out the correct channel
172 NetworkDiagnostics.OnReceive(message, channelId, endPos - startPos);
173 }
174
175 // user handler exception should not stop the whole server
176 try
177 {
178 // user implemented handler
179 handler((C)conn, message, channelId);
180 }
181 catch (Exception exception)
182 {
183 // should we disconnect on exceptions?
184 if (exceptionsDisconnect)
185 {
186 Debug.LogError($"Disconnecting connection: {conn} because handling a message of type {typeof(T)} caused an Exception. This can happen if the other side accidentally (or an attacker intentionally) sent invalid data. Reason: {exception}");
187 conn.Disconnect();
188 }
189 // otherwise log it but allow the connection to keep playing
190 else
191 {
192 Debug.LogError($"Caught an Exception when handling a message from: {conn} of type {typeof(T)}. Reason: {exception}");
193 }
194 }
195 };
