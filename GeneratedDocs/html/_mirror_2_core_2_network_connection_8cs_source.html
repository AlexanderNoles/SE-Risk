96 public void Send<T>(T message, int channelId = Channels.Reliable)
97 where T : struct, NetworkMessage
98 {
99 using (NetworkWriterPooled writer = NetworkWriterPool.Get())
100 {
101 // pack message
102 NetworkMessages.Pack(message, writer);
103
104 // validate packet size immediately.
105 // we know how much can fit into one batch at max.
106 // if it's larger, log an error immediately with the type <T>.
107 // previously we only logged in Update() when processing batches,
108 // but there we don't have type information anymore.
109 int max = NetworkMessages.MaxMessageSize(channelId);
110 if (writer.Position > max)
111 {
112 Debug.LogError($"NetworkConnection.Send: message of type {typeof(T)} with a size of {writer.Position} bytes is larger than the max allowed message size in one batch: {max}.\nThe message was dropped, please make it smaller.");
113 return;
114 }
115
116 // send allocation free
117 NetworkDiagnostics.OnSend(message, channelId, writer.Position, 1);
118 Send(writer.ToArraySegment(), channelId);
119 }
120 }

122 // Send stage two: serialized NetworkMessage as ArraySegment<byte>
123 // internal because no one except Mirror should send bytes directly to
124 // the client. they would be detected as a message. send messages instead.
125 // => make sure to validate message<T> size before calling Send<byte>!
126 [MethodImpl(MethodImplOptions.AggressiveInlining)]
127 internal virtual void Send(ArraySegment<byte> segment, int channelId = Channels.Reliable)
128 {
129 //Debug.Log($"ConnectionSend {this} bytes:{BitConverter.ToString(segment.Array, segment.Offset, segment.Count)}");
130
131 // add to batch no matter what.
132 // batching will try to fit as many as possible into MTU.
133 // but we still allow > MTU, e.g. kcp max packet size 144kb.
134 // those are simply sent as single batches.
135 //
136 // IMPORTANT: do NOT send > batch sized messages directly:
137 // - data race: large messages would be sent directly. small
138 // messages would be sent in the batch at the end of frame
139 // - timestamps: if batching assumes a timestamp, then large
140 // messages need that too.
141 //
142 // NOTE: we ALWAYS batch. it's not optional, because the
143 // receiver needs timestamps for NT etc.
144 //
145 // NOTE: we do NOT ValidatePacketSize here yet. the final packet
146 // will be the full batch, including timestamp.
147 GetBatchForChannelId(channelId).AddMessage(segment, NetworkTime.localTime);
148 }

154 internal virtual void Update()
155 {
156 // go through batches for all channels
157 // foreach ((int key, Batcher batcher) in batches) // Unity 2020 doesn't support deconstruct yet
158 foreach (KeyValuePair<int, Batcher> kvp in batches)

163 {

186 // for future reference, here is how Disconnects work in Mirror.
187 //
188 // first, there are two types of disconnects:
189 // * voluntary: the other end simply disconnected
190 // * involuntary: server disconnects a client by itself
191 //
192 // UNET had special (complex) code to handle both cases differently.
193 //
194 // Mirror handles both cases the same way:
195 // * Disconnect is called from TOP to BOTTOM
196 // NetworkServer/Client -> NetworkConnection -> Transport.Disconnect()
197 // * Disconnect is handled from BOTTOM to TOP
198 // Transport.OnDisconnected -> ...
199 //
200 // in other words, calling Disconnect() does no cleanup whatsoever.
201 // it simply asks the transport to disconnect.
202 // then later the transport events will do the clean up.
