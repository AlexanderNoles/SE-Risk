77 protected override bool RawReceiveFrom(out ArraySegment<byte> segment, out int connectionId)
78 {
79 segment = default;
80 connectionId = 0;
81
82 if (relaySocket == null) return false;
83
84 try
85 {
86 // TODO need separate buffer. don't write into result yet. only payload
87
88 if (relaySocket.ReceiveNonBlocking(relayReceiveBuffer, out ArraySegment<byte> content))
89 {
90 using (NetworkReaderPooled reader = NetworkReaderPool.Get(content))
91 {
92 // parse message type
93 if (reader.Remaining == 0)
94 {
95 Debug.LogWarning($"EdgegapServer: message of {content.Count} is too small to parse header.");
96 return false;
97 }
98 byte messageType = reader.ReadByte();
99
100 // handle message type
101 switch (messageType)
102 {
103 case (byte)MessageType.Ping:
104 {
105 // parse state
106 if (reader.Remaining < 1) return false;
107 ConnectionState last = state;
108 state = (ConnectionState)reader.ReadByte();
109
110 // log state changes for debugging.
111 if (state != last) Debug.Log($"EdgegapServer: state updated to: {state}");
112
113 // return true indicates Mirror to keep checking
114 // for further messages.
115 return true;
116 }
117 case (byte)MessageType.Data:
118 {
119 // parse connectionId and payload
120 if (reader.Remaining <= 4)
121 {
122 Debug.LogWarning($"EdgegapServer: message of {content.Count} is too small to parse connId.");
123 return false;
124 }
125
126 connectionId = reader.ReadInt();
127 segment = reader.ReadBytesSegment(reader.Remaining);
128 // Debug.Log($"EdgegapServer: receiving from connId={connectionId}: {segment.ToHexString()}");
129 return true;
130 }
131 // wrong message type. return false, don't throw.
132 default: return false;
133 }
134 }
135 }
136 }
137 catch (SocketException e)
138 {
139 Log.Info($"EdgegapServer: looks like the other end has closed the connection. This is fine: {e}");
140 }
141 return false;
142 }
