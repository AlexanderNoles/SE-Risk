51 public override void Rebuild(NetworkIdentity identity, bool initialize)
52 {
53 // clear newObservers hashset before using it
54 newObservers.Clear();
55
56 // not force hidden?
57 if (identity.visibility != Visibility.ForceHidden)
58 {
59 OnRebuildObservers(identity, newObservers);
60 }
61
62 // IMPORTANT: AFTER rebuilding add own player connection in any case
63 // to ensure player always sees himself no matter what.
64 // -> OnRebuildObservers might clear observers, so we need to add
65 // the player's own connection AFTER. 100% fail safe.
66 // -> fixes https://github.com/vis2k/Mirror/issues/692 where a
67 // player might teleport out of the ProximityChecker's cast,
68 // losing the own connection as observer.
69 if (identity.connectionToClient != null)
70 {
71 newObservers.Add(identity.connectionToClient);
72 }
73
74 bool changed = false;
75
76 // add all newObservers that aren't in .observers yet
77 foreach (NetworkConnectionToClient conn in newObservers)
78 {
79 // only add ready connections.
80 // otherwise the player might not be in the world yet or anymore
81 if (conn != null && conn.isReady)
82 {
83 if (initialize || !identity.observers.ContainsKey(conn.connectionId))
84 {
85 // new observer
86 conn.AddToObserving(identity);
87 // Debug.Log($"New Observer for {gameObject} {conn}");
88 changed = true;
89 }
90 }
91 }
92
93 // remove all old .observers that aren't in newObservers anymore
94 foreach (NetworkConnectionToClient conn in identity.observers.Values)
95 {
96 if (!newObservers.Contains(conn))
97 {
98 // removed observer
99 conn.RemoveFromObserving(identity, false);
100 // Debug.Log($"Removed Observer for {gameObject} {conn}");
101 changed = true;
102 }
103 }
104
105 // copy new observers to observers
106 if (changed)
107 {
108 identity.observers.Clear();
109 foreach (NetworkConnectionToClient conn in newObservers)
110 {
111 if (conn != null && conn.isReady)
112 identity.observers.Add(conn.connectionId, conn);
113 }
114 }
115
116 // special case for host mode: we use SetHostVisibility to hide
117 // NetworkIdentities that aren't in observer range from host.
118 // this is what games like Dota/Counter-Strike do too, where a host
119 // does NOT see all players by default. they are in memory, but
120 // hidden to the host player.
121 //
122 // this code is from UNET, it's a bit strange but it works:
123 // * it hides newly connected identities in host mode
124 // => that part was the intended behaviour
125 // * it hides ALL NetworkIdentities in host mode when the host
126 // connects but hasn't selected a character yet
127 // => this only works because we have no .localConnection != null
128 // check. at this stage, localConnection is null because
129 // StartHost starts the server first, then calls this code,
130 // then starts the client and sets .localConnection. so we can
131 // NOT add a null check without breaking host visibility here.
132 // * it hides ALL NetworkIdentities in server-only mode because
133 // observers never contain the 'null' .localConnection
134 // => that was not intended, but let's keep it as it is so we
135 // don't break anything in host mode. it's way easier than
136 // iterating all identities in a special function in StartHost.
137 if (initialize)
138 {
139 if (!newObservers.Contains(NetworkServer.localConnection))
140 {
141 SetHostVisibility(identity, false);
142 }
143 }
144 }
145 }
