19 // LateUpdate so that all spawns/despawns/changes are done
20 [ServerCallback]
21 void LateUpdate()
22 {
23 // Rebuild all dirty matches
24 // dirtyMatches will be empty if no matches changed members
25 // by spawning or destroying or changing matchId in this frame.
26 foreach (Guid dirtyMatch in dirtyMatches)
27 {
28 // rebuild always, even if matchObjects[dirtyMatch] is empty.
29 // Players might have left the match, but they may still be spawned.
30 RebuildMatchObservers(dirtyMatch);
31
32 // clean up empty entries in the dict
33 if (matchObjects[dirtyMatch].Count == 0)
34 matchObjects.Remove(dirtyMatch);
35 }
36
37 dirtyMatches.Clear();
38
39 matchCount = (ushort)matchObjects.Count;
40 }
41
42 [ServerCallback]
43 void RebuildMatchObservers(Guid matchId)
44 {
45 foreach (NetworkMatch networkMatch in matchObjects[matchId])
46 if (networkMatch.netIdentity != null)
47 NetworkServer.RebuildObservers(networkMatch.netIdentity, false);
48 }

50 // called by NetworkMatch.matchId setter
51 [ServerCallback]
52 internal void OnMatchChanged(NetworkMatch networkMatch, Guid oldMatch)
53 {
54 // This object is in a new match so observers in the prior match
55 // and the new match need to rebuild their respective observers lists.
56
57 // Remove this object from the hashset of the match it just left
58 // Guid.Empty is never a valid matchId
59 if (oldMatch != Guid.Empty)
60 {
61 dirtyMatches.Add(oldMatch);
62 matchObjects[oldMatch].Remove(networkMatch);
63 }
64
65 // Guid.Empty is never a valid matchId
66 if (networkMatch.matchId == Guid.Empty)
67 return;
68
69 dirtyMatches.Add(networkMatch.matchId);
70
71 // Make sure this new match is in the dictionary
72 if (!matchObjects.ContainsKey(networkMatch.matchId))
73 matchObjects[networkMatch.matchId] = new HashSet<NetworkMatch>();
74
75 // Add this object to the hashset of the new match
76 matchObjects[networkMatch.matchId].Add(networkMatch);
77 }

106 [ServerCallback]
107 public override void OnDestroyed(NetworkIdentity identity)
108 {
109 // Don't RebuildSceneObservers here - that will happen in LateUpdate.
110 // Multiple objects could be destroyed in same frame and we don't
111 // want to rebuild for each one...let LateUpdate do it once.
112 // We must add the current match to dirtyMatches for LateUpdate to rebuild it.
113 if (identity.TryGetComponent(out NetworkMatch currentMatch))
114 {
115 if (currentMatch.matchId != Guid.Empty &&
116 matchObjects.TryGetValue(currentMatch.matchId, out HashSet<NetworkMatch> objects) &&
117 objects.Remove(currentMatch))
118 dirtyMatches.Add(currentMatch.matchId);
119 }

144 [ServerCallback]
145 public override void OnRebuildObservers(NetworkIdentity identity, HashSet<NetworkConnectionToClient> newObservers)
146 {
147 if (!identity.TryGetComponent(out NetworkMatch networkMatch))
148 return;
149
150 // Guid.Empty is never a valid matchId
151 if (networkMatch.matchId == Guid.Empty)
152 return;
153
154 // Abort if this match hasn't been created yet by OnSpawned or OnMatchChanged
155 if (!matchObjects.TryGetValue(networkMatch.matchId, out HashSet<NetworkMatch> objects))
156 return;
157
158 // Add everything in the hashset for this object's current match
159 foreach (NetworkMatch netMatch in objects)
160 if (netMatch.netIdentity != null && netMatch.netIdentity.connectionToClient != null)
161 newObservers.Add(netMatch.netIdentity.connectionToClient);
