90 // for windows sockets, there's a rare issue where when using
91 // a server socket with multiple clients, if one of the clients
92 // is closed, the single server socket throws exceptions when
93 // sending/receiving. even if the socket is made for N clients.
94 //
95 // this actually happened to one of our users:
96 // https://github.com/MirrorNetworking/Mirror/issues/3611
97 //
98 // here's the in-depth explanation & solution:
99 //
100 // "As you may be aware, if a host receives a packet for a UDP
101 // port that is not currently bound, it may send back an ICMP
102 // "Port Unreachable" message. Whether or not it does this is
103 // dependent on the firewall, private/public settings, etc.
104 // On localhost, however, it will pretty much always send this
105 // packet back.
106 //
107 // Now, on Windows (and only on Windows), by default, a received
108 // ICMP Port Unreachable message will close the UDP socket that
109 // sent it; hence, the next time you try to receive on the
110 // socket, it will throw an exception because the socket has
111 // been closed by the OS.
112 //
113 // Obviously, this causes a headache in the multi-client,
114 // single-server socket set-up you have here, but luckily there
115 // is a fix:
116 //
117 // You need to utilise the not-often-required SIO_UDP_CONNRESET
118 // Winsock control code, which turns off this built-in behaviour
119 // of automatically closing the socket.
120 //
121 // Note that this ioctl code is only supported on Windows
122 // (XP and later), not on Linux, since it is provided by the
123 // Winsock extensions. Of course, since the described behavior
124 // is only the default behavior on Windows, this omission is not
125 // a major loss. If you are attempting to create a
126 // cross-platform library, you should cordon this off as
127 // Windows-specific code."
128 // https://stackoverflow.com/questions/74327225/why-does-sending-via-a-udpclient-cause-subsequent-receiving-to-fail
129 if (RuntimeInformation.IsOSPlatform(OSPlatform.Windows))
130 {
131 const uint IOC_IN = 0x80000000U;
132 const uint IOC_VENDOR = 0x18000000U;
133 const int SIO_UDP_CONNRESET = unchecked((int)(IOC_IN | IOC_VENDOR | 12));
134 socket.IOControl(SIO_UDP_CONNRESET, new byte[] { 0x00 }, null);
135 }
136
137 socket.Bind(new IPEndPoint(IPAddress.IPv6Any, port));
138 return socket;
139 }
140 else
141 {
142 // IPv4 socket @ "0.0.0.0" : port
143 Socket socket = new Socket(AddressFamily.InterNetwork, SocketType.Dgram, ProtocolType.Udp);
144 socket.Bind(new IPEndPoint(IPAddress.Any, port));
145 return socket;
146 }
147 }

196 // io - input.
197 // virtual so it may be modified for relays, nonalloc workaround, etc.
198 // https://github.com/vis2k/where-allocation
199 // bool return because not all receives may be valid.
200 // for example, relay may expect a certain header.
201 protected virtual bool RawReceiveFrom(out ArraySegment<byte> segment, out int connectionId)
202 {
203 segment = default;
204 connectionId = 0;
205 if (socket == null) return false;
206
207 try
208 {
209 if (socket.ReceiveFromNonBlocking(rawReceiveBuffer, out segment, ref newClientEP))
210 {
211 // set connectionId to hash from endpoint
212 connectionId = Common.ConnectionHash(newClientEP);
213 return true;
214 }
215 }
216 catch (SocketException e)
217 {
218 // NOTE: SocketException is not a subclass of IOException.
219 // the other end closing the connection is not an 'error'.
220 // but connections should never just end silently.
221 // at least log a message for easier debugging.
222 Log.Info($"[KCP] Server: ReceiveFrom failed: {e}");
223 }
224
225 return false;
226 }

250 protected virtual KcpServerConnection CreateConnection(int connectionId)
251 {
252 // generate a random cookie for this connection to avoid UDP spoofing.
253 // needs to be random, but without allocations to avoid GC.
254 uint cookie = Common.GenerateCookie();
255
256 // create empty connection without peer first.
257 // we need it to set up peer callbacks.
258 // afterwards we assign the peer.
259 // events need to be wrapped with connectionIds
260 KcpServerConnection connection = new KcpServerConnection(
261 OnConnectedCallback,
262 (message, channel) => OnData(connectionId, message, channel),
263 OnDisconnectedCallback,
264 (error, reason) => OnError(connectionId, error, reason),
265 (data) => RawSend(connectionId, data),
266 config,
267 cookie,
268 newClientEP);
269
270 return connection;

304 // receive + add + process once.
305 // best to call this as long as there is more data to receive.
306 void ProcessMessage(ArraySegment<byte> segment, int connectionId)
307 {
308 //Log.Info($"[KCP] server raw recv {msgLength} bytes = {BitConverter.ToString(buffer, 0, msgLength)}");
309
310 // is this a new connection?
311 if (!connections.TryGetValue(connectionId, out KcpServerConnection connection))
312 {
313 // create a new KcpConnection based on last received
314 // EndPoint. can be overwritten for where-allocation.
315 connection = CreateConnection(connectionId);
316
317 // DO NOT add to connections yet. only if the first message
318 // is actually the kcp handshake. otherwise it's either:
319 // * random data from the internet
320 // * or from a client connection that we just disconnected
321 // but that hasn't realized it yet, still sending data
322 // from last session that we should absolutely ignore.
323 //
324 //
325 // TODO this allocates a new KcpConnection for each new
326 // internet connection. not ideal, but C# UDP Receive
327 // already allocated anyway.
328 //
329 // expecting a MAGIC byte[] would work, but sending the raw
330 // UDP message without kcp's reliability will have low
331 // probability of being received.
332 //
333 // for now, this is fine.
334
335
336 // now input the message & process received ones
337 // connected event was set up.
338 // tick will process the first message and adds the
339 // connection if it was the handshake.
340 connection.RawInput(segment);
341 connection.TickIncoming();
342
343 // again, do not add to connections.
344 // if the first message wasn't the kcp handshake then
345 // connection will simply be garbage collected.
346 }
347 // existing connection: simply input the message into kcp
348 else
349 {
350 connection.RawInput(segment);
351 }
352 }

354 // process incoming messages. should be called before updating the world.
355 // virtual because relay may need to inject their own ping or similar.
356 readonly HashSet<int> connectionsToRemove = new HashSet<int>();
357 public virtual void TickIncoming()
358 {
359 // input all received messages into kcp
360 while (RawReceiveFrom(out ArraySegment<byte> segment, out int connectionId))
361 {
362 ProcessMessage(segment, connectionId);
363 }
364
365 // process inputs for all server connections
366 // (even if we didn't receive anything. need to tick ping etc.)
367 foreach (KcpServerConnection connection in connections.Values)
368 {
369 connection.TickIncoming();
370 }
371
372 // remove disconnected connections
373 // (can't do it in connection.OnDisconnected because Tick is called
374 // while iterating connections)
375 foreach (int connectionId in connectionsToRemove)
376 {
377 connections.Remove(connectionId);
378 }
379 connectionsToRemove.Clear();
380 }
381
382 // process outgoing messages. should be called after updating the world.
383 // virtual because relay may need to inject their own ping or similar.
384 public virtual void TickOutgoing()
385 {
386 // flush all server connections
387 foreach (KcpServerConnection connection in connections.Values)
388 {
389 connection.TickOutgoing();
390 }
391 }
392
393 // process incoming and outgoing for convenience.
394 // => ideally call ProcessIncoming() before updating the world and
395 // ProcessOutgoing() after updating the world for minimum latency
396 public virtual void Tick()
397 {
398 TickIncoming();
399 TickOutgoing();
400 }
