461 private List<Territory> EvaluateInterruptAttack(out bool attempt)
462 {
463 List<Continent> continentsOwnedByEnemies = new List<Continent>();
464 foreach(Continent continent in System.Enum.GetValues(typeof(Continent)))
465 {
466 int owner = Map.GetContinentOwner(continent);
467 if (owner != -1 && owner != GetIndex())
468 {
469 continentsOwnedByEnemies.Add(continent);
470 }
471 }
472 int minTroopRequirement=10000;
473 List<Territory> bestRoute = new List<Territory>();
474 foreach(Continent continent in continentsOwnedByEnemies)
475 {
476 foreach(Territory endTerritory in Map.continents[continent])
477 {
478 foreach(Territory startTerritory in territories)
479 {
480 int routeCost =0;
481 List<Territory> thisRoute = new List<Territory>();
482 List<Pathfinding.INode> newRoute = RouteBetweenTerritories(startTerritory, endTerritory);
483 foreach (EnemyNode node in newRoute)
484 {
485 thisRoute.Add(node.territory);
486 if (node.territory.GetOwner() != GetIndex())
487 {
488 routeCost += (node.territory.GetCurrentTroops()*2)+1;
489 }
490 }
491 if (newRoute.Count>0 && routeCost < minTroopRequirement - Random.Range(0,difficulty))
492 {
493 minTroopRequirement = routeCost;
494 bestRoute = thisRoute;
495 }
496 }
497 }
498 }
499 if (bestRoute.Count>0 && bestRoute[0].GetCurrentTroops() > minTroopRequirement - Random.Range(0, difficulty))
500 {
501 attempt = true;
502 }
503 else
504 {
505 attempt = false;
