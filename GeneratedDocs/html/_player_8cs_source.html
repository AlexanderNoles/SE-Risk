426 private Territory EvaluateExpansion(out bool attempt)
427 {
428 int currentMinNeighbours = 10000;
429 Territory currentTerritoryToExpand = null;
430 //Finds the territory with the fewest troops neighbouring it, not necessarily the fewest enemy territories
431 foreach (Territory territory in territories)
432 {
433 Debug.Log(territory);
434 int enemyNeighbourCount = 0;
435 List<Territory> enemyNeighbours = new List<Territory>();
436 foreach(Territory neighbour in territory.GetNeighbours())
437 {
438 if (neighbour.GetOwner()!= territory.GetOwner())
439 {
440 enemyNeighbourCount+=neighbour.GetCurrentTroops();
441 }
442 }
443 if (enemyNeighbourCount < currentMinNeighbours && enemyNeighbourCount!=0)
444 {
445 if (territory.GetOwner()==GetIndex())
446 {
447 currentTerritoryToExpand = territory;
448 currentMinNeighbours = enemyNeighbourCount;
449 }
450 }
451 }
452
453 attempt = false;
454 //decides if we attempt the attack
455 if (currentTerritoryToExpand != null && currentTerritoryToExpand.GetCurrentTroops() > currentMinNeighbours - Random.Range(0, difficulty)*2)

467 private List<Territory> EvaluateInterruptAttack(out bool attempt)
468 {
469 //finds the continents owned completely by an enemy
470 List<Continent> continentsOwnedByEnemies = new List<Continent>();
471 foreach(Continent continent in System.Enum.GetValues(typeof(Continent)))
472 {
473 int owner = Map.GetContinentOwner(continent);
474 if (owner != -1 && owner != GetIndex())
475 {
476 continentsOwnedByEnemies.Add(continent);
477 }
478 }
479 int minTroopRequirement=10000;
480 List<Territory> bestRoute = new List<Territory>();
481 //Finds the shortest route to break enemy control of each continent
482 foreach(Continent continent in continentsOwnedByEnemies)
483 {
484 foreach(Territory endTerritory in Map.continents[continent])
485 {
486 foreach(Territory startTerritory in territories)
487 {
488 int routeCost =0;
489 List<Territory> thisRoute = new List<Territory>();
490 List<Pathfinding.INode> newRoute = RouteBetweenTerritories(startTerritory, endTerritory);
491 foreach (EnemyNode node in newRoute)
492 {
493 thisRoute.Add(node.territory);
494 if (node.territory.GetOwner() != GetIndex())
495 {
496 routeCost += (node.territory.GetCurrentTroops()*2)+1;
497 }
498 }
499 if (newRoute.Count>0 && routeCost < minTroopRequirement - Random.Range(0,difficulty))
500 {
501 minTroopRequirement = routeCost;
502 bestRoute = thisRoute;
503 }
504 }
505 }
506 }
507 //determines if we attempt the attack
508 if (bestRoute.Count>0 && bestRoute[0].GetCurrentTroops() > minTroopRequirement - Random.Range(0, difficulty))
509 {
510 attempt = true;
511 }
512 else
513 {
514 attempt = false;
515 }
516 return bestRoute;
