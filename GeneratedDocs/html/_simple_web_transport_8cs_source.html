178 public override void ClientConnect(Uri uri)
179 {

193 client.onDisconnect += () =>

195 OnClientDisconnected.Invoke();
196 // clear client here after disconnect event has been sent
197 // there should be no more messages after disconnect
198 client = null;
199 };

201 client.onData += (ArraySegment<byte> data) => OnClientDataReceived.Invoke(data, Channels.Reliable);
202
203 // We will not invoke OnClientError if minLogLevel is set to None
204 // We only send the full exception if minLogLevel is set to Verbose
