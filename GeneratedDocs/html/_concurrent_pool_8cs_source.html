1// Pool to avoid allocations (from libuv2k)
2// API consistent with Microsoft's ObjectPool<T>.
3// concurrent for thread safe access.
4//
5// currently not in use. keep it in case we need it again.
6using System;
7using System.Collections.Concurrent;
8using System.Runtime.CompilerServices;
9
10namespace Mirror
11{
12 public class ConcurrentPool<T>
13 {
14 // Mirror is single threaded, no need for concurrent collections
15 // concurrent bag is for items who's order doesn't matter.
16 // just about right for our use case here.
17 readonly ConcurrentBag<T> objects = new ConcurrentBag<T>();
18
19 // some types might need additional parameters in their constructor, so
20 // we use a Func<T> generator
21 readonly Func<T> objectGenerator;
22
23 public ConcurrentPool(Func<T> objectGenerator, int initialCapacity)
24 {
25 this.objectGenerator = objectGenerator;
26
27 // allocate an initial pool so we have fewer (if any)
28 // allocations in the first few frames (or seconds).
29 for (int i = 0; i < initialCapacity; ++i)
30 objects.Add(objectGenerator());
31 }
32
33 // take an element from the pool, or create a new one if empty
34 [MethodImpl(MethodImplOptions.AggressiveInlining)]
35 public T Get() => objects.TryTake(out T obj) ? obj : objectGenerator();
36
37 // return an element to the pool
38 [MethodImpl(MethodImplOptions.AggressiveInlining)]
39 public void Return(T item) => objects.Add(item);
40
41 // count to see how many objects are in the pool. useful for tests.
42 public int Count => objects.Count;
43 }
44}
