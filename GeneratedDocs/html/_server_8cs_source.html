76 // the listener thread's listen function
77 // note: no maxConnections parameter. high level API should handle that.
78 // (Transport can't send a 'too full' message anyway)
79 void Listen(int port)
80 {
81 // absolutely must wrap with try/catch, otherwise thread
82 // exceptions are silent
83 try
84 {
85 // start listener on all IPv4 and IPv6 address via .Create
86 listener = TcpListener.Create(port);
87 listener.Server.NoDelay = NoDelay;
88 // IMPORTANT: do not set send/receive timeouts on listener.
89 // On linux setting the recv timeout will cause the blocking
90 // Accept call to timeout with EACCEPT (which mono interprets
91 // as EWOULDBLOCK).
92 // https://stackoverflow.com/questions/1917814/eagain-error-for-accept-on-blocking-socket/1918118#1918118
93 // => fixes https://github.com/vis2k/Mirror/issues/2695
94 //
95 //listener.Server.SendTimeout = SendTimeout;
96 //listener.Server.ReceiveTimeout = ReceiveTimeout;
97 listener.Start();
98 Log.Info("Server: listening port=" + port);
99
100 // keep accepting new clients
101 while (true)
102 {
103 // wait and accept new client
104 // note: 'using' sucks here because it will try to
105 // dispose after thread was started but we still need it
106 // in the thread
107 TcpClient client = listener.AcceptTcpClient();
108
109 // set socket options
110 client.NoDelay = NoDelay;
111 client.SendTimeout = SendTimeout;
112 client.ReceiveTimeout = ReceiveTimeout;
113
114 // generate the next connection id (thread safely)
115 int connectionId = NextConnectionId();
116
117 // add to dict immediately
118 ConnectionState connection = new ConnectionState(client, MaxMessageSize);
119 clients[connectionId] = connection;
120
121 // spawn a send thread for each client
122 Thread sendThread = new Thread(() =>
123 {
124 // wrap in try-catch, otherwise Thread exceptions
125 // are silent
126 try
127 {
128 // run the send loop
129 // IMPORTANT: DO NOT SHARE STATE ACROSS MULTIPLE THREADS!
130 ThreadFunctions.SendLoop(connectionId, client, connection.sendPipe, connection.sendPending);
131 }
132 catch (ThreadAbortException)
133 {
134 // happens on stop. don't log anything.
135 // (we catch it in SendLoop too, but it still gets
136 // through to here when aborting. don't show an
137 // error.)
138 }
139 catch (Exception exception)
140 {
141 Log.Error("Server send thread exception: " + exception);
142 }
143 });
144 sendThread.IsBackground = true;
145 sendThread.Start();
146
147 // spawn a receive thread for each client
148 Thread receiveThread = new Thread(() =>
149 {
150 // wrap in try-catch, otherwise Thread exceptions
151 // are silent
152 try
153 {
154 // run the receive loop
155 // (receive pipe is shared across all loops)
156 ThreadFunctions.ReceiveLoop(connectionId, client, MaxMessageSize, receivePipe, ReceiveQueueLimit);
157
158 // IMPORTANT: do NOT remove from clients after the
159 // thread ends. need to do it in Tick() so that the
160 // disconnect event in the pipe is still processed.
161 // (removing client immediately would mean that the
162 // pipe is lost and the disconnect event is never
163 // processed)
164
165 // sendthread might be waiting on ManualResetEvent,
166 // so let's make sure to end it if the connection
167 // closed.
168 // otherwise the send thread would only end if it's
169 // actually sending data while the connection is
170 // closed.
171 sendThread.Interrupt();
172 }
173 catch (Exception exception)
174 {
175 Log.Error("Server client thread exception: " + exception);
176 }
177 });
178 receiveThread.IsBackground = true;
179 receiveThread.Start();
180 }
181 }
182 catch (ThreadAbortException exception)
183 {
184 // UnityEditor causes AbortException if thread is still
185 // running when we press Play again next time. that's okay.
186 Log.Info("Server thread aborted. That's okay. " + exception);
187 }
188 catch (SocketException exception)
189 {
190 // calling StopServer will interrupt this thread with a
191 // 'SocketException: interrupted'. that's okay.
192 Log.Info("Server Thread stopped. That's okay. " + exception);
193 }
194 catch (Exception exception)
195 {
196 // something went wrong. probably important.
197 Log.Error("Server Exception: " + exception);
198 }

201 // start listening for new connections in a background thread and spawn
202 // a new thread for each one.
203 public bool Start(int port)
204 {
205 // not if already started
206 if (Active) return false;
207
208 // create receive pipe with max message size for pooling
209 // => create new pipes every time!
210 // if an old receive thread is still finishing up, it might still
211 // be using the old pipes. we don't want to risk any old data for
212 // our new start here.
213 receivePipe = new MagnificentReceivePipe(MaxMessageSize);
214
215 // start the listener thread
216 // (on low priority. if main thread is too busy then there is not
217 // much value in accepting even more clients)
218 Log.Info("Server: Start port=" + port);
219 listenerThread = new Thread(() => { Listen(port); });
220 listenerThread.IsBackground = true;
221 listenerThread.Priority = ThreadPriority.BelowNormal;
222 listenerThread.Start();
223 return true;
224 }

226 public void Stop()
227 {
228 // only if started
229 if (!Active) return;
230
231 Log.Info("Server: stopping...");
232
233 // stop listening to connections so that no one can connect while we
234 // close the client connections
235 // (might be null if we call Stop so quickly after Start that the
236 // thread was interrupted before even creating the listener)
237 listener?.Stop();
238
239 // kill listener thread at all costs. only way to guarantee that
240 // .Active is immediately false after Stop.
241 // -> calling .Join would sometimes wait forever
242 listenerThread?.Interrupt();
243 listenerThread = null;
244
245 // close all client connections
246 foreach (KeyValuePair<int, ConnectionState> kvp in clients)
247 {
248 TcpClient client = kvp.Value.client;
249 // close the stream if not closed yet. it may have been closed
250 // by a disconnect already, so use try/catch
251 try { client.GetStream().Close(); } catch {}
252 client.Close();
253 }
254
255 // clear clients list
256 clients.Clear();
257
258 // reset the counter in case we start up again so
259 // clients get connection ID's starting from 1
260 counter = 0;
261 }

263 // send message to client using socket connection.
264 // arraysegment for allocation free sends later.
265 // -> the segment's array is only used until Send() returns!
266 public bool Send(int connectionId, ArraySegment<byte> message)
267 {
268 // respect max message size to avoid allocation attacks.
269 if (message.Count <= MaxMessageSize)
270 {
271 // find the connection
272 if (clients.TryGetValue(connectionId, out ConnectionState connection))
273 {
274 // check send pipe limit
275 if (connection.sendPipe.Count < SendQueueLimit)
276 {
277 // add to thread safe send pipe and return immediately.
278 // calling Send here would be blocking (sometimes for long
279 // times if other side lags or wire was disconnected)
280 connection.sendPipe.Enqueue(message);
281 connection.sendPending.Set(); // interrupt SendThread WaitOne()
282 return true;
283 }
284 // disconnect if send queue gets too big.
285 // -> avoids ever growing queue memory if network is slower
286 // than input
287 // -> disconnecting is great for load balancing. better to
288 // disconnect one connection than risking every
289 // connection / the whole server
290 //
291 // note: while SendThread always grabs the WHOLE send queue
292 // immediately, it's still possible that the sending
293 // blocks for so long that the send queue just gets
294 // way too big. have a limit - better safe than sorry.
295 else
296 {
297 // log the reason
298 Log.Warning($"Server.Send: sendPipe for connection {connectionId} reached limit of {SendQueueLimit}. This can happen if we call send faster than the network can process messages. Disconnecting this connection for load balancing.");
299
300 // just close it. send thread will take care of the rest.
301 connection.client.Close();
302 return false;
303 }
304 }
305
306 // sending to an invalid connectionId is expected sometimes.
307 // for example, if a client disconnects, the server might still
308 // try to send for one frame before it calls GetNextMessages
309 // again and realizes that a disconnect happened.
310 // so let's not spam the console with log messages.
311 //Logger.Log("Server.Send: invalid connectionId: " + connectionId);
312 return false;
313 }
314 Log.Error("Server.Send: message too big: " + message.Count + ". Limit: " + MaxMessageSize);
315 return false;
316 }

343 // tick: processes up to 'limit' messages for each connection
344 // => limit parameter to avoid deadlocks / too long freezes if server or
345 // client is too slow to process network load
346 // => Mirror & DOTSNET need to have a process limit anyway.
347 // might as well do it here and make life easier.
348 // => returns amount of remaining messages to process, so the caller
349 // can call tick again as many times as needed (or up to a limit)
350 //
351 // Tick() may process multiple messages, but Mirror needs a way to stop
352 // processing immediately if a scene change messages arrives. Mirror
353 // can't process any other messages during a scene change.
354 // (could be useful for others too)
355 // => make sure to allocate the lambda only once in transports
356 public int Tick(int processLimit, Func<bool> checkEnabled = null)
357 {
358 // only if pipe was created yet (after start())
359 if (receivePipe == null)
360 return 0;
361
362 // process up to 'processLimit' messages for this connection
363 for (int i = 0; i < processLimit; ++i)
364 {
365 // check enabled in case a Mirror scene message arrived
366 if (checkEnabled != null && !checkEnabled())
367 break;
368
369 // peek first. allows us to process the first queued entry while
370 // still keeping the pooled byte[] alive by not removing anything.
371 if (receivePipe.TryPeek(out int connectionId, out EventType eventType, out ArraySegment<byte> message))
372 {
373 switch (eventType)
374 {
375 case EventType.Connected:
376 OnConnected?.Invoke(connectionId);
377 break;
378 case EventType.Data:
379 OnData?.Invoke(connectionId, message);
380 break;
381 case EventType.Disconnected:
382 OnDisconnected?.Invoke(connectionId);
383 // remove disconnected connection now that the final
384 // disconnected message was processed.
385 clients.TryRemove(connectionId, out ConnectionState _);
386 break;
387 }
388
389 // IMPORTANT: now dequeue and return it to pool AFTER we are
390 // done processing the event.
391 receivePipe.TryDequeue();
392 }
393 // no more messages. stop the loop.
394 else break;
395 }
396
397 // return what's left to process for next time
398 return receivePipe.TotalCount;
399 }
