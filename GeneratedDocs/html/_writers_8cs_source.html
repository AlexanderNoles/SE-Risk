75 //Throws GenerateWriterException when writer could not be generated for type
76 MethodReference GenerateWriter(TypeReference variableReference, ref bool WeavingFailed)
77 {
78 if (variableReference.IsByReference)
79 {
80 throw new GenerateWriterException($"Cannot pass {variableReference.Name} by reference", variableReference);
81 }
82
83 // Arrays are special, if we resolve them, we get the element type,
84 // e.g. int[] resolves to int
85 // therefore process this before checks below
86 if (variableReference.IsArray)
87 {
88 if (variableReference.IsMultidimensionalArray())
89 {
90 throw new GenerateWriterException($"{variableReference.Name} is an unsupported type. Multidimensional arrays are not supported", variableReference);
91 }
92 TypeReference elementType = variableReference.GetElementType();
93 return GenerateCollectionWriter(variableReference, elementType, nameof(NetworkWriterExtensions.WriteArray), ref WeavingFailed);
94 }
95
96 if (variableReference.Resolve()?.IsEnum ?? false)
97 {
98 // serialize enum as their base type
99 return GenerateEnumWriteFunc(variableReference, ref WeavingFailed);
100 }
101
102 // check for collections
103 if (variableReference.Is(typeof(ArraySegment<>)))
104 {
105 GenericInstanceType genericInstance = (GenericInstanceType)variableReference;
106 TypeReference elementType = genericInstance.GenericArguments[0];
107
108 return GenerateCollectionWriter(variableReference, elementType, nameof(NetworkWriterExtensions.WriteArraySegment), ref WeavingFailed);
109 }
110 if (variableReference.Is(typeof(List<>)))
111 {
112 GenericInstanceType genericInstance = (GenericInstanceType)variableReference;
113 TypeReference elementType = genericInstance.GenericArguments[0];
114
115 return GenerateCollectionWriter(variableReference, elementType, nameof(NetworkWriterExtensions.WriteList), ref WeavingFailed);
116 }
117
118 // handle both NetworkBehaviour and inheritors.
119 // fixes: https://github.com/MirrorNetworking/Mirror/issues/2939
120 if (variableReference.IsDerivedFrom<NetworkBehaviour>() || variableReference.Is<NetworkBehaviour>())
121 {
122 return GetNetworkBehaviourWriter(variableReference);
123 }
124
125 // check for invalid types
126 TypeDefinition variableDefinition = variableReference.Resolve();
127 if (variableDefinition == null)
128 {
129 throw new GenerateWriterException($"{variableReference.Name} is not a supported type. Use a supported type or provide a custom writer", variableReference);
130 }
131 if (variableDefinition.IsDerivedFrom<UnityEngine.Component>())
132 {
133 throw new GenerateWriterException($"Cannot generate writer for component type {variableReference.Name}. Use a supported type or provide a custom writer", variableReference);
134 }
135 if (variableReference.Is<UnityEngine.Object>())
136 {
137 throw new GenerateWriterException($"Cannot generate writer for {variableReference.Name}. Use a supported type or provide a custom writer", variableReference);
138 }
139 if (variableReference.Is<UnityEngine.ScriptableObject>())
140 {
141 throw new GenerateWriterException($"Cannot generate writer for {variableReference.Name}. Use a supported type or provide a custom writer", variableReference);
142 }
143 if (variableDefinition.HasGenericParameters)
144 {
145 throw new GenerateWriterException($"Cannot generate writer for generic type {variableReference.Name}. Use a supported type or provide a custom writer", variableReference);
146 }
147 if (variableDefinition.IsInterface)
148 {
149 throw new GenerateWriterException($"Cannot generate writer for interface {variableReference.Name}. Use a supported type or provide a custom writer", variableReference);
150 }
151 if (variableDefinition.IsAbstract)
152 {
153 throw new GenerateWriterException($"Cannot generate writer for abstract class {variableReference.Name}. Use a supported type or provide a custom writer", variableReference);
154 }
155
156 // generate writer for class/struct
157 return GenerateClassOrStructWriterFunction(variableReference, ref WeavingFailed);
158 }

194 MethodDefinition GenerateWriterFunc(TypeReference variable)
195 {
196 string functionName = $"_Write_{variable.FullName}";
197 // create new writer for this type
198 MethodDefinition writerFunc = new MethodDefinition(functionName,
199 MethodAttributes.Public |
200 MethodAttributes.Static |
201 MethodAttributes.HideBySig,
202 weaverTypes.Import(typeof(void)));
203
204 writerFunc.Parameters.Add(new ParameterDefinition("writer", ParameterAttributes.None, weaverTypes.Import<NetworkWriter>()));
205 writerFunc.Parameters.Add(new ParameterDefinition("value", ParameterAttributes.None, variable));
206 writerFunc.Body.InitLocals = true;
207
208 RegisterWriteFunc(variable, writerFunc);
209 return writerFunc;

212 MethodDefinition GenerateClassOrStructWriterFunction(TypeReference variable, ref bool WeavingFailed)
213 {
214 MethodDefinition writerFunc = GenerateWriterFunc(variable);
215

217

219 WriteNullCheck(worker, ref WeavingFailed);

228 void WriteNullCheck(ILProcessor worker, ref bool WeavingFailed)
229 {
230 // if (value == null)
231 // {
232 // writer.WriteBoolean(false);
233 // return;
234 // }
235 //
236
237 Instruction labelNotNull = worker.Create(OpCodes.Nop);
238 worker.Emit(OpCodes.Ldarg_1);
239 worker.Emit(OpCodes.Brtrue, labelNotNull);
240 worker.Emit(OpCodes.Ldarg_0);
241 worker.Emit(OpCodes.Ldc_I4_0);
242 worker.Emit(OpCodes.Call, GetWriteFunc(weaverTypes.Import<bool>(), ref WeavingFailed));
243 worker.Emit(OpCodes.Ret);
244 worker.Append(labelNotNull);
245
246 // write.WriteBoolean(true);
247 worker.Emit(OpCodes.Ldarg_0);
248 worker.Emit(OpCodes.Ldc_I4_1);
249 worker.Emit(OpCodes.Call, GetWriteFunc(weaverTypes.Import<bool>(), ref WeavingFailed));

252 // Find all fields in type and write them
253 bool WriteAllFields(TypeReference variable, ILProcessor worker, ref bool WeavingFailed)
254 {
255 foreach (FieldDefinition field in variable.FindAllPublicFields())
256 {
257 MethodReference writeFunc = GetWriteFunc(field.FieldType, ref WeavingFailed);
258 // need this null check till later PR when GetWriteFunc throws exception instead
259 if (writeFunc == null) { return false; }
260
261 FieldReference fieldRef = assembly.MainModule.ImportReference(field);
262
263 worker.Emit(OpCodes.Ldarg_0);
264 worker.Emit(OpCodes.Ldarg_1);
265 worker.Emit(OpCodes.Ldfld, fieldRef);
266 worker.Emit(OpCodes.Call, writeFunc);
267 }
268
269 return true;
270 }

272 MethodDefinition GenerateCollectionWriter(TypeReference variable, TypeReference elementType, string writerFunction, ref bool WeavingFailed)
273 {
274
275 MethodDefinition writerFunc = GenerateWriterFunc(variable);
276
277 MethodReference elementWriteFunc = GetWriteFunc(elementType, ref WeavingFailed);
278 MethodReference intWriterFunc = GetWriteFunc(weaverTypes.Import<int>(), ref WeavingFailed);
279
280 // need this null check till later PR when GetWriteFunc throws exception instead
281 if (elementWriteFunc == null)
282 {
283 Log.Error($"Cannot generate writer for {variable}. Use a supported type or provide a custom writer", variable);
284 WeavingFailed = true;
285 return writerFunc;

288 ModuleDefinition module = assembly.MainModule;
289 TypeReference readerExtensions = module.ImportReference(typeof(NetworkWriterExtensions));
290 MethodReference collectionWriter = Resolvers.ResolveMethod(readerExtensions, assembly, Log, writerFunction, ref WeavingFailed);
291
292 GenericInstanceMethod methodRef = new GenericInstanceMethod(collectionWriter);
293 methodRef.GenericArguments.Add(elementType);
294
295 // generates
296 // reader.WriteArray<T>(array);
297
298 ILProcessor worker = writerFunc.Body.GetILProcessor();
299 worker.Emit(OpCodes.Ldarg_0); // writer
300 worker.Emit(OpCodes.Ldarg_1); // collection
301
302 worker.Emit(OpCodes.Call, methodRef); // WriteArray
303
304 worker.Emit(OpCodes.Ret);
305
306 return writerFunc;

309 // Save a delegate for each one of the writers into Writer{T}.write
310 internal void InitializeWriters(ILProcessor worker)
311 {
312 ModuleDefinition module = assembly.MainModule;
313
314 TypeReference genericWriterClassRef = module.ImportReference(typeof(Writer<>));
315
316 System.Reflection.FieldInfo fieldInfo = typeof(Writer<>).GetField(nameof(Writer<object>.write));
317 FieldReference fieldRef = module.ImportReference(fieldInfo);
318 TypeReference networkWriterRef = module.ImportReference(typeof(NetworkWriter));
319 TypeReference actionRef = module.ImportReference(typeof(Action<,>));
320 MethodReference actionConstructorRef = module.ImportReference(typeof(Action<,>).GetConstructors()[0]);
321
322 foreach (KeyValuePair<TypeReference, MethodReference> kvp in writeFuncs)
323 {
324 TypeReference targetType = kvp.Key;
325 MethodReference writeFunc = kvp.Value;
326
327 // create a Action<NetworkWriter, T> delegate
328 worker.Emit(OpCodes.Ldnull);
329 worker.Emit(OpCodes.Ldftn, writeFunc);
330 GenericInstanceType actionGenericInstance = actionRef.MakeGenericInstanceType(networkWriterRef, targetType);
331 MethodReference actionRefInstance = actionConstructorRef.MakeHostInstanceGeneric(assembly.MainModule, actionGenericInstance);
332 worker.Emit(OpCodes.Newobj, actionRefInstance);
333
334 // save it in Writer<T>.write
335 GenericInstanceType genericInstance = genericWriterClassRef.MakeGenericInstanceType(targetType);
336 FieldReference specializedField = fieldRef.SpecializeField(assembly.MainModule, genericInstance);
337 worker.Emit(OpCodes.Stsfld, specializedField);
338 }
339 }
