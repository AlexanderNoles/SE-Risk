44 public void Add(Vector2Int position, T value)
45 {
46 // initialize set in grid if it's not in there yet
47 if (!grid.TryGetValue(position, out HashSet<T> hashSet))
48 {
49 // each grid entry may hold hundreds of entities.
50 // let's create the HashSet with a large initial capacity
51 // in order to avoid resizing & allocations.
52#if !UNITY_2021_3_OR_NEWER

54 hashSet = new HashSet<T>();
55#else
56 hashSet = new HashSet<T>(128);
57#endif

59 }

65 // helper function to get set at position without worrying
66 // -> result is passed as parameter to avoid allocations
67 // -> result is not cleared before. this allows us to pass the HashSet from
68 // GetWithNeighbours and avoid .UnionWith which is very expensive.
69 void GetAt(Vector2Int position, HashSet<T> result)
70 {
71 // return the set at position
72 if (grid.TryGetValue(position, out HashSet<T> hashSet))
73 {
74 foreach (T entry in hashSet)
75 result.Add(entry);
76 }
77 }

79 // helper function to get at position and it's 8 neighbors without worrying

81 public void GetWithNeighbours(Vector2Int position, HashSet<T> result)
82 {
83 // clear result first
84 result.Clear();
85
86 // add neighbours
87 foreach (Vector2Int offset in neighbourOffsets)

89 }

92 // IMPORTANT: we already allocated HashSet<T>s and don't want to do

94 // => so simply remove each position's entries, but keep
95 // every position in there
96 // => see 'grid' comments above!
97 // => named ClearNonAlloc to make it more obvious!
98 public void ClearNonAlloc()
99 {
100 foreach (HashSet<T> hashSet in grid.Values)
