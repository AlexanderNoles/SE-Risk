151 [Server]
152 void SyncToClients()
153 {
154 // only update if they have changed more than Sensitivity
155
156 Vector2 currentVelocity = syncVelocity ? target.velocity : default;
157 float currentAngularVelocity = syncAngularVelocity ? target.angularVelocity : default;
158
159 bool velocityChanged = syncVelocity && ((previousValue.velocity - currentVelocity).sqrMagnitude > velocitySensitivity * velocitySensitivity);
160 bool angularVelocityChanged = syncAngularVelocity && ((previousValue.angularVelocity - currentAngularVelocity) > angularVelocitySensitivity);
161

164 velocity = currentVelocity;

197 [Client]
198 void SendVelocity()
199 {
200 float now = Time.time;
201 if (now < previousValue.nextSyncTime)
202 return;
203
204 Vector2 currentVelocity = syncVelocity ? target.velocity : default;
205 float currentAngularVelocity = syncAngularVelocity ? target.angularVelocity : default;
206
207 bool velocityChanged = syncVelocity && ((previousValue.velocity - currentVelocity).sqrMagnitude > velocitySensitivity * velocitySensitivity);
208 bool angularVelocityChanged = syncAngularVelocity && previousValue.angularVelocity != currentAngularVelocity;//((previousValue.angularVelocity - currentAngularVelocity).sqrMagnitude > angularVelocitySensitivity * angularVelocitySensitivity);
209
210 // if angularVelocity has changed it is likely that velocity has also changed so just sync both values
211 // however if only velocity has changed just send velocity
212 if (angularVelocityChanged)
213 {
214 CmdSendVelocityAndAngular(currentVelocity, currentAngularVelocity);
215 previousValue.velocity = currentVelocity;
216 previousValue.angularVelocity = currentAngularVelocity;
217 }
218 else if (velocityChanged)
219 {
220 CmdSendVelocity(currentVelocity);
221 previousValue.velocity = currentVelocity;
222 }
223
224 // only update syncTime if either has changed
225 if (angularVelocityChanged || velocityChanged)
226 previousValue.nextSyncTime = now + syncInterval;
227 }
