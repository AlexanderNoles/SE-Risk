2// force all thread functions to be STATIC.
3// => Common.Send/ReceiveLoop is EXTREMELY DANGEROUS because it's too easy to
4// accidentally share Common state between threads.
5// => header buffer, payload etc. were accidentally shared once after changing
6// the thread functions from static to non static
7// => C# does not automatically detect data races. best we can do is move all of
8// our thread code into static functions and pass all state into them

10// let's even keep them in a STATIC CLASS so it's 100% obvious that this should

20 // send message (via stream) with the <size,content> message structure
21 // this function is blocking sometimes!
22 // (e.g. if someone has high latency or wire was cut off)
23 // -> payload is of multiple <<size, content, size, content, ...> parts
24 public static bool SendMessagesBlocking(NetworkStream stream, byte[] payload, int packetSize)
25 {
26 // stream.Write throws exceptions if client sends with high
27 // frequency and the server stops
28 try
29 {
30 // write the whole thing
31 stream.Write(payload, 0, packetSize);
32 return true;
33 }
34 catch (Exception exception)
35 {
36 // log as regular message because servers do shut down sometimes
37 Log.Info("Send: stream.Write exception: " + exception);
38 return false;
39 }

41 // read message (via stream) blocking.
42 // writes into byte[] and returns bytes written to avoid allocations.
43 public static bool ReadMessageBlocking(NetworkStream stream, int MaxMessageSize, byte[] headerBuffer, byte[] payloadBuffer, out int size)
44 {
45 size = 0;
46
47 // buffer needs to be of Header + MaxMessageSize
48 if (payloadBuffer.Length != 4 + MaxMessageSize)
49 {
50 Log.Error($"ReadMessageBlocking: payloadBuffer needs to be of size 4 + MaxMessageSize = {4 + MaxMessageSize} instead of {payloadBuffer.Length}");
51 return false;
52 }
53
54 // read exactly 4 bytes for header (blocking)
55 if (!stream.ReadExactly(headerBuffer, 4))
56 return false;
57
58 // convert to int
59 size = Utils.BytesToIntBigEndian(headerBuffer);
60
61 // protect against allocation attacks. an attacker might send
62 // multiple fake '2GB header' packets in a row, causing the server
63 // to allocate multiple 2GB byte arrays and run out of memory.
64 //
65 // also protect against size <= 0 which would cause issues
66 if (size > 0 && size <= MaxMessageSize)
67 {
68 // read exactly 'size' bytes for content (blocking)
69 return stream.ReadExactly(payloadBuffer, size);
70 }
71 Log.Warning("ReadMessageBlocking: possible header attack with a header of: " + size + " bytes.");
72 return false;
73 }

75 // thread receive function is the same for client and server's clients
76 public static void ReceiveLoop(int connectionId, TcpClient client, int MaxMessageSize, MagnificentReceivePipe receivePipe, int QueueLimit)
77 {
78 // get NetworkStream from client
79 NetworkStream stream = client.GetStream();
80
81 // every receive loop needs it's own receive buffer of
82 // HeaderSize + MaxMessageSize
83 // to avoid runtime allocations.
84 //
85 // IMPORTANT: DO NOT make this a member, otherwise every connection
86 // on the server would use the same buffer simulatenously
87 byte[] receiveBuffer = new byte[4 + MaxMessageSize];
88
89 // avoid header[4] allocations
90 //
91 // IMPORTANT: DO NOT make this a member, otherwise every connection
92 // on the server would use the same buffer simulatenously
93 byte[] headerBuffer = new byte[4];
94
95 // absolutely must wrap with try/catch, otherwise thread exceptions
96 // are silent
97 try
98 {
99 // add connected event to pipe
100 receivePipe.Enqueue(connectionId, EventType.Connected, default);
101
102 // let's talk about reading data.
103 // -> normally we would read as much as possible and then
104 // extract as many <size,content>,<size,content> messages
105 // as we received this time. this is really complicated
106 // and expensive to do though
107 // -> instead we use a trick:
108 // Read(2) -> size
109 // Read(size) -> content
110 // repeat
111 // Read is blocking, but it doesn't matter since the
112 // best thing to do until the full message arrives,
113 // is to wait.
114 // => this is the most elegant AND fast solution.
115 // + no resizing
116 // + no extra allocations, just one for the content
117 // + no crazy extraction logic
118 while (true)
119 {
120 // read the next message (blocking) or stop if stream closed
121 if (!ReadMessageBlocking(stream, MaxMessageSize, headerBuffer, receiveBuffer, out int size))
122 // break instead of return so stream close still happens!
123 break;
124
125 // create arraysegment for the read message
126 ArraySegment<byte> message = new ArraySegment<byte>(receiveBuffer, 0, size);
127
128 // send to main thread via pipe
129 // -> it'll copy the message internally so we can reuse the
130 // receive buffer for next read!
131 receivePipe.Enqueue(connectionId, EventType.Data, message);
132
133 // disconnect if receive pipe gets too big for this connectionId.
134 // -> avoids ever growing queue memory if network is slower
135 // than input
136 // -> disconnecting is great for load balancing. better to
137 // disconnect one connection than risking every
138 // connection / the whole server
139 if (receivePipe.Count(connectionId) >= QueueLimit)
140 {
141 // log the reason
142 Log.Warning($"receivePipe reached limit of {QueueLimit} for connectionId {connectionId}. This can happen if network messages come in way faster than we manage to process them. Disconnecting this connection for load balancing.");
143
144 // IMPORTANT: do NOT clear the whole queue. we use one
145 // queue for all connections.
146 //receivePipe.Clear();
147
148 // just break. the finally{} will close everything.
149 break;
150 }
151 }
152 }
153 catch (Exception exception)
154 {
155 // something went wrong. the thread was interrupted or the
156 // connection closed or we closed our own connection or ...
157 // -> either way we should stop gracefully
158 Log.Info("ReceiveLoop: finished receive function for connectionId=" + connectionId + " reason: " + exception);
159 }
160 finally
161 {
162 // clean up no matter what
163 stream.Close();
164 client.Close();
165
166 // add 'Disconnected' message after disconnecting properly.
167 // -> always AFTER closing the streams to avoid a race condition
168 // where Disconnected -> Reconnect wouldn't work because
169 // Connected is still true for a short moment before the stream
170 // would be closed.
171 receivePipe.Enqueue(connectionId, EventType.Disconnected, default);
172 }
173 }
174 // thread send function
175 // note: we really do need one per connection, so that if one connection
176 // blocks, the rest will still continue to get sends
177 public static void SendLoop(int connectionId, TcpClient client, MagnificentSendPipe sendPipe, ManualResetEvent sendPending)
178 {
179 // get NetworkStream from client
180 NetworkStream stream = client.GetStream();
181
182 // avoid payload[packetSize] allocations. size increases dynamically as
183 // needed for batching.
184 //
185 // IMPORTANT: DO NOT make this a member, otherwise every connection
186 // on the server would use the same buffer simulatenously
187 byte[] payload = null;
188
189 try
190 {
191 while (client.Connected) // try this. client will get closed eventually.
192 {
193 // reset ManualResetEvent before we do anything else. this
194 // way there is no race condition. if Send() is called again
195 // while in here then it will be properly detected next time
196 // -> otherwise Send might be called right after dequeue but
197 // before .Reset, which would completely ignore it until
198 // the next Send call.
199 sendPending.Reset(); // WaitOne() blocks until .Set() again
200
201 // dequeue & serialize all
202 // a locked{} TryDequeueAll is twice as fast as
203 // ConcurrentQueue, see SafeQueue.cs!
204 if (sendPipe.DequeueAndSerializeAll(ref payload, out int packetSize))
205 {
206 // send messages (blocking) or stop if stream is closed
207 if (!SendMessagesBlocking(stream, payload, packetSize))
208 // break instead of return so stream close still happens!
209 break;
210 }
211
212 // don't choke up the CPU: wait until queue not empty anymore
213 sendPending.WaitOne();
214 }
215 }
216 catch (ThreadAbortException)
217 {
218 // happens on stop. don't log anything.
219 }
220 catch (ThreadInterruptedException)
221 {
222 // happens if receive thread interrupts send thread.
223 }
224 catch (Exception exception)
225 {
226 // something went wrong. the thread was interrupted or the
227 // connection closed or we closed our own connection or ...
228 // -> either way we should stop gracefully
229 Log.Info("SendLoop Exception: connectionId=" + connectionId + " reason: " + exception);
230 }
231 finally
232 {
233 // clean up no matter what
234 // we might get SocketExceptions when sending if the 'host has
235 // failed to respond' - in which case we should close the connection
236 // which causes the ReceiveLoop to end and fire the Disconnected
237 // message. otherwise the connection would stay alive forever even
238 // though we can't send anymore.
239 stream.Close();
240 client.Close();
241 }
242 }
243 }
