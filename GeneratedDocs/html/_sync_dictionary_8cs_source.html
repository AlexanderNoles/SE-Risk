156 public override void OnDeserializeDelta(NetworkReader reader)
157 {
158 int changesCount = (int)reader.ReadUInt();
159
160 for (int i = 0; i < changesCount; i++)
161 {
162 Operation operation = (Operation)reader.ReadByte();
163
164 // apply the operation only if it is a new change
165 // that we have not applied yet
166 bool apply = changesAhead == 0;
167 TKey key = default;
168 TValue item = default;
169
170 switch (operation)
171 {
172 case Operation.OP_ADD:
173 case Operation.OP_SET:
174 key = reader.Read<TKey>();
175 item = reader.Read<TValue>();
176 if (apply)
177 {
178 // add dirty + changes.
179 // ClientToServer needs to set dirty in server OnDeserialize.
180 // no access check: server OnDeserialize can always
181 // write, even for ClientToServer (for broadcasting).
182 if (ContainsKey(key))
183 {
184 objects[key] = item; // assign after ContainsKey check
185 AddOperation(Operation.OP_SET, key, item, false);
186 }
187 else
188 {
189 objects[key] = item; // assign after ContainsKey check
190 AddOperation(Operation.OP_ADD, key, item, false);
191 }
192 }
193 break;
194
195 case Operation.OP_CLEAR:
196 if (apply)
197 {
198 objects.Clear();
199 // add dirty + changes.
200 // ClientToServer needs to set dirty in server OnDeserialize.
201 // no access check: server OnDeserialize can always
202 // write, even for ClientToServer (for broadcasting).
203 AddOperation(Operation.OP_CLEAR, default, default, false);
204 }
205 break;
206
207 case Operation.OP_REMOVE:
208 key = reader.Read<TKey>();
209 if (apply)
210 {
211 if (objects.TryGetValue(key, out item))
212 {
213 // add dirty + changes.
214 // ClientToServer needs to set dirty in server OnDeserialize.
215 // no access check: server OnDeserialize can always
216 // write, even for ClientToServer (for broadcasting).
217 objects.Remove(key);
218 AddOperation(Operation.OP_REMOVE, key, item, false);
219 }
220 }
221 break;
222 }
223
224 if (!apply)
225 {
226 // we just skipped this change
227 changesAhead--;
228 }
229 }
