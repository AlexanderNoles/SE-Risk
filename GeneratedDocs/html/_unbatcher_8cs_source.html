2// -> less complexity
3// -> easy to test

5// includes timestamp for tick batching.
6// -> allows NetworkTransform etc. to use timestamp without including it in

8using System;
9using System.Collections.Generic;

11namespace Mirror
12{
13 public class Unbatcher
14 {
15 // supporting adding multiple batches before GetNextMessage is called.
16 // just in case.
17 readonly Queue<NetworkWriterPooled> batches = new Queue<NetworkWriterPooled>();
18
19 public int BatchesCount => batches.Count;
20
21 // NetworkReader is only created once,
22 // then pointed to the first batch.
23 readonly NetworkReader reader = new NetworkReader(new byte[0]);
24
25 // timestamp that was written into the batch remotely.
26 // for the batch that our reader is currently pointed at.
27 double readerRemoteTimeStamp;
28
29 // helper function to start reading a batch.
30 void StartReadingBatch(NetworkWriterPooled batch)
31 {
32 // point reader to it
33 reader.SetBuffer(batch.ToArraySegment());
34
35 // read remote timestamp (double)
36 // -> AddBatch quarantees that we have at least 8 bytes to read
37 readerRemoteTimeStamp = reader.ReadDouble();
38 }

40 // add a new batch.
41 // returns true if valid.
42 // returns false if not, in which case the connection should be disconnected.
43 public bool AddBatch(ArraySegment<byte> batch)
44 {
45 // IMPORTANT: ArraySegment is only valid until returning. we copy it!
46 //
47 // NOTE: it's not possible to create empty ArraySegments, so we
48 // don't need to check against that.
49
50 // make sure we have at least 8 bytes to read for tick timestamp
51 if (batch.Count < Batcher.TimestampSize)
52 return false;
53
54 // put into a (pooled) writer
55 // -> WriteBytes instead of WriteSegment because the latter
56 // would add a size header. we want to write directly.
57 // -> will be returned to pool when sending!
58 NetworkWriterPooled writer = NetworkWriterPool.Get();
59 writer.WriteBytes(batch.Array, batch.Offset, batch.Count);
60
61 // first batch? then point reader there
62 if (batches.Count == 0)
63 StartReadingBatch(writer);
64
65 // add batch
66 batches.Enqueue(writer);
67 //Debug.Log($"Adding Batch {BitConverter.ToString(batch.Array, batch.Offset, batch.Count)} => batches={batches.Count} reader={reader}");
68 return true;
69 }
70
71 // get next message, unpacked from batch (if any)
72 // message ArraySegment is only valid until the next call.
73 // timestamp is the REMOTE time when the batch was created remotely.
74 public bool GetNextMessage(out ArraySegment<byte> message, out double remoteTimeStamp)
75 {
76 message = default;
77 remoteTimeStamp = 0;
78
79 // do nothing if we don't have any batches.
80 // otherwise the below queue.Dequeue() would throw an
81 // InvalidOperationException if operating on empty queue.
82 if (batches.Count == 0)
83 return false;
84
85 // was our reader pointed to anything yet?
86 if (reader.Capacity == 0)
87 return false;
88
89 // no more data to read?
90 if (reader.Remaining == 0)
91 {
92 // retire the batch
93 NetworkWriterPooled writer = batches.Dequeue();
94 NetworkWriterPool.Return(writer);
95
96 // do we have another batch?
97 if (batches.Count > 0)
98 {
99 // point reader to the next batch.
100 // we'll return the reader below.
101 NetworkWriterPooled next = batches.Peek();
102 StartReadingBatch(next);
103 }
104 // otherwise there's nothing more to read
105 else return false;
106 }
107
108 // use the current batch's remote timestamp
109 // AFTER potentially moving to the next batch ABOVE!
110 remoteTimeStamp = readerRemoteTimeStamp;
111
112 // enough data to read the size prefix?
113 if (reader.Remaining == 0)
114 return false;
115
116 // read the size prefix as varint
117 // see Batcher.AddMessage comments for explanation.
118 int size = (int)Compression.DecompressVarUInt(reader);
119
120 // validate size prefix, in case attackers send malicious data
121 if (reader.Remaining < size)
122 return false;
123
124 // return the message of size
125 message = reader.ReadBytesSegment(size);
126 return true;
127 }
128 }
