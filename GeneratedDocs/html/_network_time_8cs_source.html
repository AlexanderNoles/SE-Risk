54 // via global NetworkClient snapshot interpolated timeline (if client).

57 // I measured the accuracy of float and I got this:
58 // for the same day, accuracy is better than 1 ms
59 // after 1 day, accuracy goes down to 7 ms
60 // after 10 days, accuracy is 61 ms

73 // prediction //////////////////////////////////////////////////////////
74 // NetworkTime.time is server time, behind by bufferTime.
75 // for prediction, we want server time, ahead by latency.
76 // so that client inputs at predictedTime=2 arrive on server at time=2.
77 // the more accurate this is, the more closesly will corrections be
78 // be applied and the less jitter we will see.
79 //
80 // we'll use a two step process to calculate predicted time:
81 // 1. move snapshot interpolated time to server time, without being behind by bufferTime
82 // 2. constantly send this time to server (included in ping message)
83 // server replies with how far off it was.
84 // client averages that offset and applies it to predictedTime to get ever closer.
85 //
86 // this is also very easy to test & verify:
87 // - add LatencySimulation with 50ms latency
88 // - log predictionError on server in OnServerPing, see if it gets closer to 0
89 //
90 // credits: FakeByte, imer, NinjaKickja, mischa
91 // const because it's used immediately in _predictionError constructor.
92
93 static int PredictionErrorWindowSize = 20; // average over 20 * 100ms = 2s
94 static ExponentialMovingAverage _predictionErrorUnadjusted = new ExponentialMovingAverage(PredictionErrorWindowSize);
95 public static double predictionErrorUnadjusted => _predictionErrorUnadjusted.Value;
96 public static double predictionErrorAdjusted { get; private set; } // for debugging

99 // on client, this is based on localTime (aka Time.time) instead of the snapshot interpolated timeline.
100 // this gives much better and immediately accurate results.
101 // -> snapshot interpolation timeline tries to emulate a server timeline without hard offset corrections.
102 // -> predictedTime does have hard offset corrections, so might as well use Time.time directly for this.
103 //
104 // note that predictedTime over unreliable is enough!
105 // even with reliable components, it gives better results than if we were
106 // to implemented predictedTime over reliable channel.
107 public static double predictedTime
108 {
109 [MethodImpl(MethodImplOptions.AggressiveInlining)]

113 }

161 // client rtt calculation //////////////////////////////////////////////
162 // executed at the server when we receive a ping message
163 // reply with a pong containing the time from the client
164 // and time from the server
165 internal static void OnServerPing(NetworkConnectionToClient conn, NetworkPingMessage message)
166 {
167 // calculate the prediction offset that the client needs to apply to unadjusted time to reach server time.
168 // this will be sent back to client for corrections.
169 double unadjustedError = localTime - message.localTime;
170
171 // to see how well the client's final prediction worked, compare with adjusted time.
172 // this is purely for debugging.
173 // >0 means: server is ... seconds ahead of client's prediction (good if small)
174 // <0 means: server is ... seconds behind client's prediction.
175 // in other words, client is predicting too far ahead (not good)
176 double adjustedError = localTime - message.predictedTimeAdjusted;
177 // Debug.Log($"[Server] unadjustedError:{(unadjustedError*1000):F1}ms adjustedError:{(adjustedError*1000):F1}ms");
178
179 // Debug.Log($"OnServerPing conn:{conn}");
180 NetworkPongMessage pongMessage = new NetworkPongMessage
181 (
182 message.localTime,
183 unadjustedError,
184 adjustedError
185 );
186 conn.Send(pongMessage, Channels.Unreliable);
187 }

189 // Executed at the client when we receive a Pong message
190 // find out how long it took since we sent the Ping
191 // and update time offset & prediction offset.
192 internal static void OnClientPong(NetworkPongMessage message)
193 {
194 // prevent attackers from sending timestamps which are in the future
195 if (message.localTime > localTime) return;
196
197 // how long did this message take to come back
198 double newRtt = localTime - message.localTime;
199 _rtt.Add(newRtt);
200
201 // feed unadjusted prediction error into our exponential moving average
202 // store adjusted prediction error for debug / GUI purposes
203 _predictionErrorUnadjusted.Add(message.predictionErrorUnadjusted);
204 predictionErrorAdjusted = message.predictionErrorAdjusted;
205 // Debug.Log($"[Client] predictionError avg={(_predictionErrorUnadjusted.Value*1000):F1} ms");
206 }
