121 List<int> attackingRolls = new List<int>();
122 List<int> defendingRolls = new List<int>();
123 for (int i = 0; i < attackingDice; i++)

131 attackingRolls.Sort();
132 attackingRolls.Reverse();
133 defendingRolls.Sort();
134 defendingRolls.Reverse();
135
136 for (int i = 0; i < defendingRolls.Count && i < attackingRolls.Count; i++)
137 {
138 string newOutputLine = "(" + attackingRolls[i] + " vs " + defendingRolls[i] + ")";
139 if (attackingRolls[i] > defendingRolls[i])
140 {
141 defender.SetCurrentTroops(defender.GetCurrentTroops() - 1);
142 }
143 else
144 {
145 attacker.SetCurrentTroops(attacker.GetCurrentTroops() - 1);
146 }
147 UIManagement.AddLineToRollOutput(newOutputLine);
148 }
149
150 AttackResult attackResult = defender.GetCurrentTroops() <= 0 ? AttackResult.Won : AttackResult.Lost;
151
152 if (defender.GetCurrentTroops() <= 0)
153 {
154 Player oldOwner = defender.GetOwner();
155 defender.SetOwner(attacker.GetOwner());
156 attacker.GetOwner().AddTerritory(defender);
157 oldOwner.RemoveTerritory(defender);
158 defender.SetCurrentTroops(attacker.GetCurrentTroops() - 1 >= 3 ? 3 : attacker.GetCurrentTroops() - 1);
159 attacker.SetCurrentTroops(attacker.GetCurrentTroops() - defender.GetCurrentTroops());
160 UIManagement.AddLineToRollOutput("Territory Taken!");
161 if (oldOwner.IsDead())
162 {
163 attacker.GetOwner().Killed(oldOwner);
164 }
165 }
166 else
167 {
168 UIManagement.AddLineToRollOutput("Territory Defended!");
169 }
170
171 //Send a message to the match manager
172 //So it can check if the game is now over
173 MatchManager.WinCheck(attacker.GetOwner());
174
175 //Notify the attacker
176 //owner will be null if game was just won
177 //This is why the win check is placed above this function call
