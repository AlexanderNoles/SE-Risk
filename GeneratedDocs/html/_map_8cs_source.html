205 public static void Attack(Territory attacker, Territory defender, int attackingDice)
206 {
207 //Get max possible defending dice based on territory
208 int defendingDice = Player.GetMaxDefendingDice(defender);
209
210
211 List<int> attackingRolls = new List<int>();
212 List<int> defendingRolls = new List<int>();
213 for (int i = 0; i < attackingDice; i++)
214 {
215 attackingRolls.Add(Random.Range(1, 7));
216 }
217 for (int i = 0; i < defendingDice; i++)
218 {
219 defendingRolls.Add(Random.Range(1, 7));
220 }
221 attackingRolls.Sort();
222 attackingRolls.Reverse();
223 defendingRolls.Sort();
224 defendingRolls.Reverse();
225
226 for (int i = 0; i < defendingRolls.Count && i < attackingRolls.Count; i++)
227 {
228 string newOutputLine = "(" + attackingRolls[i] + " vs " + defendingRolls[i] + ")";
229 if (attackingRolls[i] > defendingRolls[i])
230 {
231 defender.SetCurrentTroops(defender.GetCurrentTroops() - 1);
232 }
233 else
234 {
235 attacker.SetCurrentTroops(attacker.GetCurrentTroops() - 1);
236 }
237 UIManagement.AddLineToRollOutput(newOutputLine);
238 }
239
240 AttackResult attackResult = defender.GetCurrentTroops() <= 0 ? AttackResult.Won : AttackResult.Lost;
241
242 Player attackerPlayer = MatchManager.GetPlayerFromIndex(attacker.GetOwner());
243
244 if (defender.GetCurrentTroops() <= 0)
245 {
246 int defenderIndex = defender.GetOwner();
247 defender.SetOwner(attacker.GetOwner());
248 attackerPlayer.AddTerritory(defender);
249 UIManagement.AddLineToRollOutput("Territory Taken!");
250
251 List<int> alivePlayers = GetAlivePlayers();
252 if (!alivePlayers.Contains(defenderIndex))
253 {
254 attackerPlayer.Killed(PlayerInfoHandler.CardCountForIndex(defender.GetOwner()));
255 }
256 }
257 else
258 {
259 UIManagement.AddLineToRollOutput("Territory Defended!");
260 }
261
262 //Notify the attacker
263 //owner will be null if game was just won
264 //This is why the win check is placed above this function call
265 if (attacker.GetOwner() != -1)
266 {
267 attackerPlayer.OnAttackEnd(attackResult, attacker, defender,attackingDice);
268 }
269
270 //Refresh UI
271 UIManagement.RefreshRollOutput();
272 }

428 public void SetupMap()
429 {
430 //Load territory name overriding
431 Dictionary<int, string> nameDict = MapEditor.LoadTerritoryNamesList();
432 //Load territory extra neighbours
433 Dictionary<int, List<int>> extraNeighboursDict = MapEditor.LoadExtraNeighboursList();
434
435 if (PlayOptionsManagement.IsConquestMode())

437 capitals.Clear();

439 territories = new List<Territory>();
440 continents = new Dictionary<Territory.Continent, List<Territory>>();
441 instance = this;
442 foreach (Transform child in transform)
443 {
444 if (child.TryGetComponent<Territory>(out Territory territory))
445 {
446 territories.Add(territory);
447 territory.ResetTerritory(territories.Count-1);
448
449 if (nameDict.TryGetValue(territories.Count-1, out string newTerrOverName))
450 {
451 territory.SetName(newTerrOverName);
452 }
453
454 if (!continents.ContainsKey(territory.GetContinent()))
455 {
456 continents[territory.GetContinent()] = new List<Territory>();
457 }
458 continents[territory.GetContinent()].Add(territory);
459 }
460 }
461
462 //We do this after so we can get the actual territory objects
463 for (int i = 0; i < territories.Count; i++)
464 {
465 if (extraNeighboursDict.ContainsKey(i))
466 {
467 territories[i].SetExtraNeighbours(extraNeighboursDict[i]);
