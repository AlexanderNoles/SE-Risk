65 MethodReference GenerateReader(TypeReference variableReference, ref bool WeavingFailed)
66 {
67 // Arrays are special, if we resolve them, we get the element type,
68 // so the following ifs might choke on it for scriptable objects
69 // or other objects that require a custom serializer
70 // thus check if it is an array and skip all the checks.
71 if (variableReference.IsArray)
72 {
73 if (variableReference.IsMultidimensionalArray())
74 {
75 Log.Error($"{variableReference.Name} is an unsupported type. Multidimensional arrays are not supported", variableReference);
76 WeavingFailed = true;
77 return null;
78 }
79
80 return GenerateReadCollection(variableReference, variableReference.GetElementType(), nameof(NetworkReaderExtensions.ReadArray), ref WeavingFailed);
81 }
82
83 TypeDefinition variableDefinition = variableReference.Resolve();
84
85 // check if the type is completely invalid
86 if (variableDefinition == null)
87 {
88 Log.Error($"{variableReference.Name} is not a supported type", variableReference);
89 WeavingFailed = true;
90 return null;
91 }
92 else if (variableReference.IsByReference)
93 {
94 // error??
95 Log.Error($"Cannot pass type {variableReference.Name} by reference", variableReference);
96 WeavingFailed = true;
97 return null;
98 }
99
100 // use existing func for known types
101 if (variableDefinition.IsEnum)
102 {
103 return GenerateEnumReadFunc(variableReference, ref WeavingFailed);
104 }
105 else if (variableDefinition.Is(typeof(ArraySegment<>)))
106 {
107 return GenerateArraySegmentReadFunc(variableReference, ref WeavingFailed);
108 }
109 else if (variableDefinition.Is(typeof(List<>)))
110 {
111 GenericInstanceType genericInstance = (GenericInstanceType)variableReference;
112 TypeReference elementType = genericInstance.GenericArguments[0];
113
114 return GenerateReadCollection(variableReference, elementType, nameof(NetworkReaderExtensions.ReadList), ref WeavingFailed);
115 }
116 // handle both NetworkBehaviour and inheritors.
117 // fixes: https://github.com/MirrorNetworking/Mirror/issues/2939
118 else if (variableReference.IsDerivedFrom<NetworkBehaviour>() || variableReference.Is<NetworkBehaviour>())
119 {
120 return GetNetworkBehaviourReader(variableReference);
121 }
122
123 // check if reader generation is applicable on this type
124 if (variableDefinition.IsDerivedFrom<UnityEngine.Component>())
125 {
126 Log.Error($"Cannot generate reader for component type {variableReference.Name}. Use a supported type or provide a custom reader", variableReference);
127 WeavingFailed = true;
128 return null;
129 }
130 if (variableReference.Is<UnityEngine.Object>())
131 {
132 Log.Error($"Cannot generate reader for {variableReference.Name}. Use a supported type or provide a custom reader", variableReference);
133 WeavingFailed = true;
134 return null;
135 }
136 if (variableReference.Is<UnityEngine.ScriptableObject>())
137 {
138 Log.Error($"Cannot generate reader for {variableReference.Name}. Use a supported type or provide a custom reader", variableReference);
139 WeavingFailed = true;
140 return null;
141 }
142 if (variableDefinition.HasGenericParameters)
143 {
144 Log.Error($"Cannot generate reader for generic variable {variableReference.Name}. Use a supported type or provide a custom reader", variableReference);
145 WeavingFailed = true;
146 return null;
147 }
148 if (variableDefinition.IsInterface)
149 {
150 Log.Error($"Cannot generate reader for interface {variableReference.Name}. Use a supported type or provide a custom reader", variableReference);
151 WeavingFailed = true;
152 return null;
153 }
154 if (variableDefinition.IsAbstract)
155 {
156 Log.Error($"Cannot generate reader for abstract class {variableReference.Name}. Use a supported type or provide a custom reader", variableReference);
157 WeavingFailed = true;
158 return null;
159 }
160
161 return GenerateClassOrStructReadFunction(variableReference, ref WeavingFailed);
162 }

194 MethodDefinition GenerateArraySegmentReadFunc(TypeReference variable, ref bool WeavingFailed)
195 {
196 GenericInstanceType genericInstance = (GenericInstanceType)variable;
197 TypeReference elementType = genericInstance.GenericArguments[0];
198
199 MethodDefinition readerFunc = GenerateReaderFunction(variable);
200
201 ILProcessor worker = readerFunc.Body.GetILProcessor();
202
203 // $array = reader.Read<[T]>()
204 ArrayType arrayType = elementType.MakeArrayType();
205 worker.Emit(OpCodes.Ldarg_0);
206 worker.Emit(OpCodes.Call, GetReadFunc(arrayType, ref WeavingFailed));
207
208 // return new ArraySegment<T>($array);
209 worker.Emit(OpCodes.Newobj, weaverTypes.ArraySegmentConstructorReference.MakeHostInstanceGeneric(assembly.MainModule, genericInstance));
210 worker.Emit(OpCodes.Ret);

215 {

217
218 // create new reader for this type

222 MethodAttributes.HideBySig,

232 MethodDefinition GenerateReadCollection(TypeReference variable, TypeReference elementType, string readerFunction, ref bool WeavingFailed)
233 {
234 MethodDefinition readerFunc = GenerateReaderFunction(variable);
235 // generate readers for the element
236 GetReadFunc(elementType, ref WeavingFailed);
237
238 ModuleDefinition module = assembly.MainModule;
239 TypeReference readerExtensions = module.ImportReference(typeof(NetworkReaderExtensions));
240 MethodReference listReader = Resolvers.ResolveMethod(readerExtensions, assembly, Log, readerFunction, ref WeavingFailed);
241
242 GenericInstanceMethod methodRef = new GenericInstanceMethod(listReader);
243 methodRef.GenericArguments.Add(elementType);
244
245 // generates
246 // return reader.ReadList<T>();
247
248 ILProcessor worker = readerFunc.Body.GetILProcessor();
249 worker.Emit(OpCodes.Ldarg_0); // reader
250 worker.Emit(OpCodes.Call, methodRef); // Read
251
252 worker.Emit(OpCodes.Ret);
253
254 return readerFunc;

257 MethodDefinition GenerateClassOrStructReadFunction(TypeReference variable, ref bool WeavingFailed)
258 {
259 MethodDefinition readerFunc = GenerateReaderFunction(variable);
260
261 // create local for return value
262 readerFunc.Body.Variables.Add(new VariableDefinition(variable));
263
264 ILProcessor worker = readerFunc.Body.GetILProcessor();
265
266 TypeDefinition td = variable.Resolve();
267
268 if (!td.IsValueType)
269 GenerateNullCheck(worker, ref WeavingFailed);
270
271 CreateNew(variable, worker, td, ref WeavingFailed);
272 ReadAllFields(variable, worker, ref WeavingFailed);
273
274 worker.Emit(OpCodes.Ldloc_0);
275 worker.Emit(OpCodes.Ret);
276 return readerFunc;
277 }

281 // if (!reader.ReadBoolean()) {

295 // Initialize the local variable with a new instance
296 void CreateNew(TypeReference variable, ILProcessor worker, TypeDefinition td, ref bool WeavingFailed)
297 {
298 if (variable.IsValueType)
299 {
300 // structs are created with Initobj
301 worker.Emit(OpCodes.Ldloca, 0);
302 worker.Emit(OpCodes.Initobj, variable);
303 }
304 else if (td.IsDerivedFrom<UnityEngine.ScriptableObject>())
305 {
306 GenericInstanceMethod genericInstanceMethod = new GenericInstanceMethod(weaverTypes.ScriptableObjectCreateInstanceMethod);
307 genericInstanceMethod.GenericArguments.Add(variable);
308 worker.Emit(OpCodes.Call, genericInstanceMethod);
309 worker.Emit(OpCodes.Stloc_0);
310 }
311 else
312 {
313 // classes are created with their constructor
314 MethodDefinition ctor = Resolvers.ResolveDefaultPublicCtor(variable);
315 if (ctor == null)
316 {
317 Log.Error($"{variable.Name} can't be deserialized because it has no default constructor. Don't use {variable.Name} in [SyncVar]s, Rpcs, Cmds, etc.", variable);
318 WeavingFailed = true;
319 return;
320 }
321
322 MethodReference ctorRef = assembly.MainModule.ImportReference(ctor);
323
324 worker.Emit(OpCodes.Newobj, ctorRef);
325 worker.Emit(OpCodes.Stloc_0);
326 }

329 void ReadAllFields(TypeReference variable, ILProcessor worker, ref bool WeavingFailed)
330 {
331 foreach (FieldDefinition field in variable.FindAllPublicFields())
332 {
333 // mismatched ldloca/ldloc for struct/class combinations is invalid IL, which causes crash at runtime
334 OpCode opcode = variable.IsValueType ? OpCodes.Ldloca : OpCodes.Ldloc;
335 worker.Emit(opcode, 0);
336 MethodReference readFunc = GetReadFunc(field.FieldType, ref WeavingFailed);
337 if (readFunc != null)
338 {
339 worker.Emit(OpCodes.Ldarg_0);
340 worker.Emit(OpCodes.Call, readFunc);
341 }
342 else
343 {
344 Log.Error($"{field.Name} has an unsupported type", field);
345 WeavingFailed = true;
346 }
347 FieldReference fieldRef = assembly.MainModule.ImportReference(field);
348
349 worker.Emit(OpCodes.Stfld, fieldRef);
350 }

353 // Save a delegate for each one of the readers into Reader<T>.read
354 internal void InitializeReaders(ILProcessor worker)
355 {
356 ModuleDefinition module = assembly.MainModule;
357
358 TypeReference genericReaderClassRef = module.ImportReference(typeof(Reader<>));
359
360 System.Reflection.FieldInfo fieldInfo = typeof(Reader<>).GetField(nameof(Reader<object>.read));
361 FieldReference fieldRef = module.ImportReference(fieldInfo);
362 TypeReference networkReaderRef = module.ImportReference(typeof(NetworkReader));
363 TypeReference funcRef = module.ImportReference(typeof(Func<,>));
364 MethodReference funcConstructorRef = module.ImportReference(typeof(Func<,>).GetConstructors()[0]);
365
366 foreach (KeyValuePair<TypeReference, MethodReference> kvp in readFuncs)
367 {
368 TypeReference targetType = kvp.Key;
369 MethodReference readFunc = kvp.Value;
370
371 // create a Func<NetworkReader, T> delegate
372 worker.Emit(OpCodes.Ldnull);
373 worker.Emit(OpCodes.Ldftn, readFunc);
374 GenericInstanceType funcGenericInstance = funcRef.MakeGenericInstanceType(networkReaderRef, targetType);
375 MethodReference funcConstructorInstance = funcConstructorRef.MakeHostInstanceGeneric(assembly.MainModule, funcGenericInstance);
376 worker.Emit(OpCodes.Newobj, funcConstructorInstance);
377
378 // save it in Reader<T>.read
379 GenericInstanceType genericInstance = genericReaderClassRef.MakeGenericInstanceType(targetType);
380 FieldReference specializedField = fieldRef.SpecializeField(assembly.MainModule, genericInstance);
381 worker.Emit(OpCodes.Stsfld, specializedField);
382 }
383 }
