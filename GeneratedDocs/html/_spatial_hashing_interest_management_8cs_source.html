12 [Tooltip("The maximum range that objects will be visible at.")]
13 public int visRange = 30;
14
15 // we use a 9 neighbour grid.
16 // so we always see in a distance of 2 grids.
17 // for example, our own grid and then one on top / below / left / right.
18 //
19 // this means that grid resolution needs to be distance / 2.
20 // so for example, for distance = 30 we see 2 cells = 15 * 2 distance.
21 //
22 // on first sight, it seems we need distance / 3 (we see left/us/right).
23 // but that's not the case.
24 // resolution would be 10, and we only see 1 cell far, so 10+10=20.
25 public int resolution => visRange / 2;
26
27 [Tooltip("Rebuild all every 'rebuildInterval' seconds.")]
28 public float rebuildInterval = 1;
29 double lastRebuildTime;
30
31 public enum CheckMethod
32 {
33 XZ_FOR_3D,
34 XY_FOR_2D
35 }
36 [Tooltip("Spatial Hashing supports 3D (XZ) and 2D (XY) games.")]
37 public CheckMethod checkMethod = CheckMethod.XZ_FOR_3D;
38
39 // debugging
40 public bool showSlider;

42 // the grid
43 // begin with a large capacity to avoid resizing & allocations.
44 Grid2D<NetworkConnectionToClient> grid = new Grid2D<NetworkConnectionToClient>(1024);
45
46 // project 3d world position to grid position
47 Vector2Int ProjectToGrid(Vector3 position) =>
48 checkMethod == CheckMethod.XZ_FOR_3D
49 ? Vector2Int.RoundToInt(new Vector2(position.x, position.z) / resolution)
50 : Vector2Int.RoundToInt(new Vector2(position.x, position.y) / resolution);
51
52 public override bool OnCheckObserver(NetworkIdentity identity, NetworkConnectionToClient newObserver)
53 {
54 // calculate projected positions
55 Vector2Int projected = ProjectToGrid(identity.transform.position);
56 Vector2Int observerProjected = ProjectToGrid(newObserver.identity.transform.position);
57
58 // distance needs to be at max one of the 8 neighbors, which is
59 // 1 for the direct neighbors
60 // 1.41 for the diagonal neighbors (= sqrt(2))
61 // => use sqrMagnitude and '2' to avoid computations. same result.
62 return (projected - observerProjected).sqrMagnitude <= 2;
63 }
64
65 public override void OnRebuildObservers(NetworkIdentity identity, HashSet<NetworkConnectionToClient> newObservers)
66 {
67 // add everyone in 9 neighbour grid
68 // -> pass observers to GetWithNeighbours directly to avoid allocations
69 // and expensive .UnionWith computations.
70 Vector2Int current = ProjectToGrid(identity.transform.position);
71 grid.GetWithNeighbours(current, newObservers);
72 }

80 // update everyone's position in the grid
81 // (internal so we can update from tests)
82 [ServerCallback]
83 internal void Update()
84 {
85 // NOTE: unlike Scene/MatchInterestManagement, this rebuilds ALL
86 // entities every INTERVAL. consider the other approach later.
87
88 // IMPORTANT: refresh grid every update!
89 // => newly spawned entities get observers assigned via
90 // OnCheckObservers. this can happen any time and we don't want
91 // them broadcast to old (moved or destroyed) connections.
92 // => players do move all the time. we want them to always be in the
93 // correct grid position.
94 // => note that the actual 'rebuildall' doesn't need to happen all
95 // the time.
96 // NOTE: consider refreshing grid only every 'interval' too. but not
97 // for now. stability & correctness matter.
98
99 // clear old grid results before we update everyone's position.
100 // (this way we get rid of destroyed connections automatically)
101 //
102 // NOTE: keeps allocated HashSets internally.
103 // clearing & populating every frame works without allocations
104 grid.ClearNonAlloc();
105
106 // put every connection into the grid at it's main player's position
107 // NOTE: player sees in a radius around him. NOT around his pet too.
108 foreach (NetworkConnectionToClient connection in NetworkServer.connections.Values)
109 {
110 // authenticated and joined world with a player?
111 if (connection.isAuthenticated && connection.identity != null)
112 {
113 // calculate current grid position
114 Vector2Int position = ProjectToGrid(connection.identity.transform.position);
115
116 // put into grid
117 grid.Add(position, connection);
118 }
119 }
120
121 // rebuild all spawned entities' observers every 'interval'
122 // this will call OnRebuildObservers which then returns the
123 // observers at grid[position] for each entity.
124 if (NetworkTime.localTime >= lastRebuildTime + rebuildInterval)
125 {
126 RebuildAll();
127 lastRebuildTime = NetworkTime.localTime;
128 }
129 }
