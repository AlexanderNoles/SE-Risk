1// Pool to avoid allocations (from libuv2k)
2// API consistent with Microsoft's ObjectPool<T>.
3using System;
4using System.Collections.Generic;
5using System.Runtime.CompilerServices;

7namespace Mirror
8{
9 public class Pool<T>
10 {
11 // Mirror is single threaded, no need for concurrent collections.
12 // stack increases the chance that a reused writer remains in cache.
13 readonly Stack<T> objects = new Stack<T>();
14
15 // some types might need additional parameters in their constructor, so
16 // we use a Func<T> generator
17 readonly Func<T> objectGenerator;
18
19 public Pool(Func<T> objectGenerator, int initialCapacity)
20 {
21 this.objectGenerator = objectGenerator;
22
23 // allocate an initial pool so we have fewer (if any)
24 // allocations in the first few frames (or seconds).
25 for (int i = 0; i < initialCapacity; ++i)
26 objects.Push(objectGenerator());
27 }
28
29 // take an element from the pool, or create a new one if empty
30 [MethodImpl(MethodImplOptions.AggressiveInlining)]
31 public T Get() => objects.Count > 0 ? objects.Pop() : objectGenerator();
32
33 // return an element to the pool
34 [MethodImpl(MethodImplOptions.AggressiveInlining)]
35 public void Return(T item) => objects.Push(item);
36
37 // count to see how many objects are in the pool. useful for tests.
38 public int Count => objects.Count;
39 }
