51 public override void OnRebuildObservers(NetworkIdentity identity, HashSet<NetworkConnectionToClient> newObservers)
52 {
53 // cache range and .transform because both call GetComponent.
54 int range = GetVisRange(identity);
55 Vector3 position = identity.transform.position;
56
57 // brute force distance check
58 // -> only player connections can be observers, so it's enough if we
59 // go through all connections instead of all spawned identities.
60 // -> compared to UNET's sphere cast checking, this one is orders of
61 // magnitude faster. if we have 10k monsters and run a sphere
62 // cast 10k times, we will see a noticeable lag even with physics
63 // layers. but checking to every connection is fast.
64 foreach (NetworkConnectionToClient conn in NetworkServer.connections.Values)
65 {
66 // authenticated and joined world with a player?
67 if (conn != null && conn.isAuthenticated && conn.identity != null)
68 {
69 // check distance
70 if (Vector3.Distance(conn.identity.transform.position, position) < range)
71 {
72 newObservers.Add(conn);
73 }
74 }
75 }
